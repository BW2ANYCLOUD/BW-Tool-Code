  TAB_T1 = 'BW Lineage'.
  TAB_T2 = 'FM Extractor'.
  TAB_T3 = 'Process Chains'.
  TAB_T4 = 'Reconciliation'.

AT SELECTION-SCREEN.
  CASE SSCRFIELDS-UCOMM.
    WHEN 'ONLI'.
      " Determine active tab by which primary input is filled.
      " Validate ONLY the tab that has data — never block other tabs.

      " Tab 1: only validate if Tab 1 inputs are being used
      IF ( R_PROV = 'X' AND P_INP IS NOT INITIAL )
      OR ( R_INFO = 'X' AND P_INF IS NOT INITIAL ).
        IF R_PROV = 'X' AND P_INP IS INITIAL.
          MESSAGE 'Tab1: Enter Provider name (P_INP).' TYPE 'E'.
        ENDIF.
        IF R_INFO = 'X' AND P_INF IS INITIAL.
          MESSAGE 'Tab1: Enter InfoArea (P_INF).' TYPE 'E'.
        ENDIF.
        IF P_PATH IS INITIAL.
          MESSAGE 'Tab1: Enter output folder (P_PATH).' TYPE 'E'.
        ENDIF.
      ENDIF.

      " Tab 2: only validate if FM name given
      IF P_FM IS NOT INITIAL AND P_FMDIR IS INITIAL.
        MESSAGE 'Tab2: Enter FM output folder (P_FMDIR).' TYPE 'E'.
      ENDIF.

      " Tab 3: only validate if chain name given
      IF P_PCPK IS NOT INITIAL AND P_PCDIR IS INITIAL.
        MESSAGE 'Tab3: Enter Process Chain output folder (P_PCDIR).' TYPE 'E'.
      ENDIF.

      " Tab 4: only validate if source table given
      IF P_RC_SRC IS NOT INITIAL AND P_RC_OUT IS INITIAL.
        MESSAGE 'Tab4: Enter Reconciliation output folder (P_RC_OUT).' TYPE 'E'.
      ENDIF.

      " Guard: at least one tab must have input
      IF P_INP IS INITIAL AND P_INF IS INITIAL
      AND P_FM IS INITIAL
      AND P_PCPK IS INITIAL
      AND P_RC_SRC IS INITIAL.
        MESSAGE 'Enter input on at least one tab before executing.' TYPE 'E'.
      ENDIF.

    WHEN OTHERS.
      " Tab switches — no validation
  ENDCASE.

AT SELECTION-SCREEN OUTPUT.
  LOOP AT SCREEN.
    IF R_INFO = 'X'.
      IF SCREEN-NAME = 'P_INP' OR SCREEN-NAME = '%_P_INP_%_APP_%-TEXT'.
        SCREEN-INPUT = 0. SCREEN-REQUIRED = 0. MODIFY SCREEN.
      ENDIF.
    ENDIF.
    IF R_PROV = 'X'.
      IF SCREEN-NAME = 'P_INF' OR SCREEN-NAME = '%_P_INF_%_APP_%-TEXT'.
        SCREEN-INPUT = 0. SCREEN-REQUIRED = 0. MODIFY SCREEN.
      ENDIF.
    ENDIF.
  ENDLOOP.

  IF SY-SUBRC = 0 AND LV_F4_FMDIR IS NOT INITIAL. P_FMDIR = LV_F4_FMDIR. ENDIF.

AT SELECTION-SCREEN ON VALUE-REQUEST FOR P_INF.
  DATA: LT_F4_RET TYPE STANDARD TABLE OF DDSHRETVAL WITH DEFAULT KEY,
        LT_F4_TAB TYPE STANDARD TABLE OF RSDCUBE    WITH DEFAULT KEY.

              DYNPNR = SY-DYNNR DYNPROFIELD = 'P_INF' VALUE_ORG = 'S'

INCLUDE ZBW_Tool_FORMS.

START-OF-SELECTION.

  IF SY-BATCH = 'X'.
    MESSAGE 'Run in dialog mode only (GUI download required).' TYPE 'E'.
  ENDIF.

  IF ( R_PROV = 'X' AND P_INP IS NOT INITIAL )
  OR ( R_INFO = 'X' AND P_INF IS NOT INITIAL ).

    PERFORM INIT_STOPWORDS.
    REFRESH GT_TRAN.
    SELECT * FROM RSTRAN INTO TABLE GT_TRAN WHERE OBJVERS = 'A'.

    IF R_INFO = 'X'.
      DATA: LT_IA_PROV TYPE TY_NAME_TAB,
            LV_IA_PROV TYPE CHAR30.
      REFRESH LT_IA_PROV.
      PERFORM COLLECT_INFOAREA_PROVIDERS USING P_INF CHANGING LT_IA_PROV.
      IF LT_IA_PROV IS INITIAL.
        MESSAGE 'No providers found under selected InfoArea.' TYPE 'W'.
      ENDIF.
      LOOP AT LT_IA_PROV INTO LV_IA_PROV.
        PERFORM RUN_FOR_PROVIDER USING LV_IA_PROV.
      ENDLOOP.
    ELSE.
      PERFORM RUN_FOR_PROVIDER USING P_INP.
    ENDIF.

    MESSAGE 'Tab 1: BW Lineage extraction completed.' TYPE 'S'.
  ENDIF.

  IF P_FM IS NOT INITIAL.
    IF P_MAXFM IS INITIAL OR P_MAXFM < 1. GV_MAXFM = 30. ELSE. GV_MAXFM = P_MAXFM. ENDIF.
    IF P_STDEP IS INITIAL OR P_STDEP < 1. GV_STD_DEPTH_MAX = 5. ELSE. GV_STD_DEPTH_MAX = P_STDEP. ENDIF.
    GV_STD_DEPTH = 0. GV_FM_COUNT = 0. GV_STD_SKIPPED_COUNT = '0'.
    CLEAR GV_LIMIT_REACHED.
    REFRESH: GT_TREE, GT_EDGES_TXT, GT_VISITED_FM, GT_MISSING, GT_DONE_PERF, GT_FORM_CACHE.
    PERFORM GET_FILE_SEPARATOR CHANGING GV_SEP.
    PERFORM NORMALIZE_FM_PATH USING P_FMDIR GV_SEP CHANGING GV_BASE_FOLDER.
    DATA: LV_FM_ROOT TYPE STRING.
    CONCATENATE GV_BASE_FOLDER GV_SEP P_FM INTO LV_FM_ROOT.
    PERFORM DIR_CREATE_SAFE USING LV_FM_ROOT.
    PERFORM PROCESS_FM USING P_FM 0.
    PERFORM DOWNLOAD_FILE_TO_FOLDER TABLES GT_TREE USING LV_FM_ROOT 'CALL_TREE'.
    PERFORM DOWNLOAD_FILE_TO_FOLDER TABLES GT_EDGES_TXT USING LV_FM_ROOT 'LINEAGE'.
    IF GT_MISSING IS NOT INITIAL.
      PERFORM DOWNLOAD_FILE_TO_FOLDER TABLES GT_MISSING USING LV_FM_ROOT 'MISSING_OBJECTS'.
    ENDIF.
    IF GV_STD_SKIPPED_COUNT > '0'.
      DATA: LV_SKIP_MSG TYPE STRING.
      CONCATENATE 'Standard FM depth limit hit. Skipped:' GV_STD_SKIPPED_COUNT
             INTO LV_SKIP_MSG SEPARATED BY SPACE.
      MESSAGE LV_SKIP_MSG TYPE 'W'.
    ENDIF.
    MESSAGE 'Tab 2: FM extraction completed.' TYPE 'S'.
  ENDIF.

  IF P_PCPK IS NOT INITIAL OR P_PCDIR IS NOT INITIAL.
    IF P_PCDIR IS INITIAL.
      MESSAGE 'Tab 3: P_PCDIR (output folder) is required.' TYPE 'E'.
    ENDIF.
    REFRESH: GT_PC_HDR, GT_PC_STEPS.
    IF P_PCPK IS NOT INITIAL.
      PERFORM COLLECT_PROCESS_CHAINS USING P_PCPK ''.
    ENDIF.
    PERFORM WRITE_PC_FILES USING P_PCDIR.
    MESSAGE 'Tab 3: Process Chain extraction completed.' TYPE 'S'.
  ENDIF.

  IF P_RC_SRC IS NOT INITIAL AND P_RC_TGT IS NOT INITIAL.
    IF P_RC_OUT IS INITIAL.
      MESSAGE 'Tab 4: P_RC_OUT (output folder) is required.' TYPE 'E'.
    ENDIF.
    REFRESH GT_RECON.
    PERFORM RUN_RECONCILIATION USING P_RC_SRC P_RC_TGT P_RC_KEY P_RC_OUT.
    MESSAGE 'Tab 4: Reconciliation completed.' TYPE 'S'.
  ENDIF.

FORM RUN_FOR_PROVIDER USING PV_PROV TYPE CHAR30.

  REFRESH: GT_FLOW, GT_SEEN, GT_DETAILS,
           GT_DEP_LOG, GT_DDL_TABS, GT_FM_QUEUE,
           GT_LOAD, GT_INACTIVE, GT_MD_IOBJ,
           GT_FLAG_DSO, GT_FLAG_CUBE, GT_SUMMARY.
  CLEAR: GV_CNT_TRAN, GV_CNT_TRAN_W_ROUT,
         GV_CNT_BRONZE, GV_CNT_SILVER, GV_CNT_GOLD,
         GV_ROOT_DIR, GV_FLOW_DIR, GV_DDL_DIR,
         GV_TRANS_DIR, GV_OVW_DIR, GV_MD_DIR, GV_FM_DIR.

  GV_CTX = PV_PROV.
  PERFORM PATH_INIT.
  IF GV_SEP IS INITIAL. PERFORM GET_FILE_SEPARATOR CHANGING GV_SEP. ENDIF.
  PERFORM START_FROM_INPUT USING GV_CTX.

  IF P_FLOW = 'X'. PERFORM WRITE_FLOW_CSV. ENDIF.
  IF P_JSON = 'X'.
    PERFORM WRITE_FLOW_JSON.
    PERFORM WRITE_FLOW_MERMAID.
    PERFORM WRITE_FLOW_MERMAID_GROUPED.
  ENDIF.
  IF P_TREE = 'X'. PERFORM WRITE_TREE_ASCII. ENDIF.

  IF P_MAP = 'X' OR P_ROUT = 'X'.
    PERFORM EXPORT_ALL_TRANSFORMATIONS.

    IF P_FM_IN = 'X' AND GT_FM_QUEUE IS NOT INITIAL.
      DATA: LV_INL_FMDIR TYPE STRING,
            LV_FM_OK     TYPE C,
            LV_INL_FM    TYPE RS38L_FNAM,
            LV_FM_ROOT_I TYPE STRING.
      CONCATENATE GV_ROOT_DIR 'Function_Modules' GV_SEP INTO LV_INL_FMDIR.
      PERFORM ENSURE_FRONTEND_DIR USING LV_INL_FMDIR CHANGING LV_FM_OK.
      GV_BASE_FOLDER = LV_INL_FMDIR.
      CLEAR GV_LIMIT_REACHED.
      GV_FM_COUNT = 0. GV_STD_DEPTH = 0.
      IF GV_MAXFM IS INITIAL OR GV_MAXFM < 1. GV_MAXFM = 30. ENDIF.
      IF GV_STD_DEPTH_MAX IS INITIAL. GV_STD_DEPTH_MAX = 5. ENDIF.

      DATA: LT_ALL_TREE     TYPE STANDARD TABLE OF STRING WITH EMPTY KEY,
            LT_ALL_LINEAGE  TYPE STANDARD TABLE OF STRING WITH EMPTY KEY,
            LV_FM_HDR       TYPE STRING.

      LOOP AT GT_FM_QUEUE INTO LV_INL_FM.
        REFRESH: GT_TREE, GT_EDGES_TXT, GT_VISITED_FM, GT_MISSING.
        GV_FM_COUNT = 0. CLEAR GV_LIMIT_REACHED.
        PERFORM PROCESS_FM USING LV_INL_FM 0.
        CONCATENATE GV_BASE_FOLDER GV_SEP LV_INL_FM INTO LV_FM_ROOT_I.
        PERFORM DOWNLOAD_FILE_TO_FOLDER TABLES GT_TREE USING LV_FM_ROOT_I 'CALL_TREE'.
        PERFORM DOWNLOAD_FILE_TO_FOLDER TABLES GT_EDGES_TXT USING LV_FM_ROOT_I 'LINEAGE'.
        IF GT_MISSING IS NOT INITIAL.
          PERFORM DOWNLOAD_FILE_TO_FOLDER TABLES GT_MISSING USING LV_FM_ROOT_I 'MISSING_OBJECTS'.
        ENDIF.
        CONCATENATE '=== FM: ' LV_INL_FM ' ===' INTO LV_FM_HDR.
        APPEND LV_FM_HDR TO LT_ALL_TREE.
        APPEND LINES OF GT_TREE TO LT_ALL_TREE.
        APPEND LV_FM_HDR TO LT_ALL_LINEAGE.
        APPEND LINES OF GT_EDGES_TXT TO LT_ALL_LINEAGE.
      ENDLOOP.

      IF LT_ALL_TREE IS NOT INITIAL.
        PERFORM DOWNLOAD_FILE_TO_FOLDER TABLES LT_ALL_TREE USING LV_INL_FMDIR 'FM_CALL_TREE_SUMMARY'.
      ENDIF.
      IF LT_ALL_LINEAGE IS NOT INITIAL.
        PERFORM DOWNLOAD_FILE_TO_FOLDER TABLES LT_ALL_LINEAGE USING LV_INL_FMDIR 'FM_LINEAGE_SUMMARY'.
      ENDIF.
      FREE: GT_FM_QUEUE, GT_TREE, GT_EDGES_TXT, GT_VISITED_FM, GT_MISSING.
    ENDIF.
  ENDIF.

  IF P_DDL = 'X'. PERFORM EXPORT_DDL_FILES. ENDIF.

  IF P_PC_IN = 'X'.
    DATA: LV_PC_DIR TYPE STRING,
          LV_PC_OK  TYPE C.
    CONCATENATE GV_ROOT_DIR 'Process_Chains' GV_SEP INTO LV_PC_DIR.
    PERFORM ENSURE_FRONTEND_DIR USING LV_PC_DIR CHANGING LV_PC_OK.
    REFRESH: GT_PC_HDR, GT_PC_STEPS.
    PERFORM COLLECT_PROCESS_CHAINS USING '' ''.  " collect all chains
    IF GT_PC_HDR IS NOT INITIAL.
      PERFORM WRITE_PC_FILES USING LV_PC_DIR.
    ENDIF.
  ENDIF.

  IF P_DDL IS INITIAL AND P_MD_FL = 'X'.
    PERFORM COLLECT_MD_IOBJ_FROM_FLOW.
  ENDIF.

  PERFORM CLEAN_DEP_LOG.
  PERFORM WRITE_ALL_TRANS_LOGS.
  PERFORM WRITE_GLOBAL_DEP_LOG.

  IF P_MD_FL = 'X'. PERFORM WRITE_MD_FLOWS. ENDIF.

  PERFORM BUILD_SUMMARY.
  PERFORM WRITE_SUMMARY_README.

  FREE: GT_DEP_LOG, GT_FLOW, GT_MD_IOBJ.

ENDFORM.

FORM COLLECT_MD_IOBJ_FROM_FLOW.
  DATA: LS_F      TYPE TY_FLOW,
        LV_DSO    TYPE RSDODSO-ODSOBJECT,
        LV_TABNAM TYPE DD02L-TABNAME,
        LT_DD03P  TYPE STANDARD TABLE OF DD03P WITH EMPTY KEY,
        LS_DD03P  TYPE DD03P,
        LV_IO     TYPE RSIOBJNM,
        LV_OFF    TYPE I.
  DATA: LT_DSOS TYPE SORTED TABLE OF RSDODSO-ODSOBJECT WITH UNIQUE KEY TABLE_LINE.
  REFRESH LT_DSOS.
  LOOP AT GT_FLOW INTO LS_F WHERE SOURCE_TYPE = 'DSO' OR TARGET_TYPE = 'DSO'.
    IF LS_F-SOURCE_TYPE = 'DSO' AND LS_F-SOURCE IS NOT INITIAL.
      LV_DSO = LS_F-SOURCE(30). INSERT LV_DSO INTO TABLE LT_DSOS.
    ENDIF.
    IF LS_F-TARGET_TYPE = 'DSO' AND LS_F-TARGET IS NOT INITIAL.
      LV_DSO = LS_F-TARGET(30). INSERT LV_DSO INTO TABLE LT_DSOS.
    ENDIF.
  ENDLOOP.
  LOOP AT LT_DSOS INTO LV_DSO.
    CONCATENATE '/BIC/A' LV_DSO '2' INTO LV_TABNAM.
    SELECT SINGLE TABNAME FROM DD02L INTO LV_TABNAM WHERE TABNAME = LV_TABNAM.
    IF SY-SUBRC <> 0.
      CONCATENATE '/BI0/A' LV_DSO '2' INTO LV_TABNAM.
      SELECT SINGLE TABNAME FROM DD02L INTO LV_TABNAM WHERE TABNAME = LV_TABNAM.
      IF SY-SUBRC <> 0. CONTINUE. ENDIF.
    ENDIF.
    REFRESH LT_DD03P.
    CALL FUNCTION 'DDIF_TABL_GET'
      EXPORTING NAME = LV_TABNAM TABLES DD03P_TAB = LT_DD03P EXCEPTIONS OTHERS = 1.
    IF SY-SUBRC <> 0. CONTINUE. ENDIF.
    LOOP AT LT_DD03P INTO LS_DD03P WHERE FIELDNAME CP 'SID_*'.
      LV_OFF = 4. LV_IO = LS_DD03P-FIELDNAME+LV_OFF.
      CONDENSE LV_IO NO-GAPS.
      IF LV_IO IS NOT INITIAL. INSERT LV_IO INTO TABLE GT_MD_IOBJ. ENDIF.
    ENDLOOP.
  ENDLOOP.
ENDFORM.
