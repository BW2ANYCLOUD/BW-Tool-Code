*---------------------------------------------------------------------*
* FORM GET_FILE_SEPARATOR
*---------------------------------------------------------------------*
FORM GET_FILE_SEPARATOR CHANGING P_SEP TYPE C.
  DATA: LV_SEP TYPE C.
  P_SEP = '\'.
  CLEAR LV_SEP.
  CALL METHOD CL_GUI_FRONTEND_SERVICES=>GET_FILE_SEPARATOR
    CHANGING
      FILE_SEPARATOR       = LV_SEP
    EXCEPTIONS
      NOT_SUPPORTED_BY_GUI = 1
      ERROR_NO_GUI         = 2
      CNTL_ERROR           = 3
      OTHERS               = 4.
  IF SY-SUBRC = 0 AND LV_SEP IS NOT INITIAL.
    " Old-kernel safe: read single char via offset
    P_SEP = LV_SEP+0(1).
  ENDIF.
ENDFORM.

*---------------------------------------------------------------------*
* FORM NORMALIZE_FM_PATH
* Trims trailing separators from a path string (Bug 5: output STRING)
*---------------------------------------------------------------------*
*FORM NORMALIZE_FM_PATH USING    P_IN  TYPE STRING
*                                P_SEP TYPE C
*                       CHANGING P_OUT TYPE STRING.
*  DATA: LV_LEN TYPE I,
*        LV_OFF TYPE I,
*        LV_NEW TYPE I.
*
*  P_OUT = P_IN.
*  IF P_OUT IS INITIAL.
*    MESSAGE 'FM output path (P_FMDIR) is empty.' TYPE 'E'.
*  ENDIF.
*
*  WHILE P_OUT IS NOT INITIAL.
*    DESCRIBE FIELD P_OUT LENGTH LV_LEN IN CHARACTER MODE.
*    IF LV_LEN <= 0. EXIT. ENDIF.
*    LV_OFF = LV_LEN - 1.
*    IF P_OUT+LV_OFF(1) <> P_SEP. EXIT. ENDIF.
*    LV_NEW = LV_LEN - 1.
*    P_OUT  = P_OUT(LV_NEW).
*  ENDWHILE.
*ENDFORM.

FORM NORMALIZE_FM_PATH USING    P_IN  TYPE STRING
                                 P_SEP TYPE C
                        CHANGING P_OUT TYPE STRING.

  DATA: LV_LEN TYPE I.

  P_OUT = P_IN.
  IF P_OUT IS INITIAL.
    MESSAGE 'FM output path (P_FMDIR) is empty.' TYPE 'E'.
  ENDIF.

  WHILE P_OUT IS NOT INITIAL.
    LV_LEN = STRLEN( P_OUT ).
    IF LV_LEN <= 0. EXIT. ENDIF.

    " Check last character using SUBSTRING on a STRING
    IF SUBSTRING( VAL = P_OUT OFF = LV_LEN - 1 LEN = 1 ) <> P_SEP.
      EXIT.
    ENDIF.

    " Trim the last character
    P_OUT = SUBSTRING( VAL = P_OUT OFF = 0 LEN = LV_LEN - 1 ).
  ENDWHILE.

ENDFORM.

                  CHANGING P_OUT   TYPE STRING.

  CLEAR P_OUT.
  LV_I = 0.

*---------------------------------------------------------------------*
* FORM GET_FM_INCLUDE
* Resolves FG area, main program, and specific include for an FM.
*---------------------------------------------------------------------*
FORM GET_FM_INCLUDE USING    P_FM      TYPE RS38L_FNAM
                    CHANGING C_AREA    TYPE RS38L_AREA
                             C_MAIN    TYPE PROGNAME
                             C_INCLUDE TYPE PROGNAME.
  DATA: LT_FUNCTAB TYPE STANDARD TABLE OF RS38L_INCL WITH DEFAULT KEY,
        LS_FUNCTAB TYPE RS38L_INCL.

  CLEAR: C_AREA, C_MAIN, C_INCLUDE.

  " Primary: ENLFDIR (most reliable)
  SELECT SINGLE AREA INTO C_AREA FROM ENLFDIR WHERE FUNCNAME = P_FM.

  IF SY-SUBRC <> 0 OR C_AREA IS INITIAL.
    " Fallback: TFDIR PNAME field (SAPL<area>)
    SELECT SINGLE PNAME INTO C_MAIN FROM TFDIR WHERE FUNCNAME = P_FM.
    IF SY-SUBRC = 0 AND C_MAIN CP 'SAPL*'.
      C_AREA = C_MAIN+4.   " strip 'SAPL' prefix
    ENDIF.
  ENDIF.

  IF C_AREA IS INITIAL. RETURN. ENDIF.

  CONCATENATE 'SAPL' C_AREA INTO C_MAIN.

  REFRESH LT_FUNCTAB.
  CALL FUNCTION 'RS_FUNCTION_POOL_CONTENTS'
    EXPORTING
      FUNCTION_POOL           = C_AREA
    TABLES
      FUNCTAB                 = LT_FUNCTAB
    EXCEPTIONS
      FUNCTION_POOL_NOT_FOUND = 1
      OTHERS                  = 2.
  IF SY-SUBRC <> 0. RETURN. ENDIF.

  READ TABLE LT_FUNCTAB INTO LS_FUNCTAB WITH KEY FUNCNAME = P_FM.
  IF SY-SUBRC = 0.
    C_INCLUDE = LS_FUNCTAB-INCLUDE.
  ENDIF.
ENDFORM.

*---------------------------------------------------------------------*
* FORM IS_STANDARD_FM
* Returns ABAP_TRUE for SAP-namespace / system FMs.
*---------------------------------------------------------------------*
FORM IS_STANDARD_FM USING    P_FM  TYPE RS38L_FNAM
                    CHANGING P_STD TYPE ABAP_BOOL.
  DATA: LV_UP TYPE STRING.

  LV_UP = P_FM.
  TRANSLATE LV_UP TO UPPER CASE.

  IF LV_UP CP 'RS*'   OR LV_UP CP 'RR*'   OR LV_UP CP 'RS_*'
  OR LV_UP CP 'DD*'   OR LV_UP CP 'TR*'
  OR LV_UP CP 'ENQUEUE*' OR LV_UP CP 'DEQUEUE*'
  OR LV_UP CP 'SUSR*' OR LV_UP CP 'SCWB*'
  OR LV_UP CP 'SAP*'  OR LV_UP CP 'F4IF*'
  OR LV_UP CP 'GUI_*' OR LV_UP CP 'CL_GUI*'.
    P_STD = ABAP_TRUE.
  ELSE.
    P_STD = ABAP_FALSE.
  ENDIF.
ENDFORM.

*---------------------------------------------------------------------*
* FORM ASSEMBLE_STATEMENTS
* Joins continuation lines into complete ABAP statements (up to '.').
* Bug 3 fix: all work vars declared at top, not inside loop.
*---------------------------------------------------------------------*
FORM ASSEMBLE_STATEMENTS USING    PT_SRC TYPE TY_STRINGTAB
                         CHANGING CT_STM TYPE TY_STRINGTAB.
  " All loop-local vars declared here (Bug 3 compliance)
  DATA: LV_LINE  TYPE STRING,
        LV_WORK  TYPE STRING,
        LV_STMT  TYPE STRING,
        LV_DUMMY TYPE STRING.

  REFRESH CT_STM.
  CLEAR LV_STMT.

  LOOP AT PT_SRC INTO LV_LINE.
    LV_WORK = LV_LINE.
    SHIFT LV_WORK LEFT DELETING LEADING SPACE.

    " Skip blank, full-line comment, inline comment lines
    IF LV_WORK IS INITIAL
    OR LV_WORK+0(1) = '*'
    OR LV_WORK+0(1) = '"'.
      CONTINUE.
    ENDIF.

    " Strip inline comment
    SPLIT LV_WORK AT '"' INTO LV_WORK LV_DUMMY.
    IF LV_WORK IS INITIAL. CONTINUE. ENDIF.

    IF LV_STMT IS INITIAL.
      LV_STMT = LV_WORK.
    ELSE.
      CONCATENATE LV_STMT ' ' LV_WORK INTO LV_STMT.
    ENDIF.

    IF LV_STMT CS '.'.
      APPEND LV_STMT TO CT_STM.
      CLEAR LV_STMT.
    ENDIF.
  ENDLOOP.

  " Flush incomplete final statement (no trailing '.')
  IF LV_STMT IS NOT INITIAL.
    APPEND LV_STMT TO CT_STM.
  ENDIF.
ENDFORM.

*---------------------------------------------------------------------*
* FORM GET_FORM_BLOCK
* Extracts the FORM...ENDFORM block for a given FORM name from an include.
*---------------------------------------------------------------------*
FORM GET_FORM_BLOCK USING    P_INCLUDE TYPE PROGNAME
                             P_FORM    TYPE STRING
                    CHANGING CT_BLOCK  TYPE TY_STRINGTAB
                             C_FOUND   TYPE C.
  " Bug 3: all vars declared at top
  DATA: LT_SRC     TYPE TY_STRINGTAB,
        LV_LINE    TYPE STRING,
        LV_WORK    TYPE STRING,
        LV_U       TYPE STRING,
        LV_DUMMY   TYPE STRING,
        LV_INBLOCK TYPE C,
        LV_FORM_U  TYPE STRING,
        LV_REGEX   TYPE STRING.

  REFRESH CT_BLOCK.
  C_FOUND    = SPACE.
  LV_INBLOCK = SPACE.

  READ REPORT P_INCLUDE INTO LT_SRC.
  IF SY-SUBRC <> 0. RETURN. ENDIF.

  LV_FORM_U = P_FORM.
  TRANSLATE LV_FORM_U TO UPPER CASE.
  CONCATENATE '^\s*FORM\s+' LV_FORM_U '\b' INTO LV_REGEX.

  LOOP AT LT_SRC INTO LV_LINE.
    LV_WORK = LV_LINE.
    SHIFT LV_WORK LEFT DELETING LEADING SPACE.

    IF LV_WORK IS INITIAL
    OR LV_WORK+0(1) = '*'
    OR LV_WORK+0(1) = '"'.
      IF LV_INBLOCK = 'X'. APPEND LV_LINE TO CT_BLOCK. ENDIF.
      CONTINUE.
    ENDIF.

    SPLIT LV_WORK AT '"' INTO LV_WORK LV_DUMMY.
    LV_U = LV_WORK.
    TRANSLATE LV_U TO UPPER CASE.

    IF LV_INBLOCK = SPACE.
      FIND REGEX LV_REGEX IN LV_U.
      IF SY-SUBRC = 0.
        LV_INBLOCK = 'X'.
        C_FOUND    = 'X'.
      ELSE.
        CONTINUE.
      ENDIF.
    ENDIF.

    APPEND LV_LINE TO CT_BLOCK.

    " Detect ENDFORM (with or without trailing space before '.')
    IF LV_U CS 'ENDFORM'.
      EXIT.
    ENDIF.
  ENDLOOP.
ENDFORM.

*---------------------------------------------------------------------*
* FORM EXTRACT_DEPENDENCIES
* Scans assembled statements for CALL FUNCTION and PERFORM references.
*---------------------------------------------------------------------*
FORM EXTRACT_DEPENDENCIES USING    P_PARENT TYPE RS38L_FNAM
                                   PT_SRC   TYPE TY_STRINGTAB
                          CHANGING CT_EDGES TYPE TY_EDGE_TAB
                                   CT_PERF  TYPE TY_PERF_TAB.
  DATA: LT_STM  TYPE TY_STRINGTAB,
        LV_STMT TYPE STRING,
        LV_U    TYPE STRING,
        LV_FM   TYPE STRING,
        LV_A    TYPE STRING,
        LV_B    TYPE STRING,
        LS_EDGE TYPE TY_EDGE,
        LS_PERF TYPE TY_PERF_REF.

  REFRESH CT_EDGES.
  REFRESH CT_PERF.

  PERFORM ASSEMBLE_STATEMENTS USING PT_SRC CHANGING LT_STM.

  LOOP AT LT_STM INTO LV_STMT.
    LV_U = LV_STMT.
    TRANSLATE LV_U TO UPPER CASE.

    " --- CALL FUNCTION ---
    IF LV_U CS 'CALL FUNCTION'.
      CLEAR LV_FM.
      FIND REGEX 'CALL\s+FUNCTION\s+''([^'']+)''' IN LV_U SUBMATCHES LV_FM.
      IF SY-SUBRC = 0 AND LV_FM IS NOT INITIAL.
        CLEAR LS_EDGE.
        LS_EDGE-PARENT = P_PARENT.
        LS_EDGE-KIND   = 'FM'.
        LS_EDGE-CHILD  = LV_FM.
        APPEND LS_EDGE TO CT_EDGES.
      ENDIF.
    ENDIF.

    " --- PERFORM <form> IN PROGRAM <prog> ---
    CLEAR: LV_A, LV_B.
    FIND REGEX 'PERFORM\s+(\w+)\s+IN\s+PROGRAM\s+(\w+)'
      IN LV_U SUBMATCHES LV_A LV_B.
    IF SY-SUBRC = 0 AND LV_A IS NOT INITIAL AND LV_B IS NOT INITIAL.
      CLEAR LS_PERF.
      LS_PERF-FORM = LV_A.
      LS_PERF-PROG = LV_B.
      APPEND LS_PERF TO CT_PERF.
      CONTINUE.
    ENDIF.

    " --- PERFORM <form> ---
    CLEAR LV_A.
    FIND REGEX 'PERFORM\s+(\w+)' IN LV_U SUBMATCHES LV_A.
    IF SY-SUBRC = 0 AND LV_A IS NOT INITIAL AND LV_A <> 'IN'.
      CLEAR LS_PERF.
      LS_PERF-FORM = LV_A.
      CLEAR LS_PERF-PROG.
      APPEND LS_PERF TO CT_PERF.
    ENDIF.
  ENDLOOP.

  SORT CT_PERF BY FORM PROG.
  DELETE ADJACENT DUPLICATES FROM CT_PERF COMPARING FORM PROG.
ENDFORM.

*---------------------------------------------------------------------*
* FORM RESOLVE_PERFORMS
* Resolves PERFORM references to FM edges by scanning FORM blocks.
* Safety cap: 8000 iterations.
*---------------------------------------------------------------------*
FORM RESOLVE_PERFORMS USING    P_PARENT TYPE RS38L_FNAM
                               P_MAIN   TYPE PROGNAME
                               PT_PERF  TYPE TY_PERF_TAB
                      CHANGING CT_EDGES TYPE TY_EDGE_TAB.
  DATA: LT_TODO  TYPE TY_PERF_TAB,
        LS_P     TYPE TY_PERF_REF,
        LS_DONE  TYPE TY_PERF_REF,
        LV_PROG  TYPE PROGNAME,
        LV_INC   TYPE PROGNAME,
        LT_INC   TYPE STANDARD TABLE OF PROGNAME WITH DEFAULT KEY,
        LT_BLOCK TYPE TY_STRINGTAB,
        LV_FOUND TYPE C,
        LT_E2    TYPE TY_EDGE_TAB,
        LT_P2    TYPE TY_PERF_TAB,
        LS_E2    TYPE TY_EDGE,
        LS_P2    TYPE TY_PERF_REF,
        LS_CACHE TYPE TY_FORM_CACHE,
        LV_PROC  TYPE I,
        LV_MSG   TYPE STRING.

  LT_TODO = PT_PERF.
  LV_PROC = 0.

  WHILE LT_TODO IS NOT INITIAL.
    LV_PROC = LV_PROC + 1.
    IF LV_PROC > 8000.
      CONCATENATE 'PERFORM resolution safety cap (8000) reached for FM:'
                  P_PARENT INTO LV_MSG SEPARATED BY SPACE.
      APPEND LV_MSG TO GT_MISSING.
      EXIT.
    ENDIF.

    READ TABLE LT_TODO INTO LS_P INDEX 1.
    DELETE LT_TODO INDEX 1.

    IF LS_P-PROG IS INITIAL.
      LV_PROG = P_MAIN.
    ELSE.
      LV_PROG = LS_P-PROG.
    ENDIF.

    " Visited guard (prog + form pair)
    READ TABLE GT_DONE_PERF INTO LS_DONE
      WITH KEY FORM = LS_P-FORM PROG = LV_PROG.
    IF SY-SUBRC = 0. CONTINUE. ENDIF.

    CLEAR LS_DONE.
    LS_DONE-FORM = LS_P-FORM.
    LS_DONE-PROG = LV_PROG.
    APPEND LS_DONE TO GT_DONE_PERF.

    " Cache lookup
    CLEAR LV_INC.
    READ TABLE GT_FORM_CACHE INTO LS_CACHE
      WITH KEY PROG = LV_PROG FORM = LS_P-FORM.
    IF SY-SUBRC = 0 AND LS_CACHE-INCLUDE IS NOT INITIAL.
      LV_INC = LS_CACHE-INCLUDE.
    ENDIF.

    " Search includes for the FORM block
    IF LV_INC IS INITIAL.
      REFRESH LT_INC.
      CALL FUNCTION 'RS_GET_ALL_INCLUDES'
        EXPORTING
          PROGRAM    = LV_PROG
        TABLES
          INCLUDETAB = LT_INC
        EXCEPTIONS
          OTHERS     = 1.
      IF SY-SUBRC <> 0. CONTINUE. ENDIF.

      LOOP AT LT_INC INTO LV_INC.
        REFRESH LT_BLOCK.
        LV_FOUND = SPACE.
        PERFORM GET_FORM_BLOCK USING LV_INC LS_P-FORM
                               CHANGING LT_BLOCK LV_FOUND.
        IF LV_FOUND = 'X'. EXIT. ENDIF.
        CLEAR LV_INC.
      ENDLOOP.

      " Write to cache regardless (prevents re-search)
      CLEAR LS_CACHE.
      LS_CACHE-PROG    = LV_PROG.
      LS_CACHE-FORM    = LS_P-FORM.
      LS_CACHE-INCLUDE = LV_INC.
      APPEND LS_CACHE TO GT_FORM_CACHE.
    ENDIF.

    IF LV_INC IS INITIAL. CONTINUE. ENDIF.

    " Load block if not already loaded by the include-search loop
    IF LT_BLOCK IS INITIAL.
      REFRESH LT_BLOCK.
      LV_FOUND = SPACE.
      PERFORM GET_FORM_BLOCK USING LV_INC LS_P-FORM
                             CHANGING LT_BLOCK LV_FOUND.
      IF LV_FOUND <> 'X'. CONTINUE. ENDIF.
    ENDIF.

    " Extract deps from this FORM block
    REFRESH: LT_E2, LT_P2.
    PERFORM EXTRACT_DEPENDENCIES USING P_PARENT LT_BLOCK
                                 CHANGING LT_E2 LT_P2.
    APPEND LINES OF LT_E2 TO CT_EDGES.

    LOOP AT LT_P2 INTO LS_P2.
      APPEND LS_P2 TO LT_TODO.
    ENDLOOP.
  ENDWHILE.
ENDFORM.

*---------------------------------------------------------------------*
* FORM DUMP_FG_INCLUDES_FILTERED
* Downloads TOP, UXX, main, and FORM-bearing includes for a FG.
* Bug 3 fix: LV_HASFORM, LV_UPPER, LV_LINE declared at FORM top.
*---------------------------------------------------------------------*
FORM DUMP_FG_INCLUDES_FILTERED USING P_FOLDER TYPE STRING
                                     P_MAIN   TYPE PROGNAME.
  " Bug 3: all loop-local vars declared here
  DATA: LT_INC     TYPE STANDARD TABLE OF PROGNAME WITH DEFAULT KEY,
        LV_INC     TYPE PROGNAME,
        LT_SRC     TYPE TY_STRINGTAB,
        LV_LINE    TYPE STRING,
        LV_UPPER   TYPE STRING,
        LV_HASFORM TYPE ABAP_BOOL,
        LV_TRIM    TYPE STRING.

  REFRESH LT_INC.
  CALL FUNCTION 'RS_GET_ALL_INCLUDES'
    EXPORTING
      PROGRAM    = P_MAIN
    TABLES
      INCLUDETAB = LT_INC
    EXCEPTIONS
      OTHERS     = 1.

  LOOP AT LT_INC INTO LV_INC.
    LV_UPPER = LV_INC.
    TRANSLATE LV_UPPER TO UPPER CASE.
    REFRESH LT_SRC.

    " 1. The main program itself
    IF LV_INC = P_MAIN.
      READ REPORT LV_INC INTO LT_SRC.
      IF SY-SUBRC = 0.
        PERFORM DOWNLOAD_FILE_TO_FOLDER TABLES LT_SRC USING P_FOLDER LV_INC.
      ENDIF.
      CONTINUE.
    ENDIF.

    " 2. TOP include (L<AREA>TOP)
    IF LV_UPPER CP 'L*TOP'.
      READ REPORT LV_INC INTO LT_SRC.
      IF SY-SUBRC = 0.
        PERFORM DOWNLOAD_FILE_TO_FOLDER TABLES LT_SRC USING P_FOLDER LV_INC.
      ENDIF.
      CONTINUE.
    ENDIF.

    " 3. UXX include (L<AREA>UXX)
    IF LV_UPPER CP 'L*UXX'.
      READ REPORT LV_INC INTO LT_SRC.
      IF SY-SUBRC = 0.
        PERFORM DOWNLOAD_FILE_TO_FOLDER TABLES LT_SRC USING P_FOLDER LV_INC.
      ENDIF.
      CONTINUE.
    ENDIF.

    " 4. Any include containing at least one FORM block
    READ REPORT LV_INC INTO LT_SRC.
    IF SY-SUBRC <> 0. CONTINUE. ENDIF.

    LV_HASFORM = ABAP_FALSE.
    LOOP AT LT_SRC INTO LV_LINE.
      LV_TRIM = LV_LINE.
      SHIFT LV_TRIM LEFT DELETING LEADING SPACE.
      TRANSLATE LV_TRIM TO UPPER CASE.
      IF LV_TRIM CP 'FORM *'.
        LV_HASFORM = ABAP_TRUE.
        EXIT.
      ENDIF.
    ENDLOOP.

    IF LV_HASFORM = ABAP_TRUE.
      PERFORM DOWNLOAD_FILE_TO_FOLDER TABLES LT_SRC USING P_FOLDER LV_INC.
    ENDIF.
  ENDLOOP.
ENDFORM.

*---------------------------------------------------------------------*
* FORM BUILD_UXX_OVERVIEW
* Synthesises a UXX include map (include → FM names).
*---------------------------------------------------------------------*
FORM BUILD_UXX_OVERVIEW USING P_AREA   TYPE RS38L_AREA
                              P_FOLDER TYPE STRING.
  DATA: LT_FUNCTAB   TYPE STANDARD TABLE OF RS38L_INCL WITH DEFAULT KEY,
        LS_FUNCTAB   TYPE RS38L_INCL,
        LT_OUT       TYPE TY_STRINGTAB,
        LV_LINE      TYPE STRING,
        LV_UXX       TYPE PROGNAME,
        LV_SUBRC_TXT TYPE C LENGTH 20.

  REFRESH: LT_FUNCTAB, LT_OUT.

  CALL FUNCTION 'RS_FUNCTION_POOL_CONTENTS'
    EXPORTING
      FUNCTION_POOL           = P_AREA
    TABLES
      FUNCTAB                 = LT_FUNCTAB
    EXCEPTIONS
      FUNCTION_POOL_NOT_FOUND = 1
      OTHERS                  = 2.

  IF SY-SUBRC <> 0 OR LT_FUNCTAB IS INITIAL.
    WRITE SY-SUBRC TO LV_SUBRC_TXT.
    CONCATENATE 'RS_FUNCTION_POOL_CONTENTS failed for' P_AREA
                'subrc=' LV_SUBRC_TXT INTO LV_LINE SEPARATED BY SPACE.
    APPEND LV_LINE TO GT_MISSING.
    RETURN.
  ENDIF.

  SORT LT_FUNCTAB BY INCLUDE FUNCNAME.
  CONCATENATE 'L' P_AREA 'UXX' INTO LV_UXX.

  APPEND '* Synthetic UXX overview — ZBW_FLOW_DETAIL_EXTRACTOR_4' TO LT_OUT.

*---------------------------------------------------------------------*
* FORM PROCESS_FM  – main recursive FM processor
* All inline DATA() removed; Bug 3 fix applied (vars at FORM top).
*---------------------------------------------------------------------*
FORM PROCESS_FM USING P_FM    TYPE RS38L_FNAM
                      P_LEVEL TYPE I.

  " All locals declared at top (Bug 3 compliance; no inline DATA)
  DATA: LV_AREA     TYPE RS38L_AREA,
        LV_MAIN     TYPE PROGNAME,
        LV_INC      TYPE PROGNAME,
        LV_FOLDER   TYPE STRING,        " Bug 5: STRING not CHAR200
        LT_SRC      TYPE TY_STRINGTAB,
        LT_SRC_FM   TYPE TY_STRINGTAB,
        LT_EDGES    TYPE TY_EDGE_TAB,
        LT_DEPOUT   TYPE TY_STRINGTAB,
        LT_PERF     TYPE TY_PERF_TAB,
        LV_INDENT   TYPE STRING,
        LV_LINE     TYPE STRING,
        LV_NEXT     TYPE I,
        LV_CNT_TXT  TYPE C LENGTH 20,
        LV_MAX_TXT  TYPE C LENGTH 20,
        LV_STD      TYPE ABAP_BOOL,
        LV_STD_HERE TYPE ABAP_BOOL,
        LS_E        TYPE TY_EDGE,
        LS_R        TYPE TY_EDGE.

  "--- Global limit guard
  IF GV_LIMIT_REACHED = 'X' OR GV_FM_COUNT >= GV_MAXFM.
    IF GV_LIMIT_REACHED <> 'X'.
      GV_LIMIT_REACHED = 'X'.
      WRITE GV_FM_COUNT TO LV_CNT_TXT.
      WRITE GV_MAXFM    TO LV_MAX_TXT.
      CONCATENATE '== FM limit reached: processed' LV_CNT_TXT
                  'of' LV_MAX_TXT '=='
             INTO LV_LINE SEPARATED BY SPACE.
      APPEND LV_LINE TO GT_TREE.
      APPEND LV_LINE TO GT_MISSING.
    ENDIF.
    PERFORM BUILD_INDENT USING P_LEVEL CHANGING LV_INDENT.
    CONCATENATE LV_INDENT P_FM '(skipped - limit)'
           INTO LV_LINE SEPARATED BY SPACE.
    APPEND LV_LINE TO GT_TREE.
    RETURN.
  ENDIF.

  "--- Visited guard
  READ TABLE GT_VISITED_FM WITH KEY TABLE_LINE = P_FM
    TRANSPORTING NO FIELDS.
  IF SY-SUBRC = 0.
    PERFORM BUILD_INDENT USING P_LEVEL CHANGING LV_INDENT.
    CONCATENATE LV_INDENT P_FM '(already processed)'
           INTO LV_LINE SEPARATED BY SPACE.
    APPEND LV_LINE TO GT_TREE.
    RETURN.
  ENDIF.

  "--- Mark visited
  APPEND P_FM TO GT_VISITED_FM.
  GV_FM_COUNT = GV_FM_COUNT + 1.

  "--- Locate FM include
  PERFORM GET_FM_INCLUDE USING P_FM
                         CHANGING LV_AREA LV_MAIN LV_INC.
  IF LV_INC IS INITIAL.
    CONCATENATE 'FM include not found for:' P_FM
           INTO LV_LINE SEPARATED BY SPACE.
    APPEND LV_LINE TO GT_MISSING.
    RETURN.
  ENDIF.

  "--- Create per-FM output folder
  CONCATENATE GV_BASE_FOLDER GV_SEP P_FM INTO LV_FOLDER.
  PERFORM DIR_CREATE_SAFE USING LV_FOLDER.

  PERFORM BUILD_INDENT USING P_LEVEL CHANGING LV_INDENT.
  CONCATENATE LV_INDENT P_FM INTO LV_LINE SEPARATED BY SPACE.
  APPEND LV_LINE TO GT_TREE.

  "--- Dump FG includes (skip heavy dump for nested standard FMs)
  PERFORM IS_STANDARD_FM USING P_FM CHANGING LV_STD_HERE.

  IF LV_STD_HERE = ABAP_TRUE AND GV_STD_DEPTH > 0.
    " Skip FG include dump for nested standard FMs — reduces noise
  ELSE.
    PERFORM DUMP_FG_INCLUDES_FILTERED USING LV_FOLDER LV_MAIN.
    IF LV_AREA IS NOT INITIAL.
      PERFORM BUILD_UXX_OVERVIEW USING LV_AREA LV_FOLDER.
    ENDIF.
  ENDIF.

  "--- Download the actual FM include source
  REFRESH LT_SRC_FM.
  READ REPORT LV_INC INTO LT_SRC_FM.
  IF SY-SUBRC = 0.
    PERFORM DOWNLOAD_FILE_TO_FOLDER TABLES LT_SRC_FM USING LV_FOLDER LV_INC.
  ENDIF.

  "--- Extract dependencies
  REFRESH LT_SRC.
  READ REPORT LV_INC INTO LT_SRC.

  IF SY-SUBRC <> 0.
    CONCATENATE 'MISSING FM INCLUDE:' LV_INC 'for FM:' P_FM
           INTO LV_LINE SEPARATED BY SPACE.
    APPEND LV_LINE TO GT_MISSING.
  ELSE.
    REFRESH: LT_EDGES, LT_PERF.
    PERFORM EXTRACT_DEPENDENCIES USING P_FM LT_SRC
                                 CHANGING LT_EDGES LT_PERF.

    IF LT_PERF IS NOT INITIAL.
      PERFORM RESOLVE_PERFORMS USING P_FM LV_MAIN LT_PERF
                               CHANGING LT_EDGES.
    ENDIF.

    SORT LT_EDGES BY PARENT KIND CHILD.
    DELETE ADJACENT DUPLICATES FROM LT_EDGES COMPARING PARENT KIND CHILD.

    "--- Write dependency summary file
    REFRESH LT_DEPOUT.
    APPEND 'DEPENDENCIES (CALL FUNCTION + PERFORM scan):' TO LT_DEPOUT.

    LOOP AT LT_EDGES INTO LS_E.
      CONCATENATE LS_E-KIND ':' LS_E-CHILD INTO LV_LINE SEPARATED BY SPACE.
      APPEND LV_LINE TO LT_DEPOUT.

      CONCATENATE LS_E-PARENT '->' LS_E-CHILD '[' LS_E-KIND ']'
             INTO LV_LINE SEPARATED BY SPACE.
      APPEND LV_LINE TO GT_EDGES_TXT.
    ENDLOOP.

    PERFORM DOWNLOAD_FILE_TO_FOLDER TABLES LT_DEPOUT USING LV_FOLDER 'DEPENDENCIES'.

    "--- Recurse into child FM calls
    LV_NEXT = P_LEVEL + 1.

    LOOP AT LT_EDGES INTO LS_R WHERE KIND = 'FM'.

      PERFORM IS_STANDARD_FM USING LS_R-CHILD CHANGING LV_STD.

      IF LV_STD = ABAP_TRUE.
        " Standard FM: honour depth limit
        IF GV_STD_DEPTH >= GV_STD_DEPTH_MAX.
          PERFORM APPEND_SKIPPED_STD_EDGE USING P_FM LS_R-CHILD.
          CONTINUE.
        ENDIF.
        GV_STD_DEPTH = GV_STD_DEPTH + 1.
        PERFORM PROCESS_FM USING LS_R-CHILD LV_NEXT.
        GV_STD_DEPTH = GV_STD_DEPTH - 1.
        CONTINUE.
      ENDIF.

      " Global limit check before recursing
      IF GV_LIMIT_REACHED = 'X' OR GV_FM_COUNT >= GV_MAXFM.
        PERFORM BUILD_INDENT USING LV_NEXT CHANGING LV_INDENT.
        CONCATENATE LV_INDENT LS_R-CHILD '(skipped - FM limit)'
               INTO LV_LINE SEPARATED BY SPACE.
        APPEND LV_LINE TO GT_TREE.
        CONTINUE.
      ENDIF.

      PERFORM PROCESS_FM USING LS_R-CHILD LV_NEXT.
    ENDLOOP.

  ENDIF.

  "--- FM interface file
  PERFORM DUMP_INTERFACE USING P_FM LV_FOLDER.

ENDFORM.                    " PROCESS_FM

*---------------------------------------------------------------------*
* FORM SANITIZE_FOR_FILE  – replaces non-alphanum chars with '_'
*---------------------------------------------------------------------*
FORM SANITIZE_FOR_FILE USING    PV_IN  TYPE CSEQUENCE
                       CHANGING PV_OUT TYPE STRING.
  DATA: LV_IN   TYPE STRING,
        LV_OUT  TYPE STRING,
        LV_C    TYPE C LENGTH 1,
        LV_PREV TYPE C LENGTH 1,
        LV_I    TYPE I,
        LV_LEN  TYPE I.

  LV_IN  = PV_IN.
  LV_LEN = STRLEN( LV_IN ).
  CLEAR: LV_OUT, LV_PREV.

  LV_I = 0.
  WHILE LV_I < LV_LEN.
    LV_C = LV_IN+LV_I(1).

    IF ( LV_C >= 'A' AND LV_C <= 'Z' )
    OR ( LV_C >= 'a' AND LV_C <= 'z' )
    OR ( LV_C >= '0' AND LV_C <= '9' )
    OR   LV_C = '_' OR LV_C = '-'.
      " keep
    ELSE.
      LV_C = '_'.
    ENDIF.

    " collapse consecutive underscores
    IF LV_C = '_' AND LV_PREV = '_'.
      LV_I = LV_I + 1.
      CONTINUE.
    ENDIF.

    CONCATENATE LV_OUT LV_C INTO LV_OUT IN CHARACTER MODE.
    LV_PREV = LV_C.
    LV_I = LV_I + 1.
  ENDWHILE.

  IF LV_OUT IS INITIAL. LV_OUT = 'file'. ENDIF.
  PV_OUT = LV_OUT.
ENDFORM.

*---------------------------------------------------------------------*
* FORM COLLECT_MAPPING
* Reads RSTRANFIELD + RSTRANRULE → builds enriched mapping rows.
* Uses DEFINE _FLUSH_GROUP macro for group-break logic (old-kernel safe).
*---------------------------------------------------------------------*
FORM COLLECT_MAPPING USING    PV_TRANID TYPE RSTRAN-TRANID
                     CHANGING PT_OUT    TYPE TY_MAP_TAB.

  DATA: LT_FIELDS  TYPE STANDARD TABLE OF TY_FIELD    WITH DEFAULT KEY,
        LT_RULES   TYPE STANDARD TABLE OF TY_RULE_HDR WITH DEFAULT KEY,
        LS_FIELD   TYPE TY_FIELD,
        LS_RULEH   TYPE TY_RULE_HDR,
        LT_SOURCES TYPE STANDARD TABLE OF STRING       WITH EMPTY KEY,
        LT_TARGETS TYPE STANDARD TABLE OF STRING       WITH EMPTY KEY,
        LV_RULE    TYPE STRING,
        LS_OUT     TYPE TY_MAP_ENRICHED,
        LV_SRC     TYPE STRING,
        LV_TGT     TYPE STRING,
        LV_SRC2    TYPE STRING,
        LV_TGT2    TYPE STRING,
        LV_SRC_TAB TYPE DD02L-TABNAME,
        LV_TGT_TAB TYPE DD02L-TABNAME,
        LS_FLOW    TYPE TY_FLOW,
        LV_FIELD   TYPE FIELDNAME,
        LV_CURR    TYPE INT2,
        LV_PREV    TYPE INT2.

  " Resolve physical source/target tables from flow
  CLEAR: LV_SRC_TAB, LV_TGT_TAB.
  READ TABLE GT_FLOW INTO LS_FLOW WITH KEY TRANID = PV_TRANID.
  IF SY-SUBRC = 0.
    LV_SRC_TAB = LS_FLOW-S_TABLE_NAME.
    LV_TGT_TAB = LS_FLOW-T_TABLE_NAME.
  ENDIF.

  REFRESH PT_OUT.

  " Load field participants
  SELECT TRANID RULEID PARAMTYPE FIELDNM
    FROM RSTRANFIELD
    INTO CORRESPONDING FIELDS OF TABLE LT_FIELDS
   WHERE TRANID  = PV_TRANID
     AND OBJVERS = 'A'.
  SORT LT_FIELDS BY RULEID PARAMTYPE FIELDNM.

  " Load rule headers
  SELECT TRANID RULEID RULETYPE
    FROM RSTRANRULE
    INTO CORRESPONDING FIELDS OF TABLE LT_RULES
   WHERE TRANID  = PV_TRANID
     AND OBJVERS = 'A'.
  SORT LT_RULES BY TRANID RULEID.

  IF LT_FIELDS IS INITIAL. RETURN. ENDIF.

  "--- Macro: flush accumulated source/target lists for current group
  DEFINE _FLUSH_GROUP.

    IF LT_SOURCES IS INITIAL AND LT_TARGETS IS INITIAL.
      " nothing to emit

    ELSEIF LT_SOURCES IS INITIAL.
      " targets only (e.g. constant assignment)
      LOOP AT LT_TARGETS INTO LV_TGT.
        CLEAR LS_OUT.
        LS_OUT-SOURCE_FIELD = ''.
        LS_OUT-TARGET_FIELD = LV_TGT.
        LS_OUT-RULE_TYPE    = LV_RULE.
        PERFORM ENRICH_FROM_START USING LV_TGT CHANGING LS_OUT.
        IF LV_TGT_TAB IS NOT INITIAL AND LV_TGT IS NOT INITIAL.
          LV_FIELD = LV_TGT.
          PERFORM GET_KEYFLAG USING LV_TGT_TAB LV_FIELD CHANGING LS_OUT-KEYFLAG_T.
        ENDIF.
        APPEND LS_OUT TO PT_OUT.
      ENDLOOP.

    ELSEIF LT_TARGETS IS INITIAL.
      " sources only
      LOOP AT LT_SOURCES INTO LV_SRC.
        CLEAR LS_OUT.
        LS_OUT-SOURCE_FIELD = LV_SRC.
        LS_OUT-TARGET_FIELD = ''.
        LS_OUT-RULE_TYPE    = LV_RULE.
        PERFORM ENRICH_FROM_START USING '' CHANGING LS_OUT.
        IF LV_SRC_TAB IS NOT INITIAL AND LV_SRC IS NOT INITIAL.
          LV_FIELD = LV_SRC.
          PERFORM GET_KEYFLAG USING LV_SRC_TAB LV_FIELD CHANGING LS_OUT-KEYFLAG_S.
        ENDIF.
        APPEND LS_OUT TO PT_OUT.
      ENDLOOP.

    ELSE.
      " many-to-many
      LOOP AT LT_SOURCES INTO LV_SRC2.
        LOOP AT LT_TARGETS INTO LV_TGT2.
          CLEAR LS_OUT.
          LS_OUT-SOURCE_FIELD = LV_SRC2.
          LS_OUT-TARGET_FIELD = LV_TGT2.
          LS_OUT-RULE_TYPE    = LV_RULE.
          PERFORM ENRICH_FROM_START USING LV_TGT2 CHANGING LS_OUT.
          IF LV_SRC_TAB IS NOT INITIAL AND LV_SRC2 IS NOT INITIAL.
            LV_FIELD = LV_SRC2.
            PERFORM GET_KEYFLAG USING LV_SRC_TAB LV_FIELD CHANGING LS_OUT-KEYFLAG_S.
          ENDIF.
          IF LV_TGT_TAB IS NOT INITIAL AND LV_TGT2 IS NOT INITIAL.
            LV_FIELD = LV_TGT2.
            PERFORM GET_KEYFLAG USING LV_TGT_TAB LV_FIELD CHANGING LS_OUT-KEYFLAG_T.
          ENDIF.
          APPEND LS_OUT TO PT_OUT.
        ENDLOOP.
      ENDLOOP.
    ENDIF.

    REFRESH LT_SOURCES.
    REFRESH LT_TARGETS.
    CLEAR   LV_RULE.

  END-OF-DEFINITION.

  " Walk fields, flush on group-break
  CLEAR LV_PREV.

  LOOP AT LT_FIELDS INTO LS_FIELD.
    LV_CURR = LS_FIELD-RULEID.

    " Initialise header for very first group
    IF LV_PREV IS INITIAL.
      READ TABLE LT_RULES INTO LS_RULEH
           WITH KEY TRANID = PV_TRANID RULEID = LV_CURR BINARY SEARCH.
      IF SY-SUBRC = 0.
        LV_RULE = LS_RULEH-RULETYPE.
      ELSE.
        CLEAR LV_RULE.
      ENDIF.
      LV_PREV = LV_CURR.
    ENDIF.

    " Group break
    IF LV_CURR <> LV_PREV.
      _FLUSH_GROUP.
      READ TABLE LT_RULES INTO LS_RULEH
           WITH KEY TRANID = PV_TRANID RULEID = LV_CURR BINARY SEARCH.
      IF SY-SUBRC = 0. LV_RULE = LS_RULEH-RULETYPE. ELSE. CLEAR LV_RULE. ENDIF.
      LV_PREV = LV_CURR.
    ENDIF.

    " Accumulate
    IF LS_FIELD-PARAMTYPE = '0'.
      APPEND LS_FIELD-FIELDNM TO LT_SOURCES.
    ELSEIF LS_FIELD-PARAMTYPE = '1'.
      APPEND LS_FIELD-FIELDNM TO LT_TARGETS.
    ENDIF.
  ENDLOOP.

  " Flush last group
  IF LV_PREV IS NOT INITIAL.
    _FLUSH_GROUP.
  ENDIF.

  " Remove START/END routine rows (not field-level mapping)
  DELETE PT_OUT WHERE RULE_TYPE = 'START'.
  DELETE PT_OUT WHERE RULE_TYPE = 'END'.

ENDFORM.

*---------------------------------------------------------------------*
* FORM GET_KEYFLAG  – reads DD03L-KEYFLAG for (table, field)
*---------------------------------------------------------------------*
FORM GET_KEYFLAG USING    PV_TAB     TYPE DD02L-TABNAME
                          PV_FIELD   TYPE FIELDNAME
                 CHANGING PV_KEYFLAG TYPE C.
  DATA: LV_FIELD TYPE FIELDNAME,
        LV_ORIG  TYPE FIELDNAME,
        LV_KEY   TYPE C.

  CLEAR PV_KEYFLAG.
  IF PV_TAB IS INITIAL OR PV_FIELD IS INITIAL. RETURN. ENDIF.

  LV_FIELD = PV_FIELD.
  LV_ORIG  = PV_FIELD.

  " Normalize namespace for DSO Active tables (/BIC/A*)
  IF PV_TAB CP '/BIC/A*'.
    IF LV_FIELD+0(5) <> '/BIC/' AND LV_FIELD+0(5) <> '/BI0/'.
      IF LV_FIELD+0(1) = '0'.
        CONCATENATE '/BI0/' LV_FIELD INTO LV_FIELD.
      ELSE.
        CONCATENATE '/BIC/' LV_FIELD INTO LV_FIELD.
      ENDIF.
    ENDIF.
  ENDIF.

  SELECT SINGLE KEYFLAG INTO LV_KEY
    FROM DD03L
   WHERE TABNAME   = PV_TAB
     AND FIELDNAME = LV_FIELD.

  " Fallback: try original name
  IF SY-SUBRC <> 0.
    SELECT SINGLE KEYFLAG INTO LV_KEY
      FROM DD03L
     WHERE TABNAME   = PV_TAB
       AND FIELDNAME = LV_ORIG.
  ENDIF.

  PV_KEYFLAG = LV_KEY.
ENDFORM.

FORM WRITE_MAPPING_FILE USING PV_TRANID TYPE RSTRAN-TRANID

        LV_KT   TYPE C LENGTH 1,
        LS      TYPE TY_MAP_ENRICHED,
        OK      TYPE C.

  PERFORM BUILD_TRANS_PREFIX USING PV_TRANID CHANGING LV_PREF.
  CONCATENATE LV_PREF '_Mapping.csv' INTO LV_SFX.
  PERFORM GET_FULLPATH_TRANS USING PV_TRANID LV_SFX CHANGING LV_FILE.
  PERFORM GET_DIRNAME USING LV_FILE CHANGING LV_DIR.
  PERFORM ENSURE_DIR_SMART USING LV_DIR.

  " Header
  APPEND 'TranID;SourceField;SrcKeyFlag;TargetField;TgtKeyFlag;RuleType' TO LT_OUT.

  LOOP AT PT_RULES INTO LS.
    LV_SRC = LS-SOURCE_FIELD.
    LV_TGT = LS-TARGET_FIELD.
    LV_KS  = LS-KEYFLAG_S.
    LV_KT  = LS-KEYFLAG_T.

    " Normalise /BIC/ prefix — source
    IF LV_SRC IS NOT INITIAL.
      IF LV_SRC+0(1) = '0'.
        " SAP-delivered: strip leading '0' (already stored without namespace)
      ELSEIF LV_SRC+0(5) <> '/BIC/' AND LV_SRC+0(5) <> '/BI0/'.
        CONCATENATE '/BIC/' LV_SRC INTO LV_SRC.
      ENDIF.
    ENDIF.

    " Normalise /BIC/ prefix — target
    IF LV_TGT IS NOT INITIAL.
      IF LV_TGT+0(1) = '0'.
        " SAP-delivered: leave as-is
      ELSEIF LV_TGT+0(5) <> '/BIC/' AND LV_TGT+0(5) <> '/BI0/'.
        CONCATENATE '/BIC/' LV_TGT INTO LV_TGT.
      ENDIF.
    ENDIF.

    " Build CSV row — no trailing semicolon (Bug 4)
    CONCATENATE PV_TRANID ';' LV_SRC ';' LV_KS ';'
                LV_TGT ';' LV_KT ';' LS-RULE_TYPE
           INTO LV_LINE.
    APPEND LV_LINE TO LT_OUT.
  ENDLOOP.

  IF P_PATH IS NOT INITIAL.
    PERFORM WRITE_FILE_SMART TABLES LT_OUT USING LV_FILE.
  ELSE.
    PERFORM WRITE_FILE_APPSERVER TABLES LT_OUT USING LV_FILE.
  ENDIF.
ENDFORM.

*---------------------------------------------------------------------*
* FORM RX_COLLECT_ROUTINES
* Extracts start/end/expert/field routines from transformation program.
*---------------------------------------------------------------------*
FORM RX_COLLECT_ROUTINES USING    PV_TRANID TYPE RSTRAN-TRANID
                                  PV_PROG   TYPE PROGNAME
                         CHANGING PT_ROUT   TYPE TY_ROUTINE_TAB.
  DATA: LT_SRC  TYPE STANDARD TABLE OF STRING WITH EMPTY KEY,
        LS_R    TYPE TY_ROUTINE,
        LV_LINE TYPE STRING,
        LV_ID   TYPE STRING,
        LV_METH TYPE STRING,
        LV_OK   TYPE C.

  REFRESH PT_ROUT.

  PERFORM FETCH_REPORT_SOURCE TABLES LT_SRC USING PV_PROG CHANGING LV_OK.
  IF LV_OK IS INITIAL AND LT_SRC IS INITIAL. RETURN. ENDIF.

  " START routine
  CLEAR LS_R.
  PERFORM RX_EXTRACT_METHOD_FULL TABLES LT_SRC USING 'start_routine' CHANGING LS_R.
  IF LS_R-CODE IS NOT INITIAL.
    LS_R-TRANID       = PV_TRANID.
    LS_R-ROUTINE_TYPE = 'START'.
    CLEAR LS_R-ROUTINE_ID.
    APPEND LS_R TO PT_ROUT.
  ENDIF.

  " END routine
  CLEAR LS_R.
  PERFORM RX_EXTRACT_METHOD_FULL TABLES LT_SRC USING 'end_routine' CHANGING LS_R.
  IF LS_R-CODE IS NOT INITIAL.
    LS_R-TRANID       = PV_TRANID.
    LS_R-ROUTINE_TYPE = 'END'.
    CLEAR LS_R-ROUTINE_ID.
    APPEND LS_R TO PT_ROUT.
  ENDIF.

  " EXPERT routine
  CLEAR LS_R.
  PERFORM RX_EXTRACT_METHOD_FULL TABLES LT_SRC USING 'expert_routine' CHANGING LS_R.
  IF LS_R-CODE IS NOT INITIAL.
    LS_R-TRANID       = PV_TRANID.
    LS_R-ROUTINE_TYPE = 'EXPERT'.
    CLEAR LS_R-ROUTINE_ID.
    APPEND LS_R TO PT_ROUT.
  ENDIF.

  " FIELD routines: METHOD compute_<field>
  LOOP AT LT_SRC INTO LV_LINE.
    CLEAR LV_ID.
    FIND REGEX '^\s*METHOD\s+compute_([A-Za-z0-9_/]+)\.'
      IN LV_LINE IGNORING CASE SUBMATCHES LV_ID.
    IF SY-SUBRC <> 0.
      " interface-qualified: IF_xxx~compute_<field>
      FIND REGEX '^\s*METHOD\s+IF_[A-Za-z0-9_]+~compute_([A-Za-z0-9_/]+)\.'
        IN LV_LINE IGNORING CASE SUBMATCHES LV_ID.
    ENDIF.
    IF SY-SUBRC = 0 AND LV_ID IS NOT INITIAL.
      CLEAR LV_METH.
      CONCATENATE 'compute_' LV_ID INTO LV_METH.
      CLEAR LS_R.
      PERFORM RX_EXTRACT_METHOD_FULL TABLES LT_SRC USING LV_METH CHANGING LS_R.
      IF LS_R-CODE IS NOT INITIAL.
        LS_R-TRANID       = PV_TRANID.
        LS_R-ROUTINE_TYPE = 'FIELD'.
        LS_R-ROUTINE_ID   = LV_ID.
        APPEND LS_R TO PT_ROUT.
      ENDIF.
    ENDIF.
  ENDLOOP.
ENDFORM.

*---------------------------------------------------------------------*
* FORM WRITE_ROUTINE_FILES
* Writes each routine block to <TRANID>\<prefix>_Routine_<type>.txt
* Sets scanner context and calls SCAN_ROUTINE_BLOCK per routine.
*---------------------------------------------------------------------*
FORM WRITE_ROUTINE_FILES USING PV_TRANID TYPE RSTRAN-TRANID
                               PT_ROUT   TYPE TY_ROUTINE_TAB.
  DATA: LS      TYPE TY_ROUTINE,
        LV_FILE TYPE STRING,
        LV_SFX  TYPE STRING,
        LV_DIR  TYPE STRING,
        LV_PREF TYPE STRING,
        LT_C    TYPE STANDARD TABLE OF STRING WITH EMPTY KEY,
        OK      TYPE C.

  PERFORM BUILD_TRANS_PREFIX USING PV_TRANID CHANGING LV_PREF.

  LOOP AT PT_ROUT INTO LS.
    IF LS-CODE IS INITIAL. CONTINUE. ENDIF.

    " Build filename suffix per routine type
    CASE LS-ROUTINE_TYPE.
      WHEN 'START'.
        CONCATENATE LV_PREF '_Routine_Start.txt'  INTO LV_SFX.
      WHEN 'END'.
        CONCATENATE LV_PREF '_Routine_End.txt'    INTO LV_SFX.
      WHEN 'EXPERT'.
        CONCATENATE LV_PREF '_Routine_Expert.txt' INTO LV_SFX.
      WHEN OTHERS.
        CONCATENATE LV_PREF '_Routine_Field_' LS-ROUTINE_ID '.txt' INTO LV_SFX.
    ENDCASE.

    PERFORM GET_FULLPATH_TRANS USING PV_TRANID LV_SFX CHANGING LV_FILE.
    PERFORM GET_DIRNAME USING LV_FILE CHANGING LV_DIR.
    PERFORM ENSURE_DIR_SMART USING LV_DIR.

    LT_C = LS-CODE.

    " Set global scanner context and scan for dependencies
    GV_SCAN_TRANID    = PV_TRANID.
    CASE LS-ROUTINE_TYPE.
      WHEN 'START'.  GV_SCAN_CONTAINER = 'start_routine'.
      WHEN 'END'.    GV_SCAN_CONTAINER = 'end_routine'.
      WHEN 'EXPERT'. GV_SCAN_CONTAINER = 'expert_routine'.
      WHEN OTHERS.
        CONCATENATE 'compute_' LS-ROUTINE_ID INTO GV_SCAN_CONTAINER.
    ENDCASE.

    GT_SCAN_SRC = LT_C.
    PERFORM SCAN_ROUTINE_BLOCK.

    IF P_PATH IS NOT INITIAL.
      PERFORM WRITE_FILE_SMART TABLES LT_C USING LV_FILE.
    ELSE.
      PERFORM WRITE_FILE_APPSERVER TABLES LT_C USING LV_FILE.
    ENDIF.
  ENDLOOP.
ENDFORM.

        LV_PROG   TYPE PROGNAME,

*---------------------------------------------------------------------*
* FORM CLEAN_DEP_LOG
* Validates TABLE (DDIC) and INCLUDE (TRDIR); drops noise; de-dups.
* Bug 3 fix: LV_UPPER declared outside loop.
*---------------------------------------------------------------------*
FORM CLEAN_DEP_LOG.
  DATA: LT_CLEAN      TYPE TY_DEP_TAB,
        LS            TYPE TY_DEP,
        LV_UPPER      TYPE STRING,
        LV_DROP       TYPE C,
        LV_CHOP       TYPE STRING,
        LV_DUMMY_TAB  TYPE DD02L-TABNAME,
        LV_DUMMY_PROG TYPE TRDIR-NAME.

  PERFORM INIT_STOPWORDS.

  LOOP AT GT_DEP_LOG INTO LS.
    CLEAR LV_DROP.

    " --- METHOD: prettify arrow notation
    IF LS-DEP_KIND = 'METHOD'.
      LV_CHOP = LS-DEP_NAME.
      IF LS-DEP_NAME CP '->*'.
        LV_CHOP = LS-DEP_NAME+2.
        CONCATENATE 'Instance: ' LV_CHOP INTO LS-DEP_NAME.
      ELSEIF LS-DEP_NAME CP '=>*'.
        LV_CHOP = LS-DEP_NAME+2.
        CONCATENATE 'Static: ' LV_CHOP INTO LS-DEP_NAME.
      ENDIF.
    ENDIF.

    " --- TABLE: filter work tables, stopwords, validate DDIC
    IF LS-DEP_KIND = 'TABLE'.
      IF LS-DEP_NAME CP 'LT_*' OR LS-DEP_NAME CP 'IT_*' OR LS-DEP_NAME CP 'GT_*'.
        CONTINUE.
      ENDIF.
      LV_UPPER = LS-DEP_NAME.
      TRANSLATE LV_UPPER TO UPPER CASE.
      READ TABLE GT_STOPWORDS WITH KEY TABLE_LINE = LV_UPPER TRANSPORTING NO FIELDS.
      IF SY-SUBRC = 0. CONTINUE. ENDIF.

      CLEAR LV_DUMMY_TAB.
      SELECT SINGLE TABNAME INTO LV_DUMMY_TAB FROM DD02L WHERE TABNAME = LS-DEP_NAME.
      IF SY-SUBRC <> 0. CONTINUE. ENDIF.
    ENDIF.

    " --- INCLUDE: validate program exists in TRDIR
    IF LS-DEP_KIND = 'INCLUDE'.
      CLEAR LV_DUMMY_PROG.
      SELECT SINGLE NAME INTO LV_DUMMY_PROG FROM TRDIR WHERE NAME = LS-DEP_NAME.
      IF SY-SUBRC <> 0. CONTINUE. ENDIF.
    ENDIF.

    APPEND LS TO LT_CLEAN.
  ENDLOOP.

  " De-duplicate on business keys
  SORT LT_CLEAN BY TRANID CONTAINER DEP_KIND DEP_NAME WHERE_TXT LINE_NO.
  DELETE ADJACENT DUPLICATES FROM LT_CLEAN
         COMPARING TRANID CONTAINER DEP_KIND DEP_NAME WHERE_TXT.

  GT_DEP_LOG = LT_CLEAN.
ENDFORM.

  " Move integer into TYPE STRING via WRITE to char intermediate
  DATA: LV_TMP TYPE C LENGTH 20.
  CLEAR LV_TMP. WRITE LV_CNT_I   TO LV_TMP LEFT-JUSTIFIED. CONDENSE LV_TMP NO-GAPS.
  GV_CNT_TRAN = LV_TMP.
  CLEAR LV_TMP. WRITE LV_CNT_W_I TO LV_TMP LEFT-JUSTIFIED. CONDENSE LV_TMP NO-GAPS.
  GV_CNT_TRAN_W_ROUT = LV_TMP.
ENDFORM.

FORM COLLECT_PROCESS_CHAINS
     USING PV_CHAIN TYPE RSPC_CHAIN
           PV_PACK  TYPE DEVCLASS.
* Entry point — collects root chain(s), then drives PROCESS_CHAIN_RECURSIVE.
* DDIC: RSPCCHAIN(CHAIN_ID,OBJVERS,TYPE,VARIANTE), RSPCCHAINT(CHAIN_ID,OBJVERS,LANGU,TXTLG)

  DATA: LV_CHAIN_ID TYPE RSPC_CHAIN,
        LV_CNT      TYPE I,
        LV_TXT      TYPE C LENGTH 30,
        LS_HDR2     TYPE TY_PC_HDR,
        LT_ROOTS    TYPE STANDARD TABLE OF RSPC_CHAIN WITH DEFAULT KEY.

  REFRESH: GT_PC_HDR, GT_PC_STEPS, GT_PC_SEEN, GT_PC_XLS.

  IF PV_CHAIN IS NOT INITIAL.
    APPEND PV_CHAIN TO LT_ROOTS.
  ELSE.
    SELECT CHAIN_ID INTO LV_CHAIN_ID FROM RSPCCHAIN
      WHERE OBJVERS = 'A'
      GROUP BY CHAIN_ID.
      APPEND LV_CHAIN_ID TO LT_ROOTS.
      LV_CNT = LV_CNT + 1.
      IF LV_CNT >= 50. EXIT. ENDIF.
    ENDSELECT.
  ENDIF.

  IF LT_ROOTS IS INITIAL.
    MESSAGE 'No process chains found.' TYPE 'I'. RETURN.
  ENDIF.

  " For each root chain load header into GT_PC_HDR (used by WRITE_PC_FILES)
  LOOP AT LT_ROOTS INTO LV_CHAIN_ID.
    CLEAR LS_HDR2. LS_HDR2-CHAIN = LV_CHAIN_ID.
    CLEAR LV_TXT.
    SELECT SINGLE TXTLG INTO LV_TXT FROM RSPCCHAINT
      WHERE CHAIN_ID = LV_CHAIN_ID AND OBJVERS = 'A' AND LANGU = SY-LANGU.
    IF SY-SUBRC <> 0.
      SELECT SINGLE TXTLG INTO LV_TXT FROM RSPCCHAINT
        WHERE CHAIN_ID = LV_CHAIN_ID AND OBJVERS = 'A'.
    ENDIF.
    LS_HDR2-TXTLG = LV_TXT.
    APPEND LS_HDR2 TO GT_PC_HDR.
  ENDLOOP.
ENDFORM.

*---------------------------------------------------------------------*
* FORM WRITE_PC_FILES — orchestrator: for each root chain call
*   PROCESS_CHAIN_RECURSIVE, then write global Excel + index
*---------------------------------------------------------------------*
FORM WRITE_PC_FILES USING PV_OUTDIR TYPE STRING.
  DATA: LS_HDR    TYPE TY_PC_HDR,
        LT_IDX    TYPE STANDARD TABLE OF STRING WITH EMPTY KEY,
        LV_LINE   TYPE STRING,
        LV_FILE   TYPE STRING,
        LV_ROOT   TYPE STRING,
        LV_LAST   TYPE C LENGTH 1,
        LV_LEN    TYPE I,
        LV_OFF    TYPE I,
        LV_MSG    TYPE STRING,
        LV_HC_I   TYPE I,
        LV_HC     TYPE C LENGTH 10,
        LV_SCNT_C TYPE C LENGTH 10,
        LV_SCNT   TYPE I.

  IF GT_PC_HDR IS INITIAL. RETURN. ENDIF.

  LV_ROOT = PV_OUTDIR.
  LV_LEN  = STRLEN( LV_ROOT ).
  IF LV_LEN > 0. LV_OFF = LV_LEN - 1. LV_LAST = LV_ROOT+LV_OFF(1). ENDIF.
  IF LV_LAST <> '\' AND LV_LAST <> '/'. CONCATENATE LV_ROOT '\' INTO LV_ROOT. ENDIF.
  " LV_ROOT not used - LV_ROOT passed directly

  APPEND 'Chain;Description;StepCount' TO LT_IDX.

  LOOP AT GT_PC_HDR INTO LS_HDR.
    " Recursive extraction — mirrors PROCESS_FM
    PERFORM PROCESS_CHAIN_RECURSIVE USING LS_HDR-CHAIN LV_ROOT 0.
    " Mermaid per root chain
    PERFORM WRITE_PC_MERMAID USING LS_HDR-CHAIN LV_ROOT.
  ENDLOOP.

  " Lineage tree (populated during recursion via GT_TREE)
  PERFORM WRITE_PC_LINEAGE USING LV_ROOT.

  " Global Excel CSV (all chains + subchains, all steps)
  PERFORM WRITE_PC_EXCEL_CSV USING LV_ROOT.

  " Global index
  LOOP AT GT_PC_HDR INTO LS_HDR.
    CLEAR: LV_SCNT, LV_SCNT_C.
    LOOP AT GT_PC_XLS TRANSPORTING NO FIELDS
      WHERE CHAIN = LS_HDR-CHAIN AND LEVEL = 0.
      LV_SCNT = LV_SCNT + 1.
    ENDLOOP.
    WRITE LV_SCNT TO LV_SCNT_C LEFT-JUSTIFIED. CONDENSE LV_SCNT_C NO-GAPS.
    CONCATENATE LS_HDR-CHAIN ';' LS_HDR-TXTLG ';' LV_SCNT_C INTO LV_LINE.
    APPEND LV_LINE TO LT_IDX.
  ENDLOOP.
  CONCATENATE LV_ROOT 'ProcessChain_Index.csv' INTO LV_FILE.
  PERFORM WRITE_FILE_SMART TABLES LT_IDX USING LV_FILE.

  LV_HC_I = LINES( GT_SEEN ).
  WRITE LV_HC_I TO LV_HC LEFT-JUSTIFIED. CONDENSE LV_HC NO-GAPS.
  CONCATENATE 'PC extraction done. Chains processed: ' LV_HC INTO LV_MSG.
  MESSAGE LV_MSG TYPE 'I'.

  FREE: LT_IDX, GT_PC_HDR, GT_PC_STEPS, GT_PC_SEEN, GT_PC_XLS.
ENDFORM.

*---------------------------------------------------------------------*
* FORM PROCESS_CHAIN_RECURSIVE
*   Mirrors PROCESS_FM: visited guard, folder, steps CSV, lineage,
*   per-step detail subfolder, recurse into CHAIN-type steps.
*   PV_CHAIN — chain ID
*   PV_PARENT_DIR — parent folder (chain folder created inside it)
*   PV_LEVEL — nesting depth (0 = root)
*---------------------------------------------------------------------*
FORM PROCESS_CHAIN_RECURSIVE
     USING PV_CHAIN      TYPE RSPC_CHAIN
           PV_PARENT_DIR TYPE STRING
           PV_LEVEL      TYPE I.

  DATA: LV_DIR       TYPE STRING,
        LV_FILE      TYPE STRING,
        LV_LINE      TYPE STRING,
        LV_TXT       TYPE C LENGTH 30,
        LV_POS       TYPE I,
        LV_POS_C     TYPE C LENGTH 6,
        LV_NEXT      TYPE I,
        LV_INDENT    TYPE STRING,
        LV_SUB_CHAIN TYPE RSPC_CHAIN,
        LS_STEP      TYPE TY_PC_STEP,
        LT_STEPS     TYPE TY_PC_STEP_TAB,
        LT_CSV       TYPE STANDARD TABLE OF STRING WITH EMPTY KEY,
        LT_LIN       TYPE STANDARD TABLE OF STRING WITH EMPTY KEY.

  DATA: BEGIN OF LS_CHN,
          CHAIN_ID TYPE RSPC_CHAIN,
          TYPE     TYPE RSPC_TYPE,
          VARIANTE TYPE RSPC_VARIANT,
        END OF LS_CHN.

  " ---- Visited guard (prevent infinite loops in cyclic chains) ----
  READ TABLE GT_PC_SEEN WITH KEY TABLE_LINE = PV_CHAIN
    TRANSPORTING NO FIELDS.
  IF SY-SUBRC = 0.
    " Already processed — add note to lineage and return
    DO PV_LEVEL TIMES. CONCATENATE LV_INDENT '  ' INTO LV_INDENT. ENDDO.
    CONCATENATE LV_INDENT PV_CHAIN ' (already processed - skipped)'
           INTO LV_LINE SEPARATED BY ''.
    APPEND LV_LINE TO GT_TREE.
    RETURN.
  ENDIF.
  INSERT PV_CHAIN INTO TABLE GT_PC_SEEN.

  " ---- Build folder: <parent>\<CHAIN>\ ----
  CONCATENATE PV_PARENT_DIR PV_CHAIN '\' INTO LV_DIR.
  PERFORM ENSURE_DIR_SMART USING LV_DIR.

  " ---- Add to lineage tree ----
  DO PV_LEVEL TIMES. CONCATENATE LV_INDENT '  ' INTO LV_INDENT. ENDDO.
  CLEAR LV_TXT.
  SELECT SINGLE TXTLG INTO LV_TXT FROM RSPCCHAINT
    WHERE CHAIN_ID = PV_CHAIN AND OBJVERS = 'A' AND LANGU = SY-LANGU.
  IF SY-SUBRC <> 0.
    SELECT SINGLE TXTLG INTO LV_TXT FROM RSPCCHAINT
      WHERE CHAIN_ID = PV_CHAIN AND OBJVERS = 'A'.
  ENDIF.
  CONCATENATE LV_INDENT '[CHAIN] ' PV_CHAIN ' - ' LV_TXT
         INTO LV_LINE SEPARATED BY ''.
  APPEND LV_LINE TO GT_TREE.

  " ---- Load steps from RSPCCHAIN ----
  CLEAR LV_POS.
  SELECT CHAIN_ID TYPE VARIANTE INTO (LS_CHN-CHAIN_ID, LS_CHN-TYPE, LS_CHN-VARIANTE)
    FROM RSPCCHAIN
    WHERE CHAIN_ID = PV_CHAIN AND OBJVERS = 'A'
    ORDER BY LNR.
    LV_POS = LV_POS + 1.
    CLEAR LS_STEP.
    LS_STEP-CHAIN    = PV_CHAIN.
    LS_STEP-STEPTYPE = LS_CHN-TYPE.
    LS_STEP-STEPNAME = LS_CHN-VARIANTE.
    LS_STEP-STEPPOS  = LV_POS.
    APPEND LS_STEP TO LT_STEPS.

    " Add step to lineage tree
    CLEAR LV_POS_C.
    WRITE LV_POS TO LV_POS_C LEFT-JUSTIFIED. CONDENSE LV_POS_C NO-GAPS.
    CONCATENATE LV_INDENT '  ' LV_POS_C ') ' LS_CHN-TYPE ' | ' LS_CHN-VARIANTE
           INTO LV_LINE SEPARATED BY ''.
    APPEND LV_LINE TO GT_TREE.
  ENDSELECT.

  " ---- Write Steps CSV for this chain ----
  APPEND 'Pos;StepType;StepName' TO LT_CSV.
  LOOP AT LT_STEPS INTO LS_STEP.
    CLEAR LV_POS_C.
    WRITE LS_STEP-STEPPOS TO LV_POS_C LEFT-JUSTIFIED. CONDENSE LV_POS_C NO-GAPS.
    CONCATENATE LV_POS_C ';' LS_STEP-STEPTYPE ';' LS_STEP-STEPNAME INTO LV_LINE.
    APPEND LV_LINE TO LT_CSV.
  ENDLOOP.
  CONCATENATE LV_DIR PV_CHAIN '_Steps.csv' INTO LV_FILE.
  PERFORM WRITE_FILE_SMART TABLES LT_CSV USING LV_FILE.
  FREE LT_CSV.

  " ---- Per-step: write files flat into chain folder + collect XLS row ----
  LOOP AT LT_STEPS INTO LS_STEP.
    PERFORM WRITE_PC_STEP_FILES
      USING LS_STEP LV_DIR PV_LEVEL.
  ENDLOOP.

  " ---- Recurse into CHAIN-type steps — one subfolder per subchain ----
  LV_NEXT = PV_LEVEL + 1.
  LOOP AT LT_STEPS INTO LS_STEP WHERE STEPTYPE = 'CHAIN'.
    LV_SUB_CHAIN = LS_STEP-STEPNAME.
    PERFORM PROCESS_CHAIN_RECURSIVE
      USING LV_SUB_CHAIN LV_DIR LV_NEXT.
  ENDLOOP.

ENDFORM.

*---------------------------------------------------------------------*
* FORM WRITE_PC_STEP_FILES
*   Writes all step files flat into the chain folder (no per-step subfolders).
*   File naming: <CHAIN>_DTP_<DTPNAME>_<SRC>_to_<TGT>.txt
*                <CHAIN>_IP_<IPNAME>_<DATASOURCE>.txt
*                <CHAIN>_ABAP_<VARIANT>_<PROGNAME>.txt + .abap
*                <CHAIN>_TRAN_<TRANID>.abap
*                <CHAIN>_<TYPE>_<VARIANT>.txt  (all other types)
*   Uniqueness: chain prefix + step type + object IDs in filename.
*---------------------------------------------------------------------*
FORM WRITE_PC_STEP_FILES
     USING PS_STEP  TYPE TY_PC_STEP
           PV_DIR   TYPE STRING
           PV_LEVEL TYPE I.

  DATA: LV_FILE       TYPE STRING,
        LV_FNAME      TYPE STRING,
        LV_LINE       TYPE STRING,
        LV_POS_C      TYPE C LENGTH 6,
        LT_OUT        TYPE STANDARD TABLE OF STRING WITH EMPTY KEY,
        LT_ABAP_SRC   TYPE TY_STRINGTAB,
        LV_TRAN_PRG   TYPE PROGNAME,
        LV_TRAN_CNT_I TYPE I,
        LV_TRAN_CNT   TYPE C LENGTH 10,
        LT_ABAP_SRC2  TYPE TY_STRINGTAB,
        LV_ABAP_PROG2 TYPE PROGNAME,
        LV_ABAP_CNT_I TYPE I,
        LV_ABAP_CNT   TYPE C LENGTH 10,
        LV_SEL_CNT    TYPE I,
        LV_SUB_TXT    TYPE C LENGTH 30,
        LV_CHAIN_S    TYPE STRING,
        LV_STEP_S     TYPE STRING,
        LV_SRC_S      TYPE STRING,
        LV_TGT_S      TYPE STRING,
        LV_DS_S       TYPE STRING,
        LV_PROG_S     TYPE STRING.

  DATA: BEGIN OF LS_DTP,
          DTP     TYPE C LENGTH 30,
          SRC     TYPE C LENGTH 30,
          SRCTP   TYPE C LENGTH 6,
          TGT     TYPE C LENGTH 30,
          TGTTP   TYPE C LENGTH 6,
          UPDMODE TYPE C LENGTH 2,
          DTPTXT  TYPE C LENGTH 40,
        END OF LS_DTP.
  DATA: LV_TRAN_ID  TYPE C LENGTH 30,
        LV_TRAN_SRC TYPE C LENGTH 30,
        LV_TRAN_TGT TYPE C LENGTH 30.

  DATA: BEGIN OF LS_IP,
          LOGDPID    TYPE C LENGTH 30,
          SOURCE     TYPE C LENGTH 30,
          LOGSYS     TYPE C LENGTH 10,
          OLTPSOURCE TYPE C LENGTH 30,
          OLTPTYP    TYPE C LENGTH 1,
        END OF LS_IP.
  DATA: BEGIN OF LS_SEL,
          FIELDNAME TYPE C LENGTH 30,
          SIGN      TYPE C LENGTH 1,
          OPT       TYPE C LENGTH 2,
          LOW       TYPE C LENGTH 30,
          HIGH      TYPE C LENGTH 30,
        END OF LS_SEL.

  DATA: LV_ABAP_PROG TYPE C LENGTH 30,
        LV_ABAP_VAR  TYPE C LENGTH 30.

  DATA: LS_XLS TYPE TY_PC_XLS_ROW.

  CLEAR LV_POS_C.
  WRITE PS_STEP-STEPPOS TO LV_POS_C LEFT-JUSTIFIED. CONDENSE LV_POS_C NO-GAPS.

  " Sanitize chain and step name for use in filenames
  LV_CHAIN_S = PS_STEP-CHAIN.
  LV_STEP_S  = PS_STEP-STEPNAME.
  PERFORM SANITIZE_FOR_FILE USING LV_CHAIN_S CHANGING LV_FNAME. LV_CHAIN_S = LV_FNAME.
  PERFORM SANITIZE_FOR_FILE USING LV_STEP_S  CHANGING LV_FNAME. LV_STEP_S  = LV_FNAME.

  " Init XLS row
  CLEAR LS_XLS.
  LS_XLS-CHAIN    = PS_STEP-CHAIN.
  LS_XLS-STEPPOS  = PS_STEP-STEPPOS.
  LS_XLS-STEPTYPE = PS_STEP-STEPTYPE.
  LS_XLS-STEPNAME = PS_STEP-STEPNAME.
  LS_XLS-LEVEL    = PV_LEVEL.
  SELECT SINGLE TXTLG INTO LS_XLS-CHAIN_TXT FROM RSPCCHAINT
    WHERE CHAIN_ID = PS_STEP-CHAIN AND OBJVERS = 'A' AND LANGU = SY-LANGU.

  "=================================================================
  " DTP_LOAD — file: <CHAIN>_DTP_<DTPNAME>_<SRC>_to_<TGT>.txt
  "            tran: <CHAIN>_TRAN_<TRANID>.abap
  "=================================================================
  IF PS_STEP-STEPTYPE = 'DTP_LOAD' OR PS_STEP-STEPTYPE = 'DTP'.
    CLEAR LS_DTP.
    SELECT SINGLE DTP SRC SRCTP TGT TGTTP UPDMODE
      INTO (LS_DTP-DTP, LS_DTP-SRC, LS_DTP-SRCTP,
            LS_DTP-TGT, LS_DTP-TGTTP, LS_DTP-UPDMODE)
      FROM RSBKDTP
      WHERE DTP = PS_STEP-STEPNAME AND OBJVERS = 'A'.
    IF SY-SUBRC = 0.
      SELECT SINGLE TXTLG INTO LS_DTP-DTPTXT FROM RSBKDTPT
        WHERE DTP = PS_STEP-STEPNAME AND OBJVERS = 'A' AND LANGU = SY-LANGU.

      APPEND '=== DTP Step ===' TO LT_OUT.
      CONCATENATE 'DTP         : ' LS_DTP-DTP     INTO LV_LINE. APPEND LV_LINE TO LT_OUT.
      CONCATENATE 'Description : ' LS_DTP-DTPTXT  INTO LV_LINE. APPEND LV_LINE TO LT_OUT.
      CONCATENATE 'Source Obj  : ' LS_DTP-SRC     INTO LV_LINE. APPEND LV_LINE TO LT_OUT.
      CONCATENATE 'Source Type : ' LS_DTP-SRCTP   INTO LV_LINE. APPEND LV_LINE TO LT_OUT.
      CONCATENATE 'Target Obj  : ' LS_DTP-TGT     INTO LV_LINE. APPEND LV_LINE TO LT_OUT.
      CONCATENATE 'Target Type : ' LS_DTP-TGTTP   INTO LV_LINE. APPEND LV_LINE TO LT_OUT.
      CONCATENATE 'Update Mode : ' LS_DTP-UPDMODE INTO LV_LINE. APPEND LV_LINE TO LT_OUT.

      " Transformation link
      CLEAR: LV_TRAN_ID, LV_TRAN_SRC, LV_TRAN_TGT.
      SELECT SINGLE TRANID SOURCENAME TARGETNAME
        INTO (LV_TRAN_ID, LV_TRAN_SRC, LV_TRAN_TGT)
        FROM RSTRAN
        WHERE SOURCENAME = LS_DTP-SRC AND TARGETNAME = LS_DTP-TGT
          AND OBJVERS = 'A'.
      IF SY-SUBRC = 0.
        APPEND '' TO LT_OUT.
        APPEND '--- Transformation ---' TO LT_OUT.
        CONCATENATE 'Tran ID     : ' LV_TRAN_ID  INTO LV_LINE. APPEND LV_LINE TO LT_OUT.
        CONCATENATE 'Tran Source : ' LV_TRAN_SRC INTO LV_LINE. APPEND LV_LINE TO LT_OUT.
        CONCATENATE 'Tran Target : ' LV_TRAN_TGT INTO LV_LINE. APPEND LV_LINE TO LT_OUT.
        " Download transformation ABAP: <CHAIN>_TRAN_<TRANID>.abap
        CLEAR LV_TRAN_PRG.
        CONCATENATE 'RSDRD_TP_' LV_TRAN_ID INTO LV_TRAN_PRG.
        REFRESH LT_ABAP_SRC.
        READ REPORT LV_TRAN_PRG INTO LT_ABAP_SRC.
        IF SY-SUBRC = 0.
          LV_TRAN_CNT_I = LINES( LT_ABAP_SRC ).
          WRITE LV_TRAN_CNT_I TO LV_TRAN_CNT LEFT-JUSTIFIED. CONDENSE LV_TRAN_CNT NO-GAPS.
          CONCATENATE 'Tran Lines  : ' LV_TRAN_CNT INTO LV_LINE. APPEND LV_LINE TO LT_OUT.
          CONCATENATE PV_DIR LV_CHAIN_S '_TRAN_' LV_TRAN_ID '.abap' INTO LV_FILE.
          PERFORM WRITE_FILE_SMART TABLES LT_ABAP_SRC USING LV_FILE.
        ELSE.
          APPEND '(Transformation ABAP not found)' TO LT_OUT.
        ENDIF.
        FREE LT_ABAP_SRC.
      ELSE.
        APPEND '(No RSTRAN entry found for this DTP)' TO LT_OUT.
      ENDIF.
      LS_XLS-SRC     = LS_DTP-SRC.
      LS_XLS-TGT     = LS_DTP-TGT.
      LS_XLS-UPDMODE = LS_DTP-UPDMODE.
    ELSE.
      CONCATENATE 'DTP not found: ' PS_STEP-STEPNAME INTO LV_LINE.
      APPEND LV_LINE TO LT_OUT.
    ENDIF.

    " File: <CHAIN>_DTP_<DTPNAME>_<SRC>_to_<TGT>.txt
    LV_SRC_S = LS_DTP-SRC.
    PERFORM SANITIZE_FOR_FILE USING LV_SRC_S CHANGING LV_FNAME. LV_SRC_S = LV_FNAME.
    LV_TGT_S = LS_DTP-TGT.
    PERFORM SANITIZE_FOR_FILE USING LV_TGT_S CHANGING LV_FNAME. LV_TGT_S = LV_FNAME.
    CONCATENATE PV_DIR LV_CHAIN_S '_DTP_' LV_STEP_S '_' LV_SRC_S '_to_' LV_TGT_S '.txt'
           INTO LV_FILE.
    PERFORM WRITE_FILE_SMART TABLES LT_OUT USING LV_FILE.
    REFRESH LT_OUT.
  ENDIF.

  "=================================================================
  " LOADING (InfoPackage) — file: <CHAIN>_IP_<IPNAME>_<DATASOURCE>.txt
  "=================================================================
  IF PS_STEP-STEPTYPE = 'LOADING' OR PS_STEP-STEPTYPE = 'INFOPACKAGE'.
    CLEAR LS_IP.
    SELECT SINGLE LOGDPID SOURCE LOGSYS OLTPSOURCE OLTPTYP
      INTO (LS_IP-LOGDPID, LS_IP-SOURCE, LS_IP-LOGSYS,
            LS_IP-OLTPSOURCE, LS_IP-OLTPTYP)
      FROM RSLDPIO
      WHERE LOGDPID = PS_STEP-STEPNAME AND OBJVERS = 'A'.
    IF SY-SUBRC = 0.
      APPEND '=== InfoPackage Step ===' TO LT_OUT.
      CONCATENATE 'InfoPackage : ' LS_IP-LOGDPID    INTO LV_LINE. APPEND LV_LINE TO LT_OUT.
      CONCATENATE 'DataSource  : ' LS_IP-SOURCE     INTO LV_LINE. APPEND LV_LINE TO LT_OUT.
      CONCATENATE 'LogSys      : ' LS_IP-LOGSYS     INTO LV_LINE. APPEND LV_LINE TO LT_OUT.
      CONCATENATE 'OltpSource  : ' LS_IP-OLTPSOURCE INTO LV_LINE. APPEND LV_LINE TO LT_OUT.
      CONCATENATE 'Pkg Type    : ' LS_IP-OLTPTYP    INTO LV_LINE. APPEND LV_LINE TO LT_OUT.
      APPEND '' TO LT_OUT.
      APPEND '--- Selection Conditions ---' TO LT_OUT.
      CLEAR LV_SEL_CNT.
      SELECT FIELDNAME SIGN OPT LOW HIGH
        INTO (LS_SEL-FIELDNAME, LS_SEL-SIGN, LS_SEL-OPT, LS_SEL-LOW, LS_SEL-HIGH)
        FROM RSLDPSEL
        WHERE LOGDPID = PS_STEP-STEPNAME AND OBJVERS = 'A'.
        CONCATENATE LS_SEL-FIELDNAME ' ' LS_SEL-SIGN LS_SEL-OPT
                    ' [' LS_SEL-LOW ' - ' LS_SEL-HIGH ']' INTO LV_LINE.
        APPEND LV_LINE TO LT_OUT.
        LV_SEL_CNT = LV_SEL_CNT + 1.
      ENDSELECT.
      IF LV_SEL_CNT = 0. APPEND '(No selections defined)' TO LT_OUT. ENDIF.
      LS_XLS-SRC    = LS_IP-SOURCE.
      LS_XLS-LOGSYS = LS_IP-LOGSYS.
    ELSE.
      CONCATENATE 'InfoPackage not found: ' PS_STEP-STEPNAME INTO LV_LINE.
      APPEND LV_LINE TO LT_OUT.
    ENDIF.

    " File: <CHAIN>_IP_<IPNAME>_<DATASOURCE>.txt
    LV_DS_S = LS_IP-OLTPSOURCE.
    PERFORM SANITIZE_FOR_FILE USING LV_DS_S CHANGING LV_FNAME. LV_DS_S = LV_FNAME.
    CONCATENATE PV_DIR LV_CHAIN_S '_IP_' LV_STEP_S '_' LV_DS_S '.txt' INTO LV_FILE.
    PERFORM WRITE_FILE_SMART TABLES LT_OUT USING LV_FILE.
    REFRESH LT_OUT.
  ENDIF.

  "=================================================================
  " ABAP step — file: <CHAIN>_ABAP_<VARIANT>_<PROGNAME>.txt + .abap
  "=================================================================
  IF PS_STEP-STEPTYPE = 'ABAP'.
    CLEAR: LV_ABAP_PROG, LV_ABAP_VAR.
    LV_ABAP_VAR = PS_STEP-STEPNAME.
    SELECT SINGLE LOW INTO LV_ABAP_PROG FROM RSPCVARIANT
      WHERE TYPE     = 'ABAP'
        AND VARIANTE = LV_ABAP_VAR
        AND OBJVERS  = 'A'
        AND FNAM     = 'REPORT'.

    APPEND '=== ABAP Step ===' TO LT_OUT.
    CONCATENATE 'Variant     : ' LV_ABAP_VAR INTO LV_LINE. APPEND LV_LINE TO LT_OUT.

    IF SY-SUBRC = 0 AND LV_ABAP_PROG IS NOT INITIAL.
      CONCATENATE 'ABAP Report : ' LV_ABAP_PROG INTO LV_LINE. APPEND LV_LINE TO LT_OUT.
      REFRESH LT_ABAP_SRC2.
      LV_ABAP_PROG2 = LV_ABAP_PROG.
      READ REPORT LV_ABAP_PROG2 INTO LT_ABAP_SRC2.
      IF SY-SUBRC = 0.
        LV_ABAP_CNT_I = LINES( LT_ABAP_SRC2 ).
        WRITE LV_ABAP_CNT_I TO LV_ABAP_CNT LEFT-JUSTIFIED. CONDENSE LV_ABAP_CNT NO-GAPS.
        CONCATENATE 'Source lines: ' LV_ABAP_CNT INTO LV_LINE. APPEND LV_LINE TO LT_OUT.
        " .abap file: <CHAIN>_ABAP_<VARIANT>_<PROGNAME>.abap
        LV_PROG_S = LV_ABAP_PROG.
        PERFORM SANITIZE_FOR_FILE USING LV_PROG_S CHANGING LV_FNAME. LV_PROG_S = LV_FNAME.
        CONCATENATE PV_DIR LV_CHAIN_S '_ABAP_' LV_STEP_S '_' LV_PROG_S '.abap' INTO LV_FILE.
        PERFORM WRITE_FILE_SMART TABLES LT_ABAP_SRC2 USING LV_FILE.
      ELSE.
        APPEND '(Source not found via READ REPORT)' TO LT_OUT.
      ENDIF.
      FREE LT_ABAP_SRC2.
      LS_XLS-ABAP_PRG = LV_ABAP_PROG.
    ELSE.
      APPEND '(REPORT parameter not found in RSPCVARIANT)' TO LT_OUT.
    ENDIF.

    " Detail txt: <CHAIN>_ABAP_<VARIANT>.txt
    CONCATENATE PV_DIR LV_CHAIN_S '_ABAP_' LV_STEP_S '.txt' INTO LV_FILE.
    PERFORM WRITE_FILE_SMART TABLES LT_OUT USING LV_FILE.

*---------------------------------------------------------------------*
* FORM MAKE_SQL_TYPE  (single authoritative copy — FIX 3359)
* Maps DDIC data type -> Databricks/Spark SQL type string.
*---------------------------------------------------------------------*
FORM MAKE_SQL_TYPE USING    PV_DATATYPE TYPE DD03P-DATATYPE
                             PV_LENG    TYPE DD03P-LENG
                             PV_DECIM   TYPE DD03P-DECIMALS
                   CHANGING PV_SQLTYPE  TYPE STRING.
  DATA: LV_DT TYPE DD03P-DATATYPE,
        LV    TYPE STRING.

  LV_DT = PV_DATATYPE.
  TRANSLATE LV_DT TO UPPER CASE.

  IF LV_DT = 'CHAR' OR LV_DT = 'LCHR' OR LV_DT = 'LANG' OR LV_DT = 'CLNT'
  OR LV_DT = 'NUMC'.
    CONCATENATE 'VARCHAR(' PV_LENG ')' INTO LV.
  ELSEIF LV_DT = 'RAW' OR LV_DT = 'LRAW'.
    CONCATENATE 'VARBINARY(' PV_LENG ')' INTO LV.
  ELSEIF LV_DT = 'INT1' OR LV_DT = 'INT2'.
    LV = 'SMALLINT'.
  ELSEIF LV_DT = 'INT4'.
    LV = 'INTEGER'.
  ELSEIF LV_DT = 'INT8'.
    LV = 'BIGINT'.
  ELSEIF LV_DT = 'FLTP'.
    LV = 'DOUBLE'.
  ELSEIF LV_DT = 'DEC' OR LV_DT = 'CURR' OR LV_DT = 'QUAN'.
    CONCATENATE 'DECIMAL(' PV_LENG ',' PV_DECIM ')' INTO LV.
  ELSEIF LV_DT = 'DATS'.
    LV = 'DATE'.
  ELSEIF LV_DT = 'TIMS'.
    LV = 'TIME'.
  ELSEIF LV_DT = 'ACCP'.
    LV = 'VARCHAR(6)'.
  ELSE.
    CONCATENATE 'VARCHAR(' PV_LENG ')' INTO LV.
  ENDIF.

  PV_SQLTYPE = LV.
ENDFORM.

*---------------------------------------------------------------------*
* FORM MAKE_SQL_DDL_FROM_DDIC_2  (parameter-less, global IO)
* IN : GV_TABNAME  OUT: GT_SQL
* Bug 4: trailing comma stripped via SHIFT ... DELETING TRAILING.
*---------------------------------------------------------------------*
FORM MAKE_SQL_DDL_FROM_DDIC_2.
  DATA: LT_DD03P TYPE STANDARD TABLE OF DD03P WITH EMPTY KEY,
        LS_DD03P TYPE DD03P,
        LT_KEYS  TYPE STANDARD TABLE OF STRING WITH EMPTY KEY,
        LV_LINE  TYPE STRING,
        LV_TYPE  TYPE STRING,
        LV_K     TYPE STRING,
        LV_LAST  TYPE STRING,
        LV_KLINE TYPE STRING.

  REFRESH GT_SQL.
  REFRESH LT_KEYS.

  CALL FUNCTION 'DDIF_TABL_GET'
    EXPORTING  NAME      = GV_TABNAME
    TABLES     DD03P_TAB = LT_DD03P
    EXCEPTIONS OTHERS    = 1.

  IF SY-SUBRC <> 0 OR LT_DD03P IS INITIAL. RETURN. ENDIF.

  CONCATENATE 'CREATE TABLE "' GV_TABNAME '" (' INTO LV_LINE.
  APPEND LV_LINE TO GT_SQL.

  LOOP AT LT_DD03P INTO LS_DD03P
    WHERE FIELDNAME IS NOT INITIAL AND FIELDNAME <> '.INCLUDE'.
    PERFORM MAKE_SQL_TYPE USING    LS_DD03P-DATATYPE
                                   LS_DD03P-LENG
                                   LS_DD03P-DECIMALS
                          CHANGING LV_TYPE.
    CLEAR LV_LINE.
    CONCATENATE '  "' LS_DD03P-FIELDNAME '" ' LV_TYPE ',' INTO LV_LINE.
    APPEND LV_LINE TO GT_SQL.
    IF LS_DD03P-KEYFLAG = 'X'.
      APPEND LS_DD03P-FIELDNAME TO LT_KEYS.
    ENDIF.
  ENDLOOP.

  " Remove trailing comma from last column line (Bug 4 fix)
  READ TABLE GT_SQL INTO LV_LAST INDEX LINES( GT_SQL ).
  IF SY-SUBRC = 0.
    SHIFT LV_LAST RIGHT DELETING TRAILING ','.
    CONDENSE LV_LAST.
    MODIFY GT_SQL FROM LV_LAST INDEX LINES( GT_SQL ).
  ENDIF.

  " PRIMARY KEY clause
  IF LT_KEYS IS NOT INITIAL.
    CLEAR LV_KLINE.
    LOOP AT LT_KEYS INTO LV_K.
      IF LV_KLINE IS INITIAL.
        LV_KLINE = LV_K.
      ELSE.
        CONCATENATE LV_KLINE '","' LV_K INTO LV_KLINE.
      ENDIF.
    ENDLOOP.
    CONCATENATE '  ,PRIMARY KEY ("' LV_KLINE '")' INTO LV_LINE.
    APPEND LV_LINE TO GT_SQL.
  ENDIF.

  APPEND ');' TO GT_SQL.
ENDFORM.

*---------------------------------------------------------------------*
* FORM MAKE_SQL_DDL_FROM_TS  (parameter-less, global IO)
* IN : GV_DSNAME  OUT: GT_SQL
*---------------------------------------------------------------------*
FORM MAKE_SQL_DDL_FROM_TS.
  DATA: LT_TS   TYPE STANDARD TABLE OF RSTSFIELD WITH EMPTY KEY,
        LS_TS   TYPE RSTSFIELD,
        LV_LINE TYPE STRING,
        LV_TYPE TYPE STRING,
        LV_LAST TYPE STRING.

  REFRESH GT_SQL.
  IF GV_DSNAME IS INITIAL. RETURN. ENDIF.

  SELECT * FROM RSTSFIELD INTO TABLE LT_TS
    WHERE TRANSTRU = GV_DSNAME AND OBJVERS = 'A'
    ORDER BY POSIT.
  IF SY-SUBRC <> 0 OR LT_TS IS INITIAL. RETURN. ENDIF.

  CONCATENATE 'CREATE TABLE "' GV_DSNAME '_TS" (' INTO LV_LINE.
  APPEND LV_LINE TO GT_SQL.

  LOOP AT LT_TS INTO LS_TS.
    PERFORM MAKE_SQL_TYPE USING    LS_TS-DATATYPE
                                   LS_TS-LENG
                                   LS_TS-DECIMALS
                          CHANGING LV_TYPE.
    CLEAR LV_LINE.
    CONCATENATE '  "' LS_TS-FIELDNM '" ' LV_TYPE ',' INTO LV_LINE.
    APPEND LV_LINE TO GT_SQL.
  ENDLOOP.

  " Remove trailing comma (Bug 4)
  READ TABLE GT_SQL INTO LV_LAST INDEX LINES( GT_SQL ).
  IF SY-SUBRC = 0.
    SHIFT LV_LAST RIGHT DELETING TRAILING ','.
    CONDENSE LV_LAST.
    MODIFY GT_SQL FROM LV_LAST INDEX LINES( GT_SQL ).
  ENDIF.

  APPEND ');' TO GT_SQL.
ENDFORM.

*---------------------------------------------------------------------*
* FORM COLLECT_CUBE_TABLES  (parameter-less, global IO)
* IN : GV_CUBE  OUT: GT_TABNAMES  (F*, E*, D* physical tables)
* FIX 3034/3035: GT_TABNAMES must have DEFAULT KEY, not EMPTY KEY,
*                to avoid "empty primary key" warnings on SELECT INTO TABLE.
*                Declaration must be in global data - here we use WITH DEFAULT KEY
*                for the local copy LT_TMP.

  DELETE ADJACENT DUPLICATES FROM GT_TABNAMES.

*---------------------------------------------------------------------*
* FORM RESOLVE_PSA_TABLE_NAME_RSDSSEG
* Resolves the physical PSA table name from RSDSSEG.
*---------------------------------------------------------------------*
FORM RESOLVE_PSA_TABLE_NAME_RSDSSEG
       USING    PV_DS      TYPE RSDS-DATASOURCE
                PV_LOGSYS  TYPE RSSLOGSYS
       CHANGING PV_PSA_TAB TYPE DD02L-TABNAME.

  DATA: LV_PSA_NAME TYPE C LENGTH 30,
        LV_PSA_VER  TYPE C LENGTH 4,
        LV_TECH     TYPE DD02L-TABNAME.

  CLEAR PV_PSA_TAB.
  IF PV_DS IS INITIAL. RETURN. ENDIF.

  " With LOGSYS — avoids wrong PSA in multi-source landscape
  IF PV_LOGSYS IS NOT INITIAL.
    SELECT SINGLE PSA PSAVERS INTO (LV_PSA_NAME, LV_PSA_VER)
      FROM RSDSSEG
      WHERE DATASOURCE = PV_DS AND LOGSYS = PV_LOGSYS AND OBJVERS = 'A'.
    IF SY-SUBRC <> 0.
      SELECT SINGLE PSA PSAVERS INTO (LV_PSA_NAME, LV_PSA_VER)
        FROM RSDSSEG
        WHERE DATASOURCE = PV_DS AND LOGSYS = PV_LOGSYS AND OBJVERS = 'M'.
    ENDIF.
  ENDIF.

  " Fallback without LOGSYS (flat-file / generic extractors)
  IF LV_PSA_NAME IS INITIAL.
    SELECT SINGLE PSA PSAVERS INTO (LV_PSA_NAME, LV_PSA_VER)
      FROM RSDSSEG WHERE DATASOURCE = PV_DS AND OBJVERS = 'A'.
  ENDIF.

  IF LV_PSA_NAME IS INITIAL OR LV_PSA_VER IS INITIAL. RETURN. ENDIF.

  " Resolve technical table name from RSTSODS
  SELECT SINGLE ODSNAME_TECH INTO LV_TECH FROM RSTSODS
    WHERE ODSNAME = LV_PSA_NAME AND VERSION = LV_PSA_VER AND OBJSTAT = 'ACT'.
  IF SY-SUBRC <> 0 OR LV_TECH IS INITIAL.
    SELECT SINGLE ODSNAME_TECH INTO LV_TECH FROM RSTSODS
      WHERE ODSNAME = LV_PSA_NAME AND VERSION = LV_PSA_VER.
  ENDIF.
  IF LV_TECH IS INITIAL. RETURN. ENDIF.

  IF LV_TECH(5) = '/BIC/'.
    PV_PSA_TAB = LV_TECH.
  ELSE.
    CONCATENATE '/BIC/' LV_TECH INTO PV_PSA_TAB.
  ENDIF.
ENDFORM.

*---------------------------------------------------------------------*
* FORM COLLECT_MD_FROM_FACT  (parameter-less, global IO)
* IN : GV_TABNAME (fact table)
* OUT: GT_DDL_TABS (adds P/T/H tables), GT_MD_IOBJ
*---------------------------------------------------------------------*
FORM COLLECT_MD_FROM_FACT.
  DATA: LT_DD03P TYPE STANDARD TABLE OF DD03P WITH EMPTY KEY,
        LS_DD03P TYPE DD03P,
        LV_IO    TYPE RSIOBJNM,
        LV_PREF  TYPE STRING,
        LV_TAB_P TYPE DD02L-TABNAME,
        LV_TAB_T TYPE DD02L-TABNAME,
        LV_LIKE  TYPE DD02L-TABNAME,
        LT_H     TYPE STANDARD TABLE OF DD02L-TABNAME WITH DEFAULT KEY,
        LV_H     TYPE DD02L-TABNAME,
        LV_CHECK TYPE DD02L-TABNAME.

  IF GV_TABNAME IS INITIAL. RETURN. ENDIF.

  CALL FUNCTION 'DDIF_TABL_GET'
    EXPORTING  NAME      = GV_TABNAME
    TABLES     DD03P_TAB = LT_DD03P
    EXCEPTIONS OTHERS    = 1.
  IF SY-SUBRC <> 0 OR LT_DD03P IS INITIAL. RETURN. ENDIF.

  LOOP AT LT_DD03P INTO LS_DD03P WHERE FIELDNAME CP 'SID_*'.
    LV_IO = LS_DD03P-FIELDNAME+4.
    CONDENSE LV_IO NO-GAPS.
    IF LV_IO IS INITIAL. CONTINUE. ENDIF.

    IF LV_IO CP '0*'.
      LV_PREF = '/BI0/'.
    ELSE.
      LV_PREF = '/BIC/'.
    ENDIF.

    " P-table
    CONCATENATE LV_PREF 'P' LV_IO INTO LV_TAB_P.
    CLEAR LV_CHECK.
    SELECT SINGLE TABNAME INTO LV_CHECK FROM DD02L WHERE TABNAME = LV_TAB_P.
    IF SY-SUBRC = 0. INSERT LV_TAB_P INTO TABLE GT_DDL_TABS. ENDIF.

    " T-table
    CONCATENATE LV_PREF 'T' LV_IO INTO LV_TAB_T.
    CLEAR LV_CHECK.
    SELECT SINGLE TABNAME INTO LV_CHECK FROM DD02L WHERE TABNAME = LV_TAB_T.
    IF SY-SUBRC = 0. INSERT LV_TAB_T INTO TABLE GT_DDL_TABS. ENDIF.

    " H-tables (multiple possible)
    CONCATENATE LV_PREF 'H' LV_IO '%' INTO LV_LIKE.
    REFRESH LT_H.
    SELECT TABNAME INTO TABLE LT_H FROM DD02L WHERE TABNAME LIKE LV_LIKE.
    LOOP AT LT_H INTO LV_H.
      INSERT LV_H INTO TABLE GT_DDL_TABS.
    ENDLOOP.

    INSERT LV_IO INTO TABLE GT_MD_IOBJ.
  ENDLOOP.
ENDFORM.

      LV_DSO = LS_F-TARGET. INSERT LV_DSO INTO TABLE LT_DSOS.
    ENDIF.
    IF LS_F-SOURCE_TYPE = 'DSO'.
      LV_DSO = LS_F-SOURCE(30). INSERT LV_DSO INTO TABLE LT_DSOS.
    ENDIF.
    IF LS_F-SOURCE_TYPE = 'DATASOURCE' OR LS_F-SOURCE_TYPE = 'TRANSFER STRUCTURE'.
      LV_DS_DISP = LS_F-SOURCE.
      CLEAR LV_DS.
      PERFORM EXTRACT_PLAIN_DSNAME USING LV_DS_DISP CHANGING LV_DS.
      IF LV_DS IS NOT INITIAL.
        INSERT LV_DS INTO TABLE LT_DS.
        " Collect DataSource/LogSys pair for PSA resolution
        CLEAR LS_DSL.
        LS_DSL-DS     = LV_DS.
        LS_DSL-LOGSYS = LS_F-SOURCE_LOGSYS.
        INSERT LS_DSL INTO TABLE LT_DSLS.
      ENDIF.
    ENDIF.
    IF LS_F-TARGET_TYPE = 'CUBE' OR LS_F-SOURCE_TYPE = 'CUBE'.
      IF LS_F-TARGET_TYPE = 'CUBE'.
        LV_CUBE = LS_F-TARGET.
      ELSE.
        LV_CUBE = LS_F-SOURCE.
      ENDIF.
      INSERT LV_CUBE INTO TABLE LT_CUB.
    ENDIF.
  ENDLOOP.

  " 2) Build DDL directory
  PERFORM BUILD_DDL_DIR USING GV_CTX CHANGING LV_DDL_DIR.
  PERFORM ENSURE_DDL_SUBDIRS USING LV_DDL_DIR.

  " 3) Silver: DSO Active tables
  LOOP AT LT_DSOS INTO LV_DSO.
    " Active table /BIC/A<DSO>00
    CONCATENATE '/BIC/A' LV_DSO '00' INTO LV_TAB.
    REFRESH LT_SQL. GV_TABNAME = LV_TAB.
    PERFORM MAKE_SQL_DDL_FROM_DDIC_2. LT_SQL = GT_SQL.
    IF LT_SQL IS INITIAL. CONTINUE. ENDIF.
    LV_BASE = LV_TAB.
    PERFORM SANITIZE_FOR_FILE USING LV_BASE CHANGING LV_BASE.
    PERFORM MAKE_ABS_FILE_NAME USING LV_DDL_DIR 'Silver' LV_BASE '_Silver.txt'
                               CHANGING LV_FILE.
    PERFORM WRITE_DDL_FILE TABLES LT_SQL USING LV_FILE.
    APPEND LV_FILE TO LT_LOG_SILVER.
    LV_CNT_S = LV_CNT_S + 1.
    APPEND '' TO LT_ALLSQL.
    APPEND '/* ===== SILVER (DSO Active) ===== */' TO LT_ALLSQL.
    APPEND LV_TAB TO LT_ALLSQL. APPEND LINES OF LT_SQL TO LT_ALLSQL.
  ENDLOOP.

  " 4) Bronze: PSA tables
  LOOP AT LT_DSLS INTO LS_DSL.
    PERFORM RESOLVE_PSA_TABLE_NAME_RSDSSEG
            USING LS_DSL-DS LS_DSL-LOGSYS CHANGING LV_PSA.
    IF LV_PSA IS INITIAL.
      CONCATENATE '/BIC/B' LS_DSL-DS '00' INTO LV_PSA.
    ENDIF.
    REFRESH LT_SQL. GV_TABNAME = LV_PSA.
    PERFORM MAKE_SQL_DDL_FROM_DDIC_2. LT_SQL = GT_SQL.
    IF LT_SQL IS INITIAL. CONTINUE. ENDIF.
    LV_BASE = LV_PSA.
    PERFORM SANITIZE_FOR_FILE USING LV_BASE CHANGING LV_BASE.
    PERFORM MAKE_ABS_FILE_NAME USING LV_DDL_DIR 'Bronze' LV_BASE '_PSA_Bronze.txt'
                               CHANGING LV_FILE.
    PERFORM WRITE_DDL_FILE TABLES LT_SQL USING LV_FILE.
    APPEND LV_FILE TO LT_LOG_BRONZE. LV_CNT_B = LV_CNT_B + 1.
    APPEND '' TO LT_ALLSQL.
    APPEND '/* ===== BRONZE (PSA) ===== */' TO LT_ALLSQL.
    APPEND LV_PSA TO LT_ALLSQL. APPEND LINES OF LT_SQL TO LT_ALLSQL.
  ENDLOOP.

  " 5) Silver: Transfer Structures
  LOOP AT LT_DS INTO LV_DS.
    REFRESH LT_SQL. GV_DSNAME = LV_DS.
    PERFORM MAKE_SQL_DDL_FROM_TS. LT_SQL = GT_SQL.
    IF LT_SQL IS INITIAL. CONTINUE. ENDIF.
    LV_BASE = LV_DS.
    PERFORM SANITIZE_FOR_FILE USING LV_BASE CHANGING LV_BASE.
    PERFORM MAKE_ABS_FILE_NAME USING LV_DDL_DIR 'Silver' LV_BASE '_TransferStructure.txt'
                               CHANGING LV_FILE.
    PERFORM WRITE_DDL_FILE TABLES LT_SQL USING LV_FILE.
    LV_CNT_S = LV_CNT_S + 1.
  ENDLOOP.

  " 6) Gold: InfoCube tables
  LOOP AT LT_CUB INTO LV_CUBE.
    REFRESH LT_META.
    APPEND '/* ===== GOLD (InfoCube) =====' TO LT_META.
    CONCATENATE ' CUBE: ' LV_CUBE INTO LV_FILE. APPEND LV_FILE TO LT_META.
    APPEND '  Fact/Delta: /BIC/F<cube>*, /BIC/E<cube>*' TO LT_META.
    APPEND '  Dimensions: /BIC/D<cube>*'                TO LT_META.
    APPEND '*/' TO LT_META.
    LV_BASE = LV_CUBE.
    PERFORM SANITIZE_FOR_FILE USING LV_BASE CHANGING LV_BASE.
    PERFORM MAKE_ABS_FILE_NAME USING LV_DDL_DIR 'Gold' LV_BASE '_Cube_Meta.txt'
                               CHANGING LV_FILE.
    PERFORM WRITE_DDL_FILE TABLES LT_META USING LV_FILE.
    APPEND LV_FILE TO LT_LOG_GOLD. LV_CNT_G = LV_CNT_G + 1.

    SELECT SINGLE * INTO LS_CUBE_R FROM RSDCUBE
      WHERE INFOCUBE = LV_CUBE AND OBJVERS = 'A'.
    IF SY-SUBRC = 0 AND LS_CUBE_R-CUBETYPE = 'REALTIME'.
      LV_OBJ_STR = LV_CUBE.
      INSERT LV_OBJ_STR INTO TABLE GT_FLAG_CUBE.
    ENDIF.

    GV_CUBE = LV_CUBE.
    PERFORM COLLECT_CUBE_TABLES.
    LT_TABNAMES = GT_TABNAMES.

    LOOP AT LT_TABNAMES INTO LV_TAB.
      REFRESH LT_SQL. GV_TABNAME = LV_TAB.
      PERFORM MAKE_SQL_DDL_FROM_DDIC_2. LT_SQL = GT_SQL.
      IF LT_SQL IS INITIAL. CONTINUE. ENDIF.
      CLEAR LV_BASE. LV_BASE = LV_TAB.
      PERFORM SANITIZE_FOR_FILE USING LV_BASE CHANGING LV_BASE.
      PERFORM MAKE_ABS_FILE_NAME USING LV_DDL_DIR 'Gold' LV_BASE '_Gold.txt'
                                 CHANGING LV_FILE.
      PERFORM WRITE_DDL_FILE TABLES LT_SQL USING LV_FILE.
      APPEND LV_FILE TO LT_LOG_GOLD.
      LV_CNT_G = LV_CNT_G + 1.
      APPEND '' TO LT_ALLSQL.
      APPEND '/* ===== GOLD (Cube Physical) ===== */' TO LT_ALLSQL.
      APPEND LV_TAB TO LT_ALLSQL.
      APPEND LINES OF LT_SQL TO LT_ALLSQL.

      " Collect MD from fact/delta tables
      IF LV_TAB CP '/BIC/F*' OR LV_TAB CP '/BIC/E*'.
        GV_TABNAME = LV_TAB.
        PERFORM COLLECT_MD_FROM_FACT.
      ENDIF.
    ENDLOOP.
  ENDLOOP.

  " 7) Consolidated DDL file
  PERFORM MAKE_ABS_FILE_NAME USING LV_DDL_DIR '' GV_CTX '_DDL_Consolidated.txt'
                             CHANGING LV_CON.
  PERFORM WRITE_DDL_FILE TABLES LT_ALLSQL USING LV_CON.

  " 8) DDL Log
  CONCATENATE 'DDL run: ' LV_NOW INTO LV_FILE. APPEND LV_FILE TO LT_LOG.
  CONCATENATE 'Root   : ' LV_DDL_DIR INTO LV_FILE. APPEND LV_FILE TO LT_LOG.
  APPEND '' TO LT_LOG.
  APPEND '--- SILVER (DSO Active + TS) ---' TO LT_LOG.
  APPEND LINES OF LT_LOG_SILVER TO LT_LOG.
  APPEND '' TO LT_LOG.
  APPEND '--- BRONZE (PSA) ---' TO LT_LOG.
  APPEND LINES OF LT_LOG_BRONZE TO LT_LOG.
  APPEND '' TO LT_LOG.
  APPEND '--- GOLD (Cube) ---' TO LT_LOG.
  APPEND LINES OF LT_LOG_GOLD TO LT_LOG.

  PERFORM MAKE_ABS_FILE_NAME USING LV_DDL_DIR '' 'DDL_Log' '.txt'
                             CHANGING LV_LOGFILE.
  PERFORM WRITE_DDL_FILE TABLES LT_LOG USING LV_LOGFILE.

  " 9) Commit integer counters -> TYPE STRING globals
  CLEAR LV_CNT_C.
  WRITE LV_CNT_B TO LV_CNT_C LEFT-JUSTIFIED. CONDENSE LV_CNT_C NO-GAPS.
  GV_CNT_BRONZE = LV_CNT_C.
  CLEAR LV_CNT_C.
  WRITE LV_CNT_S TO LV_CNT_C LEFT-JUSTIFIED. CONDENSE LV_CNT_C NO-GAPS.
  GV_CNT_SILVER = LV_CNT_C.
  CLEAR LV_CNT_C.
  WRITE LV_CNT_G TO LV_CNT_C LEFT-JUSTIFIED. CONDENSE LV_CNT_C NO-GAPS.
  GV_CNT_GOLD = LV_CNT_C.
ENDFORM.

FORM PROCESS_PROVIDER USING PV_PROVIDER TYPE CSEQUENCE.

  DATA: LS_TRAN     TYPE RSTRAN,
        LS_FLOW     TYPE TY_FLOW,
        LS_ROW      TYPE TY_FLOW,
        LV_SRC      TYPE C LENGTH 30,
        LV_SRC_DISP TYPE RSDS-DATASOURCE,
        LV_DS_PLAIN TYPE RSDS-DATASOURCE,
        LV_PROGRAM  TYPE PROGNAME,
        LT_LS       TYPE SORTED TABLE OF RSSLOGSYS WITH UNIQUE KEY TABLE_LINE,
        LV_LS       TYPE RSSLOGSYS.

  " visited guard
  READ TABLE GT_SEEN WITH KEY TABLE_LINE = PV_PROVIDER TRANSPORTING NO FIELDS.
  IF SY-SUBRC = 0.
    RETURN.
  ENDIF.
  INSERT PV_PROVIDER INTO TABLE GT_SEEN.

  LOOP AT GT_TRAN INTO LS_TRAN WHERE TARGETNAME = PV_PROVIDER.

    CLEAR: LS_FLOW, LV_SRC, LV_SRC_DISP, LV_DS_PLAIN, LV_PROGRAM.

    " basics
    LS_FLOW-CONTEXT_PROVIDER = GV_CTX.
    LS_FLOW-TARGET           = LS_TRAN-TARGETNAME.
    LS_FLOW-TRANID           = LS_TRAN-TRANID.

    IF LS_TRAN-TRANPROG IS NOT INITIAL.
      IF LS_TRAN-TRANPROG CS 'GP'.
        LV_PROGRAM = LS_TRAN-TRANPROG.
      ELSE.
        CONCATENATE 'GP' LS_TRAN-TRANPROG INTO LV_PROGRAM.
      ENDIF.
    ENDIF.
    LS_FLOW-PROGRAM = LV_PROGRAM.

    " target type
    PERFORM GET_PROVIDER_TYPE USING LS_FLOW-TARGET LS_TRAN-TARGETTYPE CHANGING LS_FLOW-TARGET_TYPE.

    " source name/type
    LV_SRC = LS_TRAN-SOURCENAME.
    PERFORM GET_PROVIDER_TYPE USING LV_SRC LS_TRAN-SOURCETYPE CHANGING LS_FLOW-SOURCE_TYPE.

    " plain DS (strip "(LOGSYS)")
    LV_SRC_DISP = LV_SRC.
    PERFORM EXTRACT_PLAIN_DSNAME USING LV_SRC_DISP CHANGING LV_DS_PLAIN.
    IF LV_DS_PLAIN IS INITIAL.
      LV_DS_PLAIN = LV_SRC.
    ENDIF.
    LS_FLOW-SOURCE = LV_DS_PLAIN.

    "==================== DS / TS (split per LOGSYS) ====================
    IF LS_FLOW-SOURCE_TYPE = 'DATASOURCE' OR LS_FLOW-SOURCE_TYPE = 'TRANSFER STRUCTURE'.

      " collect ALL logsys for this DS
      REFRESH LT_LS.

      " 1) Most accurate: RSDSSEG (segmenting defines PSA assignment)
      SELECT LOGSYS INTO TABLE LT_LS FROM RSDSSEG
        WHERE DATASOURCE = LV_DS_PLAIN AND OBJVERS = 'A'.
      IF LT_LS IS INITIAL.
        SELECT LOGSYS INTO TABLE LT_LS FROM RSDSSEG
          WHERE DATASOURCE = LV_DS_PLAIN AND OBJVERS = 'M'.
      ENDIF.

      " 2) Fallback: RSDS (active then modified)
      IF LT_LS IS INITIAL.
        SELECT LOGSYS INTO TABLE LT_LS FROM RSDS
          WHERE DATASOURCE = LV_DS_PLAIN AND OBJVERS = 'A'.
        IF LT_LS IS INITIAL.
          SELECT LOGSYS INTO TABLE LT_LS FROM RSDS
            WHERE DATASOURCE = LV_DS_PLAIN AND OBJVERS = 'M'.
        ENDIF.
      ENDIF.

      IF LT_LS IS INITIAL.
        " No logsys found at all → emit one row (blank logsys), PSA resolver will fallback
        CLEAR LS_FLOW-SOURCE_LOGSYS.
        PERFORM GET_TABLE_NAME
          USING    LS_FLOW-TARGET        LS_FLOW-TARGET_TYPE
                   LS_FLOW-SOURCE        LS_FLOW-SOURCE_TYPE
                   LS_FLOW-SOURCE_LOGSYS ''       " PV_T_LOGSYS unused
          CHANGING LS_FLOW-S_TABLE_NAME  LS_FLOW-T_TABLE_NAME.
        APPEND LS_FLOW TO GT_FLOW.

      ELSE.
        " one Flow row per LOGSYS (the key change)
        LOOP AT LT_LS INTO LV_LS.
          LS_ROW = LS_FLOW.
          LS_ROW-SOURCE_LOGSYS = LV_LS.

          PERFORM GET_TABLE_NAME
            USING    LS_ROW-TARGET        LS_ROW-TARGET_TYPE
                     LS_ROW-SOURCE        LS_ROW-SOURCE_TYPE
                     LS_ROW-SOURCE_LOGSYS ''
            CHANGING LS_ROW-S_TABLE_NAME  LS_ROW-T_TABLE_NAME.

          APPEND LS_ROW TO GT_FLOW.
        ENDLOOP.
      ENDIF.

      " do NOT recurse for DS
      CONTINUE.
    ENDIF.

    "==================== Non-DS sources: single row ====================
    CLEAR LS_FLOW-SOURCE_LOGSYS.
    PERFORM GET_TABLE_NAME
      USING    LS_FLOW-TARGET        LS_FLOW-TARGET_TYPE
               LS_FLOW-SOURCE        LS_FLOW-SOURCE_TYPE
               LS_FLOW-SOURCE_LOGSYS ''
      CHANGING LS_FLOW-S_TABLE_NAME  LS_FLOW-T_TABLE_NAME.
    APPEND LS_FLOW TO GT_FLOW.

    " recurse only for non-RSDS sources
    IF LS_TRAN-SOURCETYPE <> 'RSDS'.
      PERFORM PROCESS_PROVIDER USING LV_SRC.
    ENDIF.

  ENDLOOP.

ENDFORM.

FORM GET_TABLE_NAME USING    PV_TARGET      TYPE CSEQUENCE
                             PV_TARGETTYPE  TYPE CSEQUENCE
                             PV_SOURCE      TYPE CSEQUENCE
                             PV_SOURCETYPE  TYPE CSEQUENCE
                             PV_S_LOGSYS    TYPE CSEQUENCE
                             PV_T_LOGSYS    TYPE CSEQUENCE
                    CHANGING PV_S_TABNAME   TYPE STRING
                             PV_T_TABNAME   TYPE STRING.

  DATA: LV_DS_IN    TYPE RSDS-DATASOURCE,
        LV_DS_PLAIN TYPE RSDS-DATASOURCE,
        LV_PSA_TAB  TYPE DD02L-TABNAME.

  CLEAR: PV_S_TABNAME, PV_T_TABNAME.

  "----------------- Target side -----------------
  IF PV_TARGETTYPE = 'DSO'.
    CONCATENATE '/BIC/A' PV_TARGET '00' INTO PV_T_TABNAME.
  ELSEIF PV_TARGETTYPE = 'TRANSFER STRUCTURE' OR PV_TARGETTYPE = 'DATASOURCE'.
    LV_DS_IN = PV_TARGET.
    PERFORM EXTRACT_PLAIN_DSNAME USING LV_DS_IN CHANGING LV_DS_PLAIN.

    CLEAR LV_PSA_TAB.
    PERFORM RESOLVE_PSA_TABLE_NAME_RSDSSEG USING LV_DS_PLAIN PV_T_LOGSYS CHANGING LV_PSA_TAB.

    " Fallback ONLY if still blank
    IF LV_PSA_TAB IS INITIAL.
      CONCATENATE '/BIC/B' LV_DS_PLAIN '00' INTO LV_PSA_TAB.
    ENDIF.

    " Set once, never overwrite later
    PV_T_TABNAME = LV_PSA_TAB.
  ENDIF.

  "----------------- Source side -----------------
  IF PV_SOURCETYPE = 'DSO'.
    CONCATENATE '/BIC/A' PV_SOURCE '00' INTO PV_S_TABNAME.
  ELSEIF PV_SOURCETYPE = 'TRANSFER STRUCTURE' OR PV_SOURCETYPE = 'DATASOURCE'.
    LV_DS_IN = PV_SOURCE.
    PERFORM EXTRACT_PLAIN_DSNAME USING LV_DS_IN CHANGING LV_DS_PLAIN.

    CLEAR LV_PSA_TAB.
    PERFORM RESOLVE_PSA_TABLE_NAME_RSDSSEG USING LV_DS_PLAIN PV_S_LOGSYS CHANGING LV_PSA_TAB.

    IF LV_PSA_TAB IS INITIAL.
      CONCATENATE '/BIC/B' LV_DS_PLAIN '00' INTO LV_PSA_TAB.
    ENDIF.

    PV_S_TABNAME = LV_PSA_TAB.
  ENDIF.

ENDFORM.

FORM GET_GP_PROGRAM USING PV_TRANID TYPE RSTRAN-TRANID
                    CHANGING PV_PROG TYPE PROGNAME.

  DATA: LV_TRANPROG TYPE RSTRAN-TRANPROG,
        LT_SRC      TYPE STANDARD TABLE OF STRING WITH EMPTY KEY,
        LV_SRC_LINE TYPE STRING,
        LV_NAME     TYPE PROGNAME.
  CLEAR PV_PROG.

  " 1: Read transformation TRANPROG (active)
  SELECT SINGLE TRANPROG INTO LV_TRANPROG
    FROM RSTRAN
    WHERE TRANID = PV_TRANID
      AND OBJVERS = 'A'.

  " If missing, fallback to M
  IF SY-SUBRC <> 0 OR LV_TRANPROG IS INITIAL.
    SELECT SINGLE TRANPROG INTO LV_TRANPROG
      FROM RSTRAN
      WHERE TRANID = PV_TRANID
        AND OBJVERS = 'M'.
  ENDIF.

  IF LV_TRANPROG IS INITIAL.
    RETURN.
  ENDIF.

  " 2: Prefix GP if needed
  IF LV_TRANPROG CP 'GP*'.
    PV_PROG = LV_TRANPROG.
  ELSE.
    CONCATENATE 'GP' LV_TRANPROG INTO PV_PROG.
  ENDIF.

  " 3: Validate that this program contains METHOD EXECUTE
  READ REPORT PV_PROG INTO LT_SRC STATE 'A'.
  LOOP AT LT_SRC INTO LV_SRC_LINE.
    IF LV_SRC_LINE CS 'METHOD EXECUTE'.
      RETURN.
    ENDIF.
  ENDLOOP.

  " 4: Try INACTIVE version
  REFRESH LT_SRC.
  READ REPORT PV_PROG INTO LT_SRC STATE 'I'.
  LOOP AT LT_SRC INTO LV_SRC_LINE.
    IF LV_SRC_LINE CS 'METHOD EXECUTE'.
      RETURN.
    ENDIF.
  ENDLOOP.

  " 5) As a last fallback: search TRDIR for any GP* program that really has METHOD EXECUTE

  SELECT NAME
    INTO LV_NAME
    FROM TRDIR
    WHERE NAME LIKE 'GP%'        " CP not allowed in Open SQL; use LIKE
*    AND subc = '1'            " optional: restrict to type 1 (program)
*    AND genflag = ''          " optional: exclude generated templates if present in your system
  .

    " try ACTIVE
    REFRESH LT_SRC.
    READ REPORT LV_NAME INTO LT_SRC STATE 'A'.
    LOOP AT LT_SRC INTO LV_SRC_LINE.
      IF LV_SRC_LINE CS 'METHOD EXECUTE'.
        PV_PROG = LV_NAME.
        EXIT. " exit LOOP
      ENDIF.
    ENDLOOP.
    IF PV_PROG IS NOT INITIAL.
      EXIT. " exit SELECT loop as well
    ENDIF.

    " try INACTIVE
    REFRESH LT_SRC.
    READ REPORT LV_NAME INTO LT_SRC STATE 'I'.
    LOOP AT LT_SRC INTO LV_SRC_LINE.
      IF LV_SRC_LINE CS 'METHOD EXECUTE'.
        PV_PROG = LV_NAME.
        EXIT. " exit LOOP
      ENDIF.
    ENDLOOP.
    IF PV_PROG IS NOT INITIAL.
      EXIT. " exit SELECT loop
    ENDIF.

  ENDSELECT.
ENDFORM.

FORM FETCH_REPORT_SOURCE TABLES PT_SRC
                         USING  PV_PROG TYPE PROGNAME
                         CHANGING PV_OK TYPE C.
  DATA: LV_EXISTS TYPE TRDIR-NAME.

  CLEAR PV_OK.
  REFRESH PT_SRC.

  " Does program exist at all?
  SELECT SINGLE NAME INTO LV_EXISTS
    FROM TRDIR
    WHERE NAME = PV_PROG.
  IF SY-SUBRC <> 0.
    RETURN. " no such program
  ENDIF.

  " Try ACTIVE version
  READ REPORT PV_PROG INTO PT_SRC STATE 'A'.
  IF SY-SUBRC = 0 AND PT_SRC IS NOT INITIAL.
    PV_OK = 'X'.
    RETURN.
  ENDIF.

  " Try INACTIVE version (GP reports often stored here)
  READ REPORT PV_PROG INTO PT_SRC STATE 'I'.
  IF SY-SUBRC = 0 AND PT_SRC IS NOT INITIAL.
    PV_OK = 'X'.
    RETURN.
  ENDIF.

  " Last fallback: read without specifying state (some systems behave differently)
  READ REPORT PV_PROG INTO PT_SRC.
  IF SY-SUBRC = 0 AND PT_SRC IS NOT INITIAL.
    PV_OK = 'X'.
  ENDIF.

ENDFORM.

FORM PATH_INIT.

  DATA: LV_BASE      TYPE STRING,
        LV_LAST_CHAR TYPE C,
        LT_CHARS     TYPE STANDARD TABLE OF CHAR1 WITH DEFAULT KEY,
        LV_CHAR      TYPE C,
        LV_LEN       TYPE I,
        LV_PATH_OK   TYPE C.
  DATA: LV_LEN2   TYPE I,
        LV_OFFSET TYPE I,
        LV_TMP    TYPE STRING.

  " Choose base directory
  IF P_PATH IS INITIAL.
    PERFORM BUILD_OUT_DIR USING GV_CTX CHANGING LV_BASE.
  ELSE.
    LV_BASE = P_PATH.
  ENDIF.

  " Ensure trailing slash WITHOUT using string offsets
  IF LV_BASE IS NOT INITIAL.

    REFRESH LT_CHARS.
*    CALL FUNCTION 'STRING_TO_TABLE'
*      EXPORTING
*        STRING   = LV_BASE
*      TABLES
*        CHAR_TAB = LT_CHARS.

    LV_LEN2 = STRLEN( LV_BASE ).
    IF LV_LEN2 > 0.
      LV_OFFSET = LV_LEN2 - 1.
      LV_TMP = LV_BASE+LV_OFFSET(1).   " dynamic offset on STRING is OK in 7.50
      LV_LAST_CHAR = LV_TMP.
    ENDIF.

    IF LV_LAST_CHAR <> '/' AND LV_LAST_CHAR <> '\'.
      CONCATENATE LV_BASE '/' INTO LV_BASE.
    ENDIF.

    DESCRIBE TABLE LT_CHARS LINES LV_LEN.

    IF LV_LEN > 0.
      READ TABLE LT_CHARS INTO LV_LAST_CHAR INDEX LV_LEN.
    ENDIF.

    IF LV_LAST_CHAR <> '/' AND LV_LAST_CHAR <> '\'.
      CONCATENATE LV_BASE '/' INTO LV_BASE.
    ENDIF.

  ENDIF.

  " Now build root directories
  CONCATENATE LV_BASE GV_CTX '/' INTO GV_ROOT_DIR.
  PERFORM ENSURE_DIR_SMART USING GV_ROOT_DIR.

  CONCATENATE GV_ROOT_DIR 'Flow/' INTO GV_FLOW_DIR.
  PERFORM ENSURE_DIR_SMART USING GV_FLOW_DIR.

  CONCATENATE GV_ROOT_DIR 'DDL/' INTO GV_DDL_DIR.
  PERFORM ENSURE_DIR_SMART USING GV_DDL_DIR.

  PERFORM ENSURE_DDL_SUBDIRS USING GV_DDL_DIR.

  CONCATENATE GV_ROOT_DIR 'Transformations/' INTO GV_TRANS_DIR.
  PERFORM ENSURE_DIR_SMART USING GV_TRANS_DIR.

FORM PRINT_SUBTREE TABLES  PT_OUT
                   USING   PV_TARGET
                           PV_PREFIX.
  DATA: LT_CHILDREN    TYPE STANDARD TABLE OF TY_FLOW,
        LS_F           TYPE TY_FLOW,
        LV_I           TYPE I,
        LV_N           TYPE I,
        LV_BRANCH      TYPE STRING,

           INTO LV_LINE SEPARATED BY SPACE.
    APPEND LV_LINE TO PT_OUT.

    IF LS_F-SOURCE_TYPE <> 'DATASOURCE' AND LS_F-SOURCE_TYPE <> 'INFOSOURCE'.
      PERFORM PRINT_SUBTREE TABLES PT_OUT USING LS_F-SOURCE LV_NEXT_PREFIX.
    ENDIF.
  ENDLOOP.
ENDFORM.

FORM RX_SCAN_START_BODY TABLES PT_LINES
                        CHANGING PT_DET TYPE TY_DETAIL_TAB.

  DATA: LV          TYPE STRING,
        LV_TGT      TYPE STRING,
        LV_RHS      TYPE STRING,
        LV_TAB      TYPE STRING,
        LS_DET      TYPE TY_DETAIL,
        LV_CONST_SM TYPE STRING,
        LV_SRC_IOBJ TYPE STRING.
  FIELD-SYMBOLS <DET> TYPE TY_DETAIL.

  LOOP AT PT_LINES INTO LV.

    "--- 1) Assignment to target field: RESULT_FIELDS-<field> = <rhs>
    "    or E_T_* - <field> = <rhs>   (spaces/case tolerant)
    CLEAR: LV_TGT, LV_RHS.
    FIND REGEX '^\s*RESULT_FIELDS-\s*([A-Za-z0-9_]+)\s*=\s*(.+)$'
         IN LV IGNORING CASE SUBMATCHES LV_TGT LV_RHS.
    IF SY-SUBRC <> 0.
      FIND REGEX '^\s*E_T_[A-Za-z0-9_]+\s*-\s*([A-Za-z0-9_]+)\s*=\s*(.+)$'
           IN LV IGNORING CASE SUBMATCHES LV_TGT LV_RHS.
    ENDIF.

    IF SY-SUBRC = 0 AND LV_TGT IS NOT INITIAL.
      READ TABLE PT_DET ASSIGNING <DET> WITH KEY TARGET_FIELD = LV_TGT.
      IF SY-SUBRC <> 0.
        CLEAR LS_DET.
        LS_DET-TARGET_FIELD = LV_TGT.
        INSERT LS_DET INTO TABLE PT_DET ASSIGNING <DET>.
      ENDIF.

      IF <DET> IS ASSIGNED.
        <DET>-ROUTINE_FLAG = 'X'.

        " Try to capture a direct mapping: SOURCE_FIELDS-<src>
        FIND REGEX 'SOURCE_FIELDS-\s*([A-Za-z0-9_]+)'
             IN LV_RHS IGNORING CASE SUBMATCHES LV_SRC_IOBJ.
        IF SY-SUBRC = 0 AND <DET> IS ASSIGNED AND LV_SRC_IOBJ IS NOT INITIAL
           AND <DET>-SOURCE_FIELD IS INITIAL.
          <DET>-SOURCE_FIELD = LV_SRC_IOBJ.
        ENDIF.

        " 1.a Constant on RHS → pick first quoted literal
        FIND REGEX '''([^'']*)''' IN LV_RHS SUBMATCHES LV_CONST_SM.
        IF SY-SUBRC = 0 AND LV_CONST_SM IS NOT INITIAL.
          <DET>-CONSTANT_VAL = LV_CONST_SM.
        ELSE.
          " 1.b Otherwise treat as formula if it has operators/paren
          IF LV_RHS CS '+' OR LV_RHS CS '-' OR LV_RHS CS '*' OR LV_RHS CS '/' OR
             LV_RHS CS '(' OR LV_RHS CS ')'.
            <DET>-FORMULA_TEXT = LV_RHS.
          ENDIF.
        ENDIF.
      ENDIF.
    ENDIF.

    "--- 2) Master data source: SELECT ... FROM <table> (tied to last LV_TGT)
    IF LV CS 'SELECT' AND ( LV CS '/BIC/' OR LV CS '/BI0/' OR LV CS ' FROM ' ).
      " *** FIXED: no (?i), no \b — use IGNORING CASE for case-insensitive match
      FIND REGEX 'FROM\s+([A-Za-z0-9_/]+)'
           IN LV IGNORING CASE SUBMATCHES LV_TAB.
      IF SY-SUBRC = 0 AND LV_TGT IS NOT INITIAL AND LV_TAB IS NOT INITIAL.
        READ TABLE PT_DET ASSIGNING <DET> WITH KEY TARGET_FIELD = LV_TGT.
        IF SY-SUBRC <> 0.
          CLEAR LS_DET.
          LS_DET-TARGET_FIELD = LV_TGT.
          INSERT LS_DET INTO TABLE PT_DET ASSIGNING <DET>.
        ENDIF.
        IF <DET> IS ASSIGNED AND <DET>-MASTER_SOURCE IS INITIAL.
          <DET>-MASTER_SOURCE = LV_TAB.
        ENDIF.
      ENDIF.
    ELSEIF LV CS 'READ TABLE'.
      IF LV_TGT IS NOT INITIAL.
        READ TABLE PT_DET ASSIGNING <DET> WITH KEY TARGET_FIELD = LV_TGT.
        IF SY-SUBRC <> 0.
          CLEAR LS_DET.
          LS_DET-TARGET_FIELD = LV_TGT.
          INSERT LS_DET INTO TABLE PT_DET ASSIGNING <DET>.
        ENDIF.
        IF <DET> IS ASSIGNED AND <DET>-MASTER_SOURCE IS INITIAL.
          <DET>-MASTER_SOURCE = 'READ TABLE ...'.
        ENDIF.
      ENDIF.
    ENDIF.

  ENDLOOP.
ENDFORM.

FORM RX_EXTRACT_METHOD_BODY TABLES  PT_SRC
                                    PT_BODY
                            USING    PV_NAME TYPE STRING.
  DATA: LV_BEGIN TYPE I VALUE 0,
        LV_END   TYPE I VALUE 0,
        LV_LINE  TYPE STRING,
        LV_PAT1  TYPE STRING,
        LV_PAT2  TYPE STRING,
        LV_IDX   TYPE I,
        LV_MAX   TYPE I.
  REFRESH PT_BODY.
  CONCATENATE '^\s*METHOD\s+' PV_NAME '\.' INTO LV_PAT1.
  CONCATENATE '^\s*METHOD\s+IF_[A-Z0-9_]+~' PV_NAME '\.' INTO LV_PAT2.
  DESCRIBE TABLE PT_SRC LINES LV_MAX.
  LV_IDX = 1.
  WHILE LV_IDX <= LV_MAX.
    READ TABLE PT_SRC INTO LV_LINE INDEX LV_IDX.
    IF SY-SUBRC <> 0. EXIT. ENDIF.
    IF LV_BEGIN = 0.
      FIND REGEX LV_PAT1 IN LV_LINE IGNORING CASE.
      IF SY-SUBRC <> 0. FIND REGEX LV_PAT2 IN LV_LINE IGNORING CASE. ENDIF.
      IF SY-SUBRC = 0. LV_BEGIN = LV_IDX + 1. ENDIF.
    ELSE.
      FIND REGEX '^\s*ENDMETHOD\.' IN LV_LINE IGNORING CASE.
      IF SY-SUBRC = 0. LV_END = LV_IDX - 1. EXIT. ENDIF.
    ENDIF.
    LV_IDX = LV_IDX + 1.
  ENDWHILE.
  IF LV_BEGIN = 0 OR LV_END < LV_BEGIN. RETURN. ENDIF.
  LV_IDX = LV_BEGIN.
  WHILE LV_IDX <= LV_END.
    READ TABLE PT_SRC INTO LV_LINE INDEX LV_IDX.
    IF SY-SUBRC <> 0. EXIT. ENDIF.
    APPEND LV_LINE TO PT_BODY.
    LV_IDX = LV_IDX + 1.
  ENDWHILE.
ENDFORM.

    ENDIF.
  ENDLOOP.

FORM SCAN_ROUTINE_BLOCK.

  FIELD-SYMBOLS <S> TYPE STRING.

  DATA: LN   TYPE I,
        LV   TYPE STRING,
        LVU  TYPE STRING,
        C    TYPE C LENGTH 1,
        TAB  TYPE DD02L-TABNAME,
        FM   TYPE RS38L_FNAM,
        MTH  TYPE STRING,
        INC  TYPE PROGNAME,
        DEP  TYPE TY_DEP,
        I    TYPE I,
        J    TYPE I,
        K    TYPE I,
        LEN  TYPE I,
        IDX  TYPE I,
        WORK TYPE STRING.

  LN = 0.
  LOOP AT GT_SCAN_SRC ASSIGNING <S>.
    LN  = LN + 1.
    LV  = <S>.
    LVU = LV.
    TRANSLATE LVU TO UPPER CASE.
    LEN = STRLEN( LVU ).

    " --------------------
    " SELECT ... FROM <TAB>
    CLEAR TAB. IDX = -1.
    FIND FIRST OCCURRENCE OF ' FROM ' IN LVU MATCH OFFSET IDX.
    IF SY-SUBRC = 0 AND IDX GE 0.
      IDX = IDX + 6. " skip ' FROM '
      WHILE IDX < LEN AND LVU+IDX(1) = ' '. IDX = IDX + 1. ENDWHILE.
      CLEAR TAB. I = IDX.
      WHILE I < LEN.
        C = LVU+I(1).
        IF ( ( C GE 'A' AND C LE 'Z' ) OR ( C GE '0' AND C LE '9' ) OR C = '_' OR C = '/' ).
          CONCATENATE TAB C INTO TAB.
          I = I + 1.
        ELSE.
          EXIT.
        ENDIF.
      ENDWHILE.
      IF TAB IS NOT INITIAL.
        CLEAR DEP.
        DEP-TRANID    = GV_SCAN_TRANID.
        DEP-CONTAINER = GV_SCAN_CONTAINER.
        DEP-DEP_KIND  = 'TABLE'.
        DEP-DEP_NAME  = TAB.
        DEP-WHERE_TXT = 'SELECT FROM'.
        DEP-LINE_NO   = LN.
        APPEND DEP TO GT_DEP_LOG.
        INSERT TAB INTO TABLE GT_DDL_TABS.
      ENDIF.
    ENDIF.

    " UPDATE <TAB>
    CLEAR TAB. IDX = -1.
    FIND FIRST OCCURRENCE OF ' UPDATE ' IN LVU MATCH OFFSET IDX.
    IF SY-SUBRC = 0 AND IDX GE 0.
      IDX = IDX + 8.
      WHILE IDX < LEN AND LVU+IDX(1) = ' '. IDX = IDX + 1. ENDWHILE.
      CLEAR TAB. I = IDX.
      WHILE I < LEN.
        C = LVU+I(1).
        IF ( ( C GE 'A' AND C LE 'Z' ) OR ( C GE '0' AND C LE '9' ) OR C = '_' OR C = '/' ).
          CONCATENATE TAB C INTO TAB.
          I = I + 1.
        ELSE.
          EXIT.
        ENDIF.
      ENDWHILE.
      IF TAB IS NOT INITIAL.
        CLEAR DEP.
        DEP-TRANID    = GV_SCAN_TRANID.
        DEP-CONTAINER = GV_SCAN_CONTAINER.
        DEP-DEP_KIND  = 'TABLE'.
        DEP-DEP_NAME  = TAB.
        DEP-WHERE_TXT = 'UPDATE'.
        DEP-LINE_NO   = LN.
        APPEND DEP TO GT_DEP_LOG.
        INSERT TAB INTO TABLE GT_DDL_TABS.
      ENDIF.
    ENDIF.

    " INSERT [INTO] <TAB>
    CLEAR TAB. IDX = -1.
    FIND FIRST OCCURRENCE OF ' INSERT INTO ' IN LVU MATCH OFFSET IDX.
    IF SY-SUBRC = 0 AND IDX GE 0.
      IDX = IDX + 13.
    ELSE.
      FIND FIRST OCCURRENCE OF ' INSERT ' IN LVU MATCH OFFSET IDX.
      IF SY-SUBRC = 0 AND IDX GE 0.
        IDX = IDX + 8.
      ENDIF.
    ENDIF.
    IF IDX GE 0.
      WHILE IDX < LEN AND LVU+IDX(1) = ' '. IDX = IDX + 1. ENDWHILE.
      CLEAR TAB. I = IDX.
      WHILE I < LEN.
        C = LVU+I(1).
        IF ( ( C GE 'A' AND C LE 'Z' ) OR ( C GE '0' AND C LE '9' ) OR C = '_' OR C = '/' ).
          CONCATENATE TAB C INTO TAB.
          I = I + 1.
        ELSE.
          EXIT.
        ENDIF.
      ENDWHILE.
      IF TAB IS NOT INITIAL.
        CLEAR DEP.
        DEP-TRANID    = GV_SCAN_TRANID.
        DEP-CONTAINER = GV_SCAN_CONTAINER.
        DEP-DEP_KIND  = 'TABLE'.
        DEP-DEP_NAME  = TAB.
        DEP-WHERE_TXT = 'INSERT'.
        DEP-LINE_NO   = LN.
        APPEND DEP TO GT_DEP_LOG.
        INSERT TAB INTO TABLE GT_DDL_TABS.
      ENDIF.
    ENDIF.

    " DELETE [FROM] <TAB>
    CLEAR TAB. IDX = -1.
    FIND FIRST OCCURRENCE OF ' DELETE FROM ' IN LVU MATCH OFFSET IDX.
    IF SY-SUBRC = 0 AND IDX GE 0.
      IDX = IDX + 13.
    ELSE.
      FIND FIRST OCCURRENCE OF ' DELETE ' IN LVU MATCH OFFSET IDX.
      IF SY-SUBRC = 0 AND IDX GE 0.
        IDX = IDX + 8.
      ENDIF.
    ENDIF.
    IF IDX GE 0.
      WHILE IDX < LEN AND LVU+IDX(1) = ' '. IDX = IDX + 1. ENDWHILE.
      CLEAR TAB. I = IDX.
      WHILE I < LEN.
        C = LVU+I(1).
        IF ( ( C GE 'A' AND C LE 'Z' ) OR ( C GE '0' AND C LE '9' ) OR C = '_' OR C = '/' ).
          CONCATENATE TAB C INTO TAB.
          I = I + 1.
        ELSE.
          EXIT.
        ENDIF.
      ENDWHILE.
      IF TAB IS NOT INITIAL.
        CLEAR DEP.
        DEP-TRANID    = GV_SCAN_TRANID.
        DEP-CONTAINER = GV_SCAN_CONTAINER.
        DEP-DEP_KIND  = 'TABLE'.
        DEP-DEP_NAME  = TAB.
        DEP-WHERE_TXT = 'DELETE'.
        DEP-LINE_NO   = LN.
        APPEND DEP TO GT_DEP_LOG.
        INSERT TAB INTO TABLE GT_DDL_TABS.
      ENDIF.
    ENDIF.

    " JOIN <TAB> (multiple per line)
    WORK = LVU.
    DO.
      CLEAR TAB. IDX = -1.
      FIND FIRST OCCURRENCE OF ' JOIN ' IN WORK MATCH OFFSET IDX.
      IF SY-SUBRC <> 0 OR IDX LT 0.
        EXIT.
      ENDIF.
      IDX = IDX + 6.
      WHILE IDX < STRLEN( WORK ) AND WORK+IDX(1) = ' '. IDX = IDX + 1. ENDWHILE.
      CLEAR TAB. I = IDX.
      WHILE I < STRLEN( WORK ).
        C = WORK+I(1).
        IF ( ( C GE 'A' AND C LE 'Z' ) OR ( C GE '0' AND C LE '9' ) OR C = '_' OR C = '/' ).
          CONCATENATE TAB C INTO TAB.
          I = I + 1.
        ELSE.
          EXIT.
        ENDIF.
      ENDWHILE.
      IF TAB IS NOT INITIAL.
        CLEAR DEP.
        DEP-TRANID    = GV_SCAN_TRANID.
        DEP-CONTAINER = GV_SCAN_CONTAINER.
        DEP-DEP_KIND  = 'TABLE'.
        DEP-DEP_NAME  = TAB.
        DEP-WHERE_TXT = 'JOIN'.
        DEP-LINE_NO   = LN.
        APPEND DEP TO GT_DEP_LOG.
        INSERT TAB INTO TABLE GT_DDL_TABS.
        REPLACE FIRST OCCURRENCE OF TAB IN WORK WITH SPACE.
      ELSE.
        EXIT.
      ENDIF.
    ENDDO.

    " --------------------
    " CALL FUNCTION 'FM'
    CLEAR FM. IDX = -1.
    FIND FIRST OCCURRENCE OF 'CALL FUNCTION' IN LVU MATCH OFFSET IDX.
    IF SY-SUBRC = 0 AND IDX GE 0.
      I = IDX + STRLEN( 'CALL FUNCTION' ).
      J = -1.
      K = I.
      WHILE K < LEN.
        C = LVU+K(1).
        IF C = ''''. J = K. EXIT. ENDIF.
        K = K + 1.
      ENDWHILE.
      IF J GE 0.
        CLEAR FM. K = J + 1.
        WHILE K < LEN.
          C = LVU+K(1).
          IF C = ''''. EXIT. ENDIF.
          CONCATENATE FM C INTO FM.
          K = K + 1.
        ENDWHILE.
      ELSE.
        K = I.
        WHILE K < LEN AND LVU+K(1) = ' '. K = K + 1. ENDWHILE.
        CLEAR FM.
        WHILE K < LEN.
          C = LVU+K(1).
          IF ( ( C GE 'A' AND C LE 'Z' ) OR ( C GE '0' AND C LE '9' ) OR C = '_' ).
            CONCATENATE FM C INTO FM.
            K = K + 1.
          ELSE.
            EXIT.
          ENDIF.
        ENDWHILE.
      ENDIF.

      IF FM IS NOT INITIAL.
        CLEAR DEP.
        DEP-TRANID    = GV_SCAN_TRANID.
        DEP-CONTAINER = GV_SCAN_CONTAINER.
        DEP-DEP_KIND  = 'FM'.
        DEP-DEP_NAME  = FM.
        DEP-WHERE_TXT = 'CALL FUNCTION'.
        DEP-LINE_NO   = LN.
        APPEND DEP TO GT_DEP_LOG.
        INSERT FM INTO TABLE GT_FM_QUEUE.
      ENDIF.
    ENDIF.

    " --------------------
    " METHODS (class=>method and ref->method)
    CLEAR MTH. IDX = -1.
    FIND FIRST OCCURRENCE OF '=>' IN LVU MATCH OFFSET IDX.
    IF SY-SUBRC = 0 AND IDX GE 0.
      I = IDX + 2.
      WHILE I < LEN AND LVU+I(1) = ' '. I = I + 1. ENDWHILE.
      WHILE I < LEN.
        C = LVU+I(1).
        IF ( ( C GE 'A' AND C LE 'Z' ) OR ( C GE '0' AND C LE '9' ) OR C = '_' ).
          CONCATENATE MTH C INTO MTH.
          I = I + 1.
        ELSE.
          EXIT.
        ENDIF.
      ENDWHILE.
      IF MTH IS NOT INITIAL.
        CLEAR DEP.
        DEP-TRANID    = GV_SCAN_TRANID.
        DEP-CONTAINER = GV_SCAN_CONTAINER.
        DEP-DEP_KIND  = 'METHOD'.
        CONCATENATE '=>' MTH INTO DEP-DEP_NAME.
        DEP-WHERE_TXT = 'CALL METHOD'.
        DEP-LINE_NO   = LN.
        APPEND DEP TO GT_DEP_LOG.
      ENDIF.
    ENDIF.

    CLEAR MTH. IDX = -1.
    FIND FIRST OCCURRENCE OF '->' IN LVU MATCH OFFSET IDX.
    IF SY-SUBRC = 0 AND IDX GE 0.
      I = IDX + 2.
      WHILE I < LEN AND LVU+I(1) = ' '. I = I + 1. ENDWHILE.
      WHILE I < LEN.
        C = LVU+I(1).
        IF ( ( C GE 'A' AND C LE 'Z' ) OR ( C GE '0' AND C LE '9' ) OR C = '_' ).
          CONCATENATE MTH C INTO MTH.
          I = I + 1.
        ELSE.
          EXIT.
        ENDIF.
      ENDWHILE.
      IF MTH IS NOT INITIAL.
        CLEAR DEP.
        DEP-TRANID    = GV_SCAN_TRANID.
        DEP-CONTAINER = GV_SCAN_CONTAINER.
        DEP-DEP_KIND  = 'METHOD'.
        CONCATENATE '->' MTH INTO DEP-DEP_NAME.
        DEP-WHERE_TXT = 'CALL METHOD'.
        DEP-LINE_NO   = LN.
        APPEND DEP TO GT_DEP_LOG.
      ENDIF.
    ENDIF.

    " --------------------
    " INCLUDE <prog>.
    CLEAR INC. IDX = -1.
    FIND FIRST OCCURRENCE OF 'INCLUDE ' IN LVU MATCH OFFSET IDX.
    IF SY-SUBRC = 0 AND IDX GE 0.
      I = IDX + 8.
      WHILE I < LEN AND LVU+I(1) = ' '. I = I + 1. ENDWHILE.
      CLEAR INC.
      WHILE I < LEN.
        C = LVU+I(1).
        IF ( ( C GE 'A' AND C LE 'Z' ) OR ( C GE '0' AND C LE '9' ) OR C = '_' OR C = '/' ).
          CONCATENATE INC C INTO INC.
          I = I + 1.
        ELSEIF C = '.'.
          EXIT.
        ELSE.
          EXIT.
        ENDIF.
      ENDWHILE.
      IF INC IS NOT INITIAL.
        CLEAR DEP.
        DEP-TRANID    = GV_SCAN_TRANID.
        DEP-CONTAINER = GV_SCAN_CONTAINER.
        DEP-DEP_KIND  = 'INCLUDE'.
        DEP-DEP_NAME  = INC.
        DEP-WHERE_TXT = 'INCLUDE'.
        DEP-LINE_NO   = LN.
        APPEND DEP TO GT_DEP_LOG.
      ENDIF.
    ENDIF.

  ENDLOOP.

ENDFORM.

FORM RX_ANALYZE_EXECUTE USING    PV_PROG TYPE PROGNAME
                        CHANGING PT_DET  TYPE TY_DETAIL_TAB.
  DATA: LT_V2F TYPE STANDARD TABLE OF TY_V2F WITH EMPTY KEY,
        LT_V2M TYPE STANDARD TABLE OF TY_V2M WITH EMPTY KEY,
        LT_V2E TYPE STANDARD TABLE OF TY_V2E WITH EMPTY KEY.

  " -------- Declarations --------
  DATA: LT_SRC  TYPE STANDARD TABLE OF STRING,
        LV_OK   TYPE C,
        LT_EXEC TYPE STANDARD TABLE OF STRING,
        LT_FULL TYPE STANDARD TABLE OF STRING,
        LT_SEEN TYPE TY_SEEN_TAB.

  DATA: LV_LINE      TYPE STRING,
        LV_FLD       TYPE STRING,
        LV_VAR       TYPE STRING,
        LV_METH      TYPE STRING,
        LV_EXPR      TYPE STRING,
        LV_MDTAB     TYPE STRING,
        LV_D1        TYPE STRING,
        LV_D2        TYPE STRING,
        LV_MVIEW_TAB TYPE STRING,
        LV_RESVAR    TYPE STRING,
        LV_MDEXPR    TYPE STRING,
        LV_MDTAB_M   TYPE STRING,
        LS_DET       TYPE TY_DETAIL,
        LS_V2M       TYPE TY_V2M,
        LS_V2M_1     TYPE TY_V2M,
        LS_V2E       TYPE TY_V2E,
        LS_V2F_1     TYPE TY_V2F,
        LS_V2F_2     TYPE TY_V2F,
        LT_MB        TYPE STANDARD TABLE OF STRING WITH EMPTY KEY,
        LT_MFULL     TYPE STANDARD TABLE OF STRING WITH EMPTY KEY,
        LT_SEEN2     TYPE TY_SEEN_TAB,
        LV_SRC       TYPE STRING,
        LV_C         TYPE STRING,
        LV_ATTR      TYPE STRING,
        LV_TMP       TYPE STRING,
        LVLEN        TYPE I,
        LV_OFF_LAST  TYPE I,
        LV_INNER_LEN TYPE I,
        LV_TMP_CONST TYPE STRING.
  FIELD-SYMBOLS <DET> TYPE TY_DETAIL.
  FIELD-SYMBOLS <DET1> TYPE TY_DETAIL.

  DATA: IN_CALL TYPE C VALUE ' ',
        CALLBUF TYPE STRING.

  PERFORM FETCH_REPORT_SOURCE TABLES LT_SRC USING PV_PROG CHANGING LV_OK.

  " Fallback: try state-less read if LV_OK is still initial
  IF LV_OK IS INITIAL OR LT_SRC IS INITIAL.
    READ REPORT PV_PROG INTO LT_SRC. " state-less (kernel-dependent but useful in GP cases)
    IF SY-SUBRC = 0 AND LT_SRC IS NOT INITIAL.
      LV_OK = 'X'.
    ENDIF.
  ENDIF.

  " Only give up if the buffer is really empty
  IF LT_SRC IS INITIAL.
    RETURN.
  ENDIF.

  REFRESH: LT_EXEC, LT_FULL, LT_SEEN.
  PERFORM RX_EXTRACT_METHOD_BODY TABLES LT_SRC LT_EXEC USING 'execute'.
  IF LT_EXEC IS INITIAL.
    RETURN.
  ENDIF.

  PERFORM RX_EXPAND_INCLUDES TABLES LT_EXEC LT_FULL CHANGING LT_SEEN.
  IF LT_FULL IS INITIAL.
    LT_FULL = LT_EXEC.
  ENDIF.

  " -------- 1) Direct patterns in EXECUTE --------
  "     - RESULT_FIELDS-<field> = 'const' | <expr>
  "     - SELECT ... FROM <table>
  PERFORM RX_SCAN_START_BODY TABLES LT_FULL CHANGING PT_DET.

  " -------- 2) Map temp var -> target field (new GP patterns) --------
  " Build LT_V2F (VAR -> FIELD) from:
  "   a) _G_1-<field> = <_R_xxx>.
  "   b) CONVERSION_EXIT ... INPUT <_R_xxx> ... OUTPUT _G_1-<field>.
  REFRESH LT_V2F.

*  DATA : LS_V2F_2 TYPE TY_V2F.

  " 2.a  _G_1-<field> = _R_... .
  LOOP AT LT_FULL INTO LV_LINE.
    CLEAR: LV_FLD, LV_VAR.
    FIND REGEX '^\s*_G_1-\s*([A-Za-z0-9_/]+)\s*=\s*([A-Za-z0-9_\-/]+)\s*\.'
         IN LV_LINE IGNORING CASE SUBMATCHES LV_FLD LV_VAR.
    IF SY-SUBRC = 0 AND LV_FLD IS NOT INITIAL AND LV_VAR CP '_R_*'.
      CLEAR LS_V2F_2.
      LS_V2F_2-VAR   = LV_VAR.
      LS_V2F_2-FIELD = LV_FLD.
      APPEND LS_V2F_2 TO LT_V2F.
    ENDIF.
  ENDLOOP.

  " 2.b  CONVERSION_EXIT multi-line: pair INPUT _R_*  with  OUTPUT _G_1-<field>
  DATA: IN_CONV   TYPE C,
        INPUT_VAR TYPE STRING,
        OUT_FLD   TYPE STRING.

  CLEAR: IN_CONV, INPUT_VAR, OUT_FLD.

  LOOP AT LT_FULL INTO LV_LINE.

    " enter a CONVERSION_EXIT ... block (ALLOW both CALL FUNCTION and CALL '...')
    IF IN_CONV IS INITIAL.
      FIND REGEX '^\s*CALL\s+(FUNCTION\s+)?''CONVERSION_EXIT_[A-Za-z0-9_]+'
           IN LV_LINE IGNORING CASE.
      IF SY-SUBRC = 0.
        IN_CONV = 'X'.
        CLEAR: INPUT_VAR, OUT_FLD.
      ENDIF.
    ENDIF.

    IF IN_CONV = 'X'.
      " capture INPUT var
      IF INPUT_VAR IS INITIAL.
        FIND REGEX 'ID\s+''INPUT''\s+FIELD\s+([A-Za-z0-9_\-/]+)'
             IN LV_LINE IGNORING CASE SUBMATCHES LV_VAR.
        IF SY-SUBRC = 0.
          INPUT_VAR = LV_VAR.
        ENDIF.
      ENDIF.

      " capture OUTPUT target (G_1-<field>)
      IF OUT_FLD IS INITIAL.
        FIND REGEX 'ID\s+''OUTPUT''\s+FIELD\s+_G_1-([A-Za-z0-9_/]+)'
             IN LV_LINE IGNORING CASE SUBMATCHES LV_FLD.
        IF SY-SUBRC = 0.
          OUT_FLD = LV_FLD.
        ENDIF.
      ENDIF.

      " end of the CALL ... block — the period closes it
      IF LV_LINE CS '.'.
        IF INPUT_VAR CP '_R_*' AND OUT_FLD IS NOT INITIAL.
          CLEAR LS_V2F_2.
          LS_V2F_2-VAR   = INPUT_VAR.
          LS_V2F_2-FIELD = OUT_FLD.
          APPEND LS_V2F_2 TO LT_V2F.
        ENDIF.
        CLEAR: IN_CONV, INPUT_VAR, OUT_FLD.
      ENDIF.
    ENDIF.

  ENDLOOP.

  DELETE ADJACENT DUPLICATES FROM LT_V2F COMPARING VAR FIELD.

  " -------- 2.x  Enrich PT_DET with DIRECT, CONSTANT, MASTER ----------

  " 2.x.a  DIRECT: _G_1-<tgt> = <_YS_SC_1>-<src>.
  LOOP AT LT_FULL INTO LV_LINE.
    CLEAR: LV_FLD.
    FIND REGEX '^\s*_G_1-\s*([A-Za-z0-9_/]+)\s*=\s*<\s*_YS_SC_1\s*>\s*-\s*([A-Za-z0-9_/]+)\s*\.'
         IN LV_LINE IGNORING CASE SUBMATCHES LV_FLD LV_SRC.
    IF SY-SUBRC = 0 AND LV_FLD IS NOT INITIAL.
      READ TABLE PT_DET ASSIGNING <DET> WITH KEY TARGET_FIELD = LV_FLD.
      IF SY-SUBRC <> 0.
        CLEAR LS_DET. LS_DET-TARGET_FIELD = LV_FLD.
        INSERT LS_DET INTO TABLE PT_DET ASSIGNING <DET>.
      ENDIF.
      IF <DET> IS ASSIGNED.
        <DET>-SOURCE_FIELD = LV_SRC.
        <DET>-ROUTINE_FLAG = 'X'.
      ENDIF.
    ENDIF.
  ENDLOOP.

  " 2.x.b  CONSTANT: _G_1-<tgt> = 'literal'.
  LOOP AT LT_FULL INTO LV_LINE.
    CLEAR: LV_FLD.
    FIND REGEX '^\s*_G_1-\s*([A-Za-z0-9_/]+)\s*=\s*''([^'']*)''\s*\.'
         IN LV_LINE IGNORING CASE SUBMATCHES LV_FLD LV_C.
    IF SY-SUBRC = 0 AND LV_FLD IS NOT INITIAL.
      READ TABLE PT_DET ASSIGNING <DET> WITH KEY TARGET_FIELD = LV_FLD.
      IF SY-SUBRC <> 0.
        CLEAR LS_DET. LS_DET-TARGET_FIELD = LV_FLD.
        INSERT LS_DET INTO TABLE PT_DET ASSIGNING <DET>.
      ENDIF.
      IF <DET> IS ASSIGNED AND <DET>-CONSTANT_VAL IS INITIAL.
        <DET>-CONSTANT_VAL = LV_C.
        <DET>-ROUTINE_FLAG = 'X'.
      ENDIF.
    ENDIF.
  ENDLOOP.

  " Discover MVIEW table once (e.g., /BI0/MMATERIAL) to tag master source
  CLEAR LV_MVIEW_TAB.
  LOOP AT LT_SRC INTO LV_LINE.
    FIND REGEX 'MVIEW\s+TYPE\s+([A-Za-z0-9_/]+)' IN LV_LINE IGNORING CASE SUBMATCHES LV_MVIEW_TAB.
    IF SY-SUBRC = 0 AND LV_MVIEW_TAB IS NOT INITIAL.
      EXIT.
    ENDIF.
  ENDLOOP.

  " 2.x.c  MASTER: _G_1-<tgt> = _B_1-MVIEW-<attr>.
  LOOP AT LT_FULL INTO LV_LINE.
    CLEAR: LV_FLD.
    FIND REGEX '^\s*_G_1-\s*([A-Za-z0-9_/]+)\s*=\s*_B_1-MVIEW-([A-Za-z0-9_]+)\s*\.'
         IN LV_LINE IGNORING CASE SUBMATCHES LV_FLD LV_ATTR.
    IF SY-SUBRC = 0 AND LV_FLD IS NOT INITIAL.
      READ TABLE PT_DET ASSIGNING <DET> WITH KEY TARGET_FIELD = LV_FLD.
      IF SY-SUBRC <> 0.
        CLEAR LS_DET. LS_DET-TARGET_FIELD = LV_FLD.
        INSERT LS_DET INTO TABLE PT_DET ASSIGNING <DET>.
      ENDIF.
      IF <DET> IS ASSIGNED AND <DET>-MASTER_SOURCE IS INITIAL.
        IF LV_MVIEW_TAB IS INITIAL.
          <DET>-MASTER_SOURCE = 'READ TABLE ...'.
        ELSE.
          <DET>-MASTER_SOURCE = LV_MVIEW_TAB.
        ENDIF.
        <DET>-ROUTINE_FLAG = 'X'.
      ENDIF.
    ENDIF.
  ENDLOOP.
  " -------- 3) Map temp var -> FORMULA__r_s (accumulate lines until '.') --------
  CLEAR: IN_CALL, CALLBUF.
  LOOP AT LT_FULL INTO LV_LINE.
    IF IN_CALL IS INITIAL.
      " Start of a call when a line contains the token 'FORMULA__'
      IF LV_LINE CS 'FORMULA__'.
        IN_CALL = 'X'.
        CALLBUF = LV_LINE.
        IF LV_LINE CS '.'.
          IN_CALL = ' '.
        ENDIF.
      ENDIF.
    ELSE.
      CONCATENATE CALLBUF ' ' LV_LINE INTO CALLBUF.
      IF LV_LINE CS '.'.
        IN_CALL = ' '.
      ENDIF.
    ENDIF.

    IF IN_CALL IS INITIAL AND CALLBUF IS NOT INITIAL.
      CLEAR: LV_METH, LV_VAR.
      " Take method token FORMULA__r_s (no objref / no CALL keyword needed)
      FIND REGEX '(FORMULA__([0-9]+)_([0-9]+))'
           IN CALLBUF IGNORING CASE SUBMATCHES LV_METH LV_D1 LV_D2.
      " Find 'result = <var>' part
      FIND REGEX 'result\s*=\s*([-A-Za-z0-9_/]+)'
           IN CALLBUF IGNORING CASE SUBMATCHES LV_VAR.
      IF LV_METH IS NOT INITIAL AND LV_VAR IS NOT INITIAL.
        TRANSLATE LV_METH TO LOWER CASE.      " -> formula__r_s
        LS_V2M-METH = LV_METH.
        LS_V2M-VAR  = LV_VAR.
        APPEND LS_V2M TO LT_V2M.
      ENDIF.
      CLEAR CALLBUF.
    ENDIF.
  ENDLOOP.

  " -------- 4) Open each formula__r_s, read RESULT = <expr> and FROM <tab> -----
  LOOP AT LT_V2M INTO LS_V2M_1.
    REFRESH: LT_MB, LT_MFULL, LT_SEEN2.
    CLEAR: LV_RESVAR, LV_MDEXPR, LV_MDTAB_M.

    PERFORM RX_EXTRACT_METHOD_BODY TABLES LT_SRC LT_MB USING LS_V2M_1-METH.
    IF LT_MB IS INITIAL.
      CONTINUE.
    ENDIF.
    PERFORM RX_EXPAND_INCLUDES TABLES LT_MB LT_MFULL CHANGING LT_SEEN2.
    IF LT_MFULL IS INITIAL.
      LT_MFULL = LT_MB.
    ENDIF.

    " --- 4.a Determine name of temp variable assigned to RESULT ---
    CLEAR LV_RESVAR.

    LOOP AT LT_MFULL INTO LV_LINE.
      FIND REGEX '^\s*RESULT\s*=\s*([A-Za-z0-9_]+)\s*\.'
           IN LV_LINE IGNORING CASE SUBMATCHES LV_RESVAR.
      IF SY-SUBRC = 0 AND LV_RESVAR IS NOT INITIAL.
        EXIT.
      ENDIF.
    ENDLOOP.

    " --- 4.b Find the expression used to calculate that temp variable ---
    CLEAR LV_EXPR.

    IF LV_RESVAR IS NOT INITIAL.
      LOOP AT LT_MFULL INTO LV_LINE.
        FIND REGEX '^\s*' && LV_RESVAR && '\s*=\s*(.+)\.'
             IN LV_LINE IGNORING CASE SUBMATCHES LV_TMP.
        IF SY-SUBRC = 0 AND LV_TMP IS NOT INITIAL.
          LV_EXPR = LV_TMP.
          EXIT.
        ENDIF.
      ENDLOOP.
    ENDIF.

    " fallback if not found
    IF LV_EXPR IS INITIAL AND LV_RESVAR IS NOT INITIAL.
      LV_EXPR = LV_RESVAR.  " same as old behavior, but real case should not happen
    ENDIF.


    CLEAR LV_MDTAB.
    LOOP AT LT_MFULL INTO LV_LINE.
      FIND REGEX 'FROM\s+([A-Za-z0-9_/]+)'
           IN LV_LINE IGNORING CASE SUBMATCHES LV_MDTAB.
      IF SY-SUBRC = 0 AND LV_MDTAB IS NOT INITIAL.
        EXIT.
      ENDIF.
    ENDLOOP.

    LS_V2E-VAR   = LS_V2M_1-VAR.
    LS_V2E-EXPR  = LV_EXPR.
    LS_V2E-MDTAB = LV_MDTAB.
    APPEND LS_V2E TO LT_V2E.
  ENDLOOP.

  " -------- 5) Merge onto PT_DET by target field ----------------------
  LOOP AT LT_V2F INTO LS_V2F_1.
    READ TABLE LT_V2E INTO LS_V2E WITH KEY VAR = LS_V2F_1-VAR.
    IF SY-SUBRC <> 0.
      CONTINUE.
    ENDIF.


    READ TABLE PT_DET ASSIGNING <DET1> WITH KEY TARGET_FIELD = LS_V2F_1-FIELD.
    IF SY-SUBRC <> 0.
*      DATA LS_DET TYPE TY_DETAIL.
      CLEAR LS_DET.
      LS_DET-TARGET_FIELD = LS_V2F_1-FIELD.
      INSERT LS_DET INTO TABLE PT_DET ASSIGNING <DET1>.
    ENDIF.

    IF <DET1> IS ASSIGNED.
      <DET1>-ROUTINE_FLAG = 'X'.

      IF <DET1>-FORMULA_TEXT IS INITIAL AND LS_V2E-EXPR IS NOT INITIAL.
        <DET1>-FORMULA_TEXT = LS_V2E-EXPR.
      ENDIF.

      IF <DET1>-MASTER_SOURCE IS INITIAL AND LS_V2E-MDTAB IS NOT INITIAL.
        <DET1>-MASTER_SOURCE = LS_V2E-MDTAB.
      ENDIF.

      " Pure quoted literal -> constant
      IF <DET1>-CONSTANT_VAL IS INITIAL AND LS_V2E-EXPR IS NOT INITIAL.
        CLEAR: LVLEN, LV_OFF_LAST, LV_INNER_LEN, LV_TMP_CONST.
        LVLEN = STRLEN( LS_V2E-EXPR ).
        IF LVLEN GE 2.
          LV_OFF_LAST = LVLEN - 1.
          IF LS_V2E-EXPR+0(1) = '''' AND LS_V2E-EXPR+LV_OFF_LAST(1) = ''''.
            LV_INNER_LEN = LVLEN - 2.
            IF LV_INNER_LEN GT 0.
              LV_TMP_CONST = LS_V2E-EXPR+1(LV_INNER_LEN).
              <DET1>-CONSTANT_VAL = LV_TMP_CONST.
            ENDIF.
          ENDIF.
        ENDIF.
      ENDIF.
    ENDIF.
  ENDLOOP.

ENDFORM.

FORM DUMP_DETAILS USING PV_TRANID TYPE RSTRAN-TRANID.
  DATA: LT     TYPE STANDARD TABLE OF STRING WITH EMPTY KEY,
        L      TYPE STRING,
        F      TYPE STRING,
        LS_DET TYPE TY_DETAIL.

  APPEND 'TargetField;Constant;Formula;Master;Flag' TO LT.
  LOOP AT GT_DETAILS INTO LS_DET.
    CONCATENATE LS_DET-TARGET_FIELD LS_DET-CONSTANT_VAL LS_DET-FORMULA_TEXT LS_DET-MASTER_SOURCE LS_DET-ROUTINE_FLAG
           INTO L SEPARATED BY ';'.

*---------------------------------------------------------------------*
* FORM RC_DOWNLOAD_TABLE
* Writes download spec file (field list + WHERE) for SAP table export
* Note: actual row-level download via SE16N or SQVI is recommended;
* this writes the spec so the user can replicate the exact download
* in Databricks / external tool with the same WHERE clause.
*---------------------------------------------------------------------*
FORM RC_DOWNLOAD_TABLE USING PV_TABLE  TYPE DD02L-TABNAME
                              PV_WHERE   TYPE STRING
                              PV_MAX     TYPE I
                              PV_OUTDIR  TYPE STRING.
  DATA: LT_OUT    TYPE STANDARD TABLE OF STRING WITH EMPTY KEY,
        LT_FIELDS TYPE STANDARD TABLE OF DFIES  WITH EMPTY KEY,
        LS_FIELDS TYPE DFIES,
        LV_LINE   TYPE STRING,
        LV_FILE   TYPE STRING,
        LV_DIR    TYPE STRING,
        LV_LAST   TYPE C LENGTH 1,
        LV_LEN    TYPE I,
        LV_OFF    TYPE I,
        LV_OK     TYPE C,
        LV_CHECK  TYPE DD02L-TABNAME,
        LV_MSG    TYPE STRING,
        LV_CNT_C  TYPE C LENGTH 20.

  LV_DIR = PV_OUTDIR.
  LV_LEN = STRLEN( LV_DIR ).
  IF LV_LEN > 0. LV_OFF = LV_LEN - 1. LV_LAST = LV_DIR+LV_OFF(1). ENDIF.
  IF LV_LAST <> '\' AND LV_LAST <> '/'. CONCATENATE LV_DIR '\' INTO LV_DIR. ENDIF.
  PERFORM ENSURE_DIR_SMART USING LV_DIR.

  SELECT SINGLE TABNAME INTO LV_CHECK FROM DD02L WHERE TABNAME = PV_TABLE.
  IF SY-SUBRC <> 0.
    CONCATENATE 'Table not in DDIC: ' PV_TABLE INTO LV_MSG.
    MESSAGE LV_MSG TYPE 'W'. RETURN.
  ENDIF.

  CALL FUNCTION 'DDIF_FIELDINFO_GET'
    EXPORTING TABNAME = PV_TABLE LANGU = SY-LANGU ALL_TYPES = 'X'
    TABLES    DFIES_TAB = LT_FIELDS
    EXCEPTIONS OTHERS = 1.

  " Download spec
  CONCATENATE 'Table: ' PV_TABLE INTO LV_LINE. APPEND LV_LINE TO LT_OUT.
  IF PV_WHERE IS NOT INITIAL.
    CONCATENATE 'WHERE: ' PV_WHERE INTO LV_LINE. APPEND LV_LINE TO LT_OUT.
  ENDIF.
  IF PV_MAX > 0.
    WRITE PV_MAX TO LV_CNT_C LEFT-JUSTIFIED. CONDENSE LV_CNT_C NO-GAPS.
    CONCATENATE 'MaxRows: ' LV_CNT_C INTO LV_LINE. APPEND LV_LINE TO LT_OUT.
  ENDIF.

  CONCATENATE LV_DIR PV_TABLE '_download_spec.txt' INTO LV_FILE.
  PERFORM WRITE_FILE_SMART TABLES LT_OUT USING LV_FILE.

  " Field list CSV
  REFRESH LT_OUT.
  APPEND 'FieldName;DataType;Length;KeyFlag;Description' TO LT_OUT.
  LOOP AT LT_FIELDS INTO LS_FIELDS.
    CONCATENATE LS_FIELDS-FIELDNAME ';' LS_FIELDS-DATATYPE ';'
                LS_FIELDS-LENG ';' LS_FIELDS-KEYFLAG ';'
                LS_FIELDS-SCRTEXT_M INTO LV_LINE.
    APPEND LV_LINE TO LT_OUT.
  ENDLOOP.
  CONCATENATE LV_DIR PV_TABLE '_fields.csv' INTO LV_FILE.
  PERFORM WRITE_FILE_SMART TABLES LT_OUT USING LV_FILE.

  CONCATENATE 'Download spec written: ' PV_TABLE INTO LV_MSG.
  MESSAGE LV_MSG TYPE 'I'.
ENDFORM.

FORM PARSE_EDGES_TO_PAIRS CHANGING CT_PAIRS TYPE TY_PAIR_TAB.
  DATA: LV_LINE  TYPE STRING,
        LV_LEFT  TYPE STRING,
        LV_RIGHT TYPE STRING,
        LV_POS   TYPE I,
        LV_BRK   TYPE I,
        LV_SRS   TYPE I,
        LS_P     TYPE TY_PAIR.
  FIELD-SYMBOLS <L> TYPE STRING.
  REFRESH CT_PAIRS.
  LOOP AT GT_EDGES_TXT ASSIGNING <L>.
    LV_LINE = <L>.
    IF LV_LINE IS INITIAL. CONTINUE. ENDIF.
    SHIFT LV_LINE LEFT DELETING LEADING SPACE.
    IF LV_LINE IS INITIAL OR LV_LINE+0(1) = '*'. CONTINUE. ENDIF.
    CLEAR LV_POS. FIND '->' IN LV_LINE MATCH OFFSET LV_POS.
    IF SY-SUBRC <> 0. CONTINUE. ENDIF.
    IF LV_POS > 0. LV_LEFT = LV_LINE(LV_POS). ELSE. CONTINUE. ENDIF.
    LV_SRS = LV_POS + 2.
    IF LV_SRS >= STRLEN( LV_LINE ). CONTINUE. ENDIF.
    LV_RIGHT = LV_LINE+LV_SRS.
    CLEAR LV_BRK. FIND '[' IN LV_RIGHT MATCH OFFSET LV_BRK.
    IF SY-SUBRC = 0 AND LV_BRK > 0. LV_RIGHT = LV_RIGHT(LV_BRK). ENDIF.
    SHIFT LV_LEFT  RIGHT DELETING TRAILING SPACE.
    SHIFT LV_LEFT  LEFT  DELETING LEADING  SPACE.
    SHIFT LV_RIGHT RIGHT DELETING TRAILING SPACE.
    SHIFT LV_RIGHT LEFT  DELETING LEADING  SPACE.
    IF LV_LEFT IS INITIAL OR LV_RIGHT IS INITIAL. CONTINUE. ENDIF.
    CLEAR LS_P. LS_P-PARENT = LV_LEFT. LS_P-CHILD = LV_RIGHT.
    APPEND LS_P TO CT_PAIRS.
  ENDLOOP.
  SORT CT_PAIRS BY PARENT CHILD.
  DELETE ADJACENT DUPLICATES FROM CT_PAIRS COMPARING PARENT CHILD.
ENDFORM.

FORM EMIT_JSON_NODE USING    P_FM     TYPE RS38L_FNAM
                             P_INDENT TYPE I
                             PT_PAIRS TYPE TY_PAIR_TAB
                    CHANGING PT_JSON  TYPE TY_STRINGTAB.
  DATA: LV_IND  TYPE STRING, LV_IND2 TYPE STRING,
        LV_LINE TYPE STRING, LT_KIDS TYPE TY_FM_TAB,
        LV_CHILD TYPE RS38L_FNAM, LV_COUNT TYPE I,
        LV_I TYPE I, LV_IND_NEXT TYPE I.
  PERFORM BUILD_INDENT USING P_INDENT CHANGING LV_IND.
  PERFORM BUILD_INDENT USING P_INDENT CHANGING LV_IND2.
  CONCATENATE LV_IND2 '  ' INTO LV_IND2.
  CONCATENATE LV_IND GC_LBRACE INTO LV_LINE. APPEND LV_LINE TO PT_JSON.
  CONCATENATE LV_IND2 GC_QUOTE 'fm' GC_QUOTE GC_COLON GC_QUOTE P_FM GC_QUOTE
    INTO LV_LINE. APPEND LV_LINE TO PT_JSON.
  CONCATENATE LV_IND2 GC_QUOTE 'children' GC_QUOTE GC_COLON GC_LBRACK
    INTO LV_LINE. APPEND LV_LINE TO PT_JSON.
  REFRESH LT_KIDS.
  PERFORM BUILD_CHILD_LIST USING P_FM PT_PAIRS CHANGING LT_KIDS.
  DESCRIBE TABLE LT_KIDS LINES LV_COUNT. CLEAR LV_I.
  LOOP AT LT_KIDS INTO LV_CHILD.
    ADD 1 TO LV_I.
    LV_IND_NEXT = P_INDENT + 4.
    PERFORM EMIT_JSON_NODE USING LV_CHILD LV_IND_NEXT PT_PAIRS CHANGING PT_JSON.
    IF LV_I < LV_COUNT.
      CONCATENATE LV_IND2 GC_COMMA INTO LV_LINE. APPEND LV_LINE TO PT_JSON.
    ENDIF.
  ENDLOOP.
  CONCATENATE LV_IND2 GC_RBRACK INTO LV_LINE. APPEND LV_LINE TO PT_JSON.
  CONCATENATE LV_IND GC_RBRACE INTO LV_LINE. APPEND LV_LINE TO PT_JSON.
ENDFORM.

FORM BUILD_JSON_TREE USING    P_ROOT  TYPE RS38L_FNAM
                     CHANGING PT_JSON TYPE TY_STRINGTAB.
  DATA: LT_PAIRS TYPE TY_PAIR_TAB, LT_KIDS TYPE TY_FM_TAB,
        LV_LINE TYPE STRING, LV_CNT TYPE C LENGTH 20, LV_MAX TYPE C LENGTH 20,

    INTO LV_LINE. APPEND LV_LINE TO PT_JSON.
  IF GV_LIMIT_REACHED = 'X'.
    WRITE GV_FM_COUNT TO LV_CNT. WRITE GV_MAXFM TO LV_MAX.
    CONCATENATE LV_IND2 GC_QUOTE 'limit_reached' GC_QUOTE GC_COLON GC_QUOTE 'true' GC_QUOTE
      INTO LV_LINE. APPEND LV_LINE TO PT_JSON.
  ENDIF.
  CONCATENATE LV_IND2 GC_QUOTE 'children' GC_QUOTE GC_COLON GC_LBRACK
    INTO LV_LINE. APPEND LV_LINE TO PT_JSON.
  PERFORM BUILD_CHILD_LIST USING P_ROOT LT_PAIRS CHANGING LT_KIDS.
  DESCRIBE TABLE LT_KIDS LINES LV_COUNT. CLEAR LV_I.
  LOOP AT LT_KIDS INTO LV_CHILD.
    ADD 1 TO LV_I.
    PERFORM EMIT_JSON_NODE USING LV_CHILD 4 LT_PAIRS CHANGING PT_JSON.
    IF LV_I < LV_COUNT.
      CONCATENATE LV_IND2 GC_COMMA INTO LV_LINE. APPEND LV_LINE TO PT_JSON.
    ENDIF.
  ENDLOOP.
  CONCATENATE LV_IND2 GC_RBRACK INTO LV_LINE. APPEND LV_LINE TO PT_JSON.
  APPEND GC_RBRACE TO PT_JSON.
ENDFORM.

*---------------------------------------------------------------------*
* FORM WRITE_FILE_APPSERVER
* Server-side write using OPEN DATASET / TRANSFER / CLOSE DATASET.
* PV_FILE must be an absolute AL11 path.
*---------------------------------------------------------------------*
FORM WRITE_FILE_APPSERVER TABLES PT_TAB
                          USING  PV_FILE TYPE STRING.
  DATA: LV_LINE TYPE STRING,
        LV_MSG  TYPE STRING,
        LV_SUBRC_TXT TYPE C LENGTH 20.

  OPEN DATASET PV_FILE FOR OUTPUT IN TEXT MODE ENCODING DEFAULT.
  IF SY-SUBRC <> 0.
    WRITE SY-SUBRC TO LV_SUBRC_TXT.
    CONCATENATE 'OPEN DATASET failed (subrc=' LV_SUBRC_TXT '): ' PV_FILE
           INTO LV_MSG.
    " Log to missing table rather than dump in batch
    APPEND LV_MSG TO GT_MISSING.
    RETURN.
  ENDIF.

  LOOP AT PT_TAB INTO LV_LINE.
    TRANSFER LV_LINE TO PV_FILE.
  ENDLOOP.

  CLOSE DATASET PV_FILE.
ENDFORM.

*---------------------------------------------------------------------*
* FORM WRITE_PROGRESS_LOG
* Appends a timestamped progress entry to both GT_CHECKPOINT and
* to a progress log file on the appserver (batch) or ignores (dialog).
*---------------------------------------------------------------------*
FORM WRITE_PROGRESS_LOG USING PV_PROV   TYPE STRING
                               PV_STATUS TYPE STRING.
  DATA: LS_CK   TYPE TY_CHECKPOINT,
        LT_LINE TYPE STANDARD TABLE OF STRING WITH EMPTY KEY,
        LV_LINE TYPE STRING,
        LV_TS   TYPE C LENGTH 14.

  CONCATENATE SY-DATUM SY-UZEIT INTO LV_TS.

  CLEAR LS_CK.
  LS_CK-PROVIDER  = PV_PROV.
  LS_CK-STATUS    = PV_STATUS.
  LS_CK-TIMESTAMP = LV_TS.

  " Update or insert
  DATA: LS_EXIST TYPE TY_CHECKPOINT.
  READ TABLE GT_CHECKPOINT INTO LS_EXIST
    WITH KEY PROVIDER = PV_PROV.
  IF SY-SUBRC = 0.
    MODIFY GT_CHECKPOINT FROM LS_CK TRANSPORTING STATUS TIMESTAMP
      WHERE PROVIDER = PV_PROV.
  ELSE.
    APPEND LS_CK TO GT_CHECKPOINT.
  ENDIF.

  " Write line to console log in batch
  IF GV_BATCH = 'X'.
    CONCATENATE LV_TS ';' PV_PROV ';' PV_STATUS INTO LV_LINE.
    APPEND LV_LINE TO LT_LINE.
    IF GV_CHKPT_FILE IS NOT INITIAL.
      " Append mode via OPEN FOR APPENDING
      DATA: LV_L TYPE STRING.
      OPEN DATASET GV_CHKPT_FILE FOR APPENDING IN TEXT MODE ENCODING DEFAULT.
      IF SY-SUBRC = 0.
        LOOP AT LT_LINE INTO LV_L.
          TRANSFER LV_L TO GV_CHKPT_FILE.
        ENDLOOP.
        CLOSE DATASET GV_CHKPT_FILE.
      ENDIF.
    ENDIF.
  ENDIF.
ENDFORM.

*---------------------------------------------------------------------*
* FORM LOAD_CHECKPOINT
* Reads checkpoint file into GT_CHECKPOINT at job start.
* Skips providers already marked DONE.
*---------------------------------------------------------------------*
FORM LOAD_CHECKPOINT.
  DATA: LV_LINE TYPE STRING,        LS_OLD      TYPE TY_CHECKPOINT,
        LV_F    TYPE STRING,
        LS_CK   TYPE TY_CHECKPOINT,
        LV_TS   TYPE C LENGTH 14,
        LV_PROV TYPE C LENGTH 30,
        LV_STAT TYPE C LENGTH 10,
        LV_POS1 TYPE I,
        LV_POS2 TYPE I,
        LV_LEN  TYPE I.

  REFRESH GT_CHECKPOINT.
  IF GV_CHKPT_FILE IS INITIAL. RETURN. ENDIF.

  OPEN DATASET GV_CHKPT_FILE FOR INPUT IN TEXT MODE ENCODING DEFAULT.
  IF SY-SUBRC <> 0. RETURN. ENDIF.  " No checkpoint yet — fresh start

  DO.
    READ DATASET GV_CHKPT_FILE INTO LV_LINE.
    IF SY-SUBRC <> 0. EXIT. ENDIF.
    IF LV_LINE IS INITIAL. CONTINUE. ENDIF.

    " Format: YYYYMMDDHHMMSS;PROVIDER;STATUS
    CLEAR: LV_TS, LV_PROV, LV_STAT, LV_POS1, LV_POS2.
    FIND ';' IN LV_LINE MATCH OFFSET LV_POS1.
    IF SY-SUBRC <> 0. CONTINUE. ENDIF.
    LV_TS = SUBSTRING( VAL = LV_LINE OFF = 0 LEN = LV_POS1 ).

    LV_F = SUBSTRING( VAL = LV_LINE OFF = LV_POS1 + 1
                      LEN = STRLEN( LV_LINE ) - LV_POS1 - 1 ).
    FIND ';' IN LV_F MATCH OFFSET LV_POS2.
    IF SY-SUBRC <> 0. CONTINUE. ENDIF.
    LV_PROV = SUBSTRING( VAL = LV_F OFF = 0 LEN = LV_POS2 ).
    LV_STAT = SUBSTRING( VAL = LV_F OFF = LV_POS2 + 1
                         LEN = STRLEN( LV_F ) - LV_POS2 - 1 ).

    CONDENSE: LV_PROV NO-GAPS, LV_STAT NO-GAPS.
    IF LV_PROV IS INITIAL. CONTINUE. ENDIF.

    CLEAR LS_CK.
    LS_CK-PROVIDER  = LV_PROV.
    LS_CK-STATUS    = LV_STAT.
    LS_CK-TIMESTAMP = LV_TS.

    READ TABLE GT_CHECKPOINT INTO LS_OLD WITH KEY PROVIDER = LV_PROV.
    IF SY-SUBRC = 0.
      MODIFY GT_CHECKPOINT FROM LS_CK TRANSPORTING STATUS TIMESTAMP
        WHERE PROVIDER = LV_PROV.
    ELSE.
      APPEND LS_CK TO GT_CHECKPOINT.
    ENDIF.
  ENDDO.

  CLOSE DATASET GV_CHKPT_FILE.
ENDFORM.

*---------------------------------------------------------------------*
* FORM SAVE_CHECKPOINT
* Rewrites full checkpoint file with current state.
*---------------------------------------------------------------------*
FORM SAVE_CHECKPOINT.
  DATA: LS_CK  TYPE TY_CHECKPOINT,
        LV_LINE TYPE STRING.

  IF GV_CHKPT_FILE IS INITIAL. RETURN. ENDIF.

  OPEN DATASET GV_CHKPT_FILE FOR OUTPUT IN TEXT MODE ENCODING DEFAULT.
  IF SY-SUBRC <> 0. RETURN. ENDIF.

  LOOP AT GT_CHECKPOINT INTO LS_CK.
    CONCATENATE LS_CK-TIMESTAMP ';' LS_CK-PROVIDER ';' LS_CK-STATUS
           INTO LV_LINE.
    TRANSFER LV_LINE TO GV_CHKPT_FILE.
  ENDLOOP.

  CLOSE DATASET GV_CHKPT_FILE.
ENDFORM.

*---------------------------------------------------------------------*
* FORM CHECK_CHECKPOINT
* Returns 'DONE' if provider was successfully completed in prior run.
*---------------------------------------------------------------------*
FORM CHECK_CHECKPOINT USING    PV_PROV   TYPE STRING
                      CHANGING PV_STATUS TYPE STRING.
  DATA: LS_CK TYPE TY_CHECKPOINT.
  CLEAR PV_STATUS.
  READ TABLE GT_CHECKPOINT INTO LS_CK WITH KEY PROVIDER = PV_PROV.
  IF SY-SUBRC = 0.
    PV_STATUS = LS_CK-STATUS.
  ENDIF.
ENDFORM.

*---------------------------------------------------------------------*
* FORM TRIM_FM_CACHE
* Keeps GT_FORM_CACHE at or below GC_FM_CACHE_CAP entries.
* Removes oldest entries (from table start) when cap exceeded.
* Prevents unbounded cache growth during large InfoArea runs.
*---------------------------------------------------------------------*
FORM TRIM_FM_CACHE.
  DATA: LV_CNT  TYPE I,
        LV_TRIM TYPE I,
        LV_I    TYPE I.

  LV_CNT = LINES( GT_FORM_CACHE ).
  IF LV_CNT > GC_FM_CACHE_CAP.
    LV_TRIM = LV_CNT - GC_FM_CACHE_CAP.
    LV_I = 0.
    WHILE LV_I < LV_TRIM.
      DELETE GT_FORM_CACHE INDEX 1.
      LV_I = LV_I + 1.
    ENDWHILE.
  ENDIF.
ENDFORM.

* Builds an appserver file path: P_AL11 / subdir / filename

                                    PV_SUBDIR TYPE STRING
                                    PV_FILE   TYPE STRING
                           CHANGING PV_PATH   TYPE STRING.
  DATA: LV_BASE TYPE STRING,
        LV_LEN  TYPE I,
        LV_LAST TYPE C LENGTH 1,

  LV_BASE = PV_BASE.
  LV_LEN  = STRLEN( LV_BASE ).

  IF PV_SUBDIR IS NOT INITIAL.
    CONCATENATE LV_BASE PV_SUBDIR '/' INTO LV_BASE.
  ENDIF.

  CONCATENATE LV_BASE PV_FILE INTO PV_PATH.
ENDFORM.
