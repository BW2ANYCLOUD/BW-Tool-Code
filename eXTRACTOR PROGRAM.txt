*---------------------------------------------------------------------*
* Start-of-selection  (v13.1+ controller)                             *
*---------------------------------------------------------------------*
START-OF-SELECTION.

  " 0) Clear run state
  REFRESH: GT_FLOW, GT_TRAN, GT_SEEN, GT_DETAILS.
  CLEAR:   GV_OUTDIR, GV_MSG.

  " 1) Derive effective output path from radio button
  DATA GV_OUTPATH TYPE STRING.
  IF RB_FE = 'X'.
    GV_OUTPATH = P_FEDIR.
    " Map back to P_OUTDIR which the rest of the code uses for frontend writes
    " (non-initial P_OUTDIR triggers GUI_DOWNLOAD)
    P_OUTDIR = P_FEDIR.
  ELSE.
    GV_OUTPATH = P_APPDIR.
    CLEAR P_OUTDIR.  " blank = appserver OPEN DATASET
  ENDIF.

  " Validate: at least one input required
  IF SO_PROV[] IS INITIAL AND SO_MD[] IS INITIAL.
    MESSAGE 'Enter at least one provider or master data InfoObject.' TYPE 'E'.
  ENDIF.

  " 2) Context label = first low value from the active range table
  DATA LV_CTX_CAND TYPE CHAR30.
  IF RB_PROV = 'X'.
    READ TABLE SO_PROV INDEX 1 INTO DATA(LS_SR).
    IF SY-SUBRC = 0. LV_CTX_CAND = LS_SR-LOW. ENDIF.
  ELSE.
    READ TABLE SO_MD INDEX 1 INTO DATA(LS_MD).
    IF SY-SUBRC = 0. LV_CTX_CAND = LS_MD-LOW. ENDIF.
  ENDIF.
  CONDENSE LV_CTX_CAND NO-GAPS.
  TRANSLATE LV_CTX_CAND TO UPPER CASE.
  GV_CTX = LV_CTX_CAND.

  PERFORM INIT_STOPWORDS.
  PERFORM BUILD_OUT_DIR USING GV_CTX CHANGING GV_OUTDIR.
  PERFORM BUILD_PROVIDER_ROOT.
  PERFORM BUILD_SUBFOLDERS.

  " 3) Preload ALL active transformations once
  SELECT * FROM RSTRAN INTO TABLE GT_TRAN WHERE OBJVERS = 'A'.

  " 4a) Transactional provider traversal
  IF RB_PROV = 'X'.
    DATA LV_IP TYPE CHAR30.
    LOOP AT SO_PROV INTO DATA(LS_PROV) WHERE SIGN = 'I' AND OPTION = 'EQ'.
      LV_IP = LS_PROV-LOW. CONDENSE LV_IP NO-GAPS.
      TRANSLATE LV_IP TO UPPER CASE.
      IF LV_IP IS NOT INITIAL.
        PERFORM START_FROM_INPUT USING LV_IP.
      ENDIF.
    ENDLOOP.
    " Also handle BT (between) and CP (pattern) ranges
    LOOP AT SO_PROV INTO LS_PROV WHERE SIGN = 'I'
                                   AND OPTION <> 'EQ'.
      " For pattern/range: collect matching DSOs
      DATA LT_DSO_R TYPE STANDARD TABLE OF RSDODSO-ODSOBJECT WITH EMPTY KEY.
      SELECT ODSOBJECT INTO TABLE LT_DSO_R FROM RSDODSO
        WHERE ODSOBJECT IN SO_PROV AND OBJVERS = 'A'.
      LOOP AT LT_DSO_R INTO DATA(LV_DSO_R).
        DATA LV_P TYPE CHAR30. LV_P = LV_DSO_R.
        PERFORM START_FROM_INPUT USING LV_P.
      ENDLOOP.
      EXIT. " one range-expansion pass is enough
    ENDLOOP.
  ENDIF.

  " 4b) Master Data InfoObject traversal
  IF RB_MD = 'X'.
    DATA LV_MD_IO TYPE CHAR30.
    LOOP AT SO_MD INTO DATA(LS_MDIN) WHERE SIGN = 'I' AND OPTION = 'EQ'.
      LV_MD_IO = LS_MDIN-LOW. CONDENSE LV_MD_IO NO-GAPS.
      TRANSLATE LV_MD_IO TO UPPER CASE.
      IF LV_MD_IO IS NOT INITIAL.
        PERFORM PROCESS_MD_INFOOBJECT USING LV_MD_IO.
      ENDIF.
    ENDLOOP.
  ENDIF.

  " 5) Rebuild output dirs (context may have changed)
  PERFORM BUILD_OUT_DIR USING GV_CTX CHANGING GV_OUTDIR.
  PERFORM BUILD_PROVIDER_ROOT.
  PERFORM BUILD_SUBFOLDERS.

  " 5) Optional outputs based on lineage (gt_flow)
  IF P_FLOW = 'X'.
    PERFORM WRITE_FLOW_CSV.
  ENDIF.

  IF P_LOAD = 'X'.
    PERFORM COLLECT_ACTIVITY_PROVIDERS.
    PERFORM RESOLVE_LAST_LOAD_FOR_ALL.
    PERFORM BUILD_INACTIVE_LOG.
    PERFORM WRITE_LOAD_ACTIVITY.
    PERFORM WRITE_INACTIVE_LOG.
  ENDIF.

  IF P_JSON = 'X'.
    PERFORM WRITE_FLOW_JSON.
    PERFORM WRITE_FLOW_MERMAID.
    PERFORM WRITE_FLOW_MERMAID_GROUPED.   " NEW grouped version
  ENDIF.

*  IF P_TREE = 'X'.
*    PERFORM WRITE_TREE_ASCII.
*  ENDIF.

  " 6) Export per-TRANID only if requested (mapping/routines)
  IF P_MAP = 'X' OR P_ROUT = 'X'.
    PERFORM EXPORT_ALL_TRANSFORMATIONS.
  ENDIF.

* DDL files (DSO/PSA/Cube)
  IF P_DDL = 'X'.
    PERFORM EXPORT_DDL_FILES.
  ENDIF.

  " Direct Update DSO extraction
  PERFORM EXTRACT_DIRECT_UPDATE_DSOS.

  " InfoSet extraction
  PERFORM EXTRACT_INFOSETS.

  " MultiProvider extraction
  PERFORM EXTRACT_MULTIPROVIDERS.

  " BEx Query extraction (auto-detect always on; P_BEX adds full provider extract)
  PERFORM EXTRACT_BEX_QUERIES.

  " 7) Final message (actual write root)
  DATA LV_LOC TYPE STRING.
  IF P_OUTDIR IS INITIAL.
    LV_LOC = GV_OUTDIR.
  ELSE.
    LV_LOC = P_OUTDIR.
  ENDIF.

  IF P_ROUT = 'X'.

    "1) Make sure all transformations have been scanned at this point
    PERFORM CLEAN_DEP_LOG.              " <— cleans GT_DEP_LOG (DDIC/TRDIR validation, stop-words, method prettify)

    " 2) Per-TRANS logs from the cleaned set
    PERFORM WRITE_ALL_TRANS_LOGS.       " <— new: writes <TRANID>_log.txt + Deps_Index.txt per TRANID

    "  produce files / ALV from the cleaned list
    PERFORM WRITE_GLOBAL_DEP_LOG.     " .../Transformations/Dependencies_Log.txt
    PERFORM WRITE_DEPS_DDL.           " .../DDL/Deps/*.txt (only real tables)
    PERFORM SHOW_DEPENDENCIES_ALV.    " optional: readable ALV
    " Summary (uses final counters, MD list, flags)
*    PERFORM BUILD_SUMMARY.
  ENDIF.

  " ALWAYS generate summary
  PERFORM BUILD_SUMMARY.
  PERFORM WRITE_SUMMARY_README.

  CONCATENATE 'Extraction finished. Files under' LV_LOC INTO GV_MSG SEPARATED BY SPACE.
  MESSAGE GV_MSG TYPE 'S'.

FORM START_FROM_INPUT USING PV_INPUT TYPE CHAR30.
  DATA LV_TYPE TYPE STRING.
  PERFORM GET_PROVIDER_TYPE USING PV_INPUT '' CHANGING LV_TYPE.

  IF LV_TYPE = 'MULTIPROVIDER'.
    DATA LT_PARTS TYPE STANDARD TABLE OF RSDCUBEMULTI-PARTCUBE.
    DATA LV_PART  TYPE RSDCUBEMULTI-PARTCUBE.
    SELECT PARTCUBE FROM RSDCUBEMULTI INTO TABLE LT_PARTS
    WHERE INFOCUBE = PV_INPUT AND OBJVERS = 'A'.
    LOOP AT LT_PARTS INTO LV_PART.
      PERFORM PROCESS_PROVIDER USING LV_PART.
    ENDLOOP.
  ELSE.
    PERFORM PROCESS_PROVIDER USING PV_INPUT.
  ENDIF.
ENDFORM.

FORM PROCESS_PROVIDER USING PV_PROVIDER TYPE CHAR30.

  DATA: LS_TRAN     TYPE RSTRAN,
        LS_FLOW     TYPE TY_FLOW,
        LV_SRC      TYPE CHAR30,
        LV_SRC_DISP TYPE RSDS-DATASOURCE,
        LV_DS_PLAIN TYPE RSDS-DATASOURCE,
        LV_PROGRAM  TYPE PROGNAME.

  " Reset runtime clock to prevent TIME_OUT on large flows
  SET RUN TIME CLOCK RESOLUTION LOW.

  " visited guard
  READ TABLE GT_SEEN WITH KEY TABLE_LINE = PV_PROVIDER TRANSPORTING NO FIELDS.
  IF SY-SUBRC = 0.
    RETURN.
  ENDIF.
  INSERT PV_PROVIDER INTO TABLE GT_SEEN.

  LOOP AT GT_TRAN INTO LS_TRAN WHERE TARGETNAME = PV_PROVIDER.

    CLEAR: LS_FLOW, LV_SRC, LV_SRC_DISP, LV_DS_PLAIN, LV_PROGRAM.

    " basics
    LS_FLOW-CONTEXT_PROVIDER = GV_CTX.
    LS_FLOW-TARGET           = LS_TRAN-TARGETNAME.
    LS_FLOW-TRANID           = LS_TRAN-TRANID.

    IF LS_TRAN-TRANPROG IS NOT INITIAL.
      IF LS_TRAN-TRANPROG CS 'GP'.
        LV_PROGRAM = LS_TRAN-TRANPROG.
      ELSE.
        CONCATENATE 'GP' LS_TRAN-TRANPROG INTO LV_PROGRAM.
      ENDIF.
    ENDIF.
    LS_FLOW-PROGRAM = LV_PROGRAM.

    " target type
    PERFORM GET_PROVIDER_TYPE USING LS_FLOW-TARGET LS_TRAN-TARGETTYPE CHANGING LS_FLOW-TARGET_TYPE.

    " source name/type
    LV_SRC = LS_TRAN-SOURCENAME.
    PERFORM GET_PROVIDER_TYPE USING LV_SRC LS_TRAN-SOURCETYPE CHANGING LS_FLOW-SOURCE_TYPE.

    " plain DS (strip "(LOGSYS)")
    LV_SRC_DISP = LV_SRC.
    PERFORM EXTRACT_PLAIN_DSNAME USING LV_SRC_DISP CHANGING LV_DS_PLAIN.
    IF LV_DS_PLAIN IS INITIAL.
      LV_DS_PLAIN = LV_SRC.
    ENDIF.
    LS_FLOW-SOURCE = LV_DS_PLAIN.

    "==================== DS / TS / IS (split per LOGSYS) ====================
    IF LS_FLOW-SOURCE_TYPE = 'DATASOURCE' OR LS_FLOW-SOURCE_TYPE = 'TRANSFER STRUCTURE'
                                          OR LS_FLOW-SOURCE_TYPE = 'INFOSOURCE'.

      " collect ALL logsys for this DS
      DATA LT_LS TYPE SORTED TABLE OF RSSLOGSYS WITH UNIQUE KEY TABLE_LINE.
      DATA LV_LS TYPE RSSLOGSYS.
      REFRESH LT_LS.

      " 1) Most accurate: RSDSSEG (segmenting defines PSA assignment)
      SELECT LOGSYS INTO TABLE LT_LS FROM RSDSSEG
        WHERE DATASOURCE = LV_DS_PLAIN AND OBJVERS = 'A'.
      IF LT_LS IS INITIAL.
        SELECT LOGSYS INTO TABLE LT_LS FROM RSDSSEG
          WHERE DATASOURCE = LV_DS_PLAIN AND OBJVERS = 'M'.
      ENDIF.

      " 2) Fallback: RSDS (active then modified)
      IF LT_LS IS INITIAL.
        SELECT LOGSYS INTO TABLE LT_LS FROM RSDS
          WHERE DATASOURCE = LV_DS_PLAIN AND OBJVERS = 'A'.
        IF LT_LS IS INITIAL.
          SELECT LOGSYS INTO TABLE LT_LS FROM RSDS
            WHERE DATASOURCE = LV_DS_PLAIN AND OBJVERS = 'M'.
        ENDIF.
      ENDIF.

      IF LT_LS IS INITIAL.
        " No logsys found at all → emit one row (blank logsys), PSA resolver will fallback
        CLEAR LS_FLOW-SOURCE_LOGSYS.
        PERFORM GET_TABLE_NAME
          USING    LS_FLOW-TARGET        LS_FLOW-TARGET_TYPE
                   LS_FLOW-SOURCE        LS_FLOW-SOURCE_TYPE
                   LS_FLOW-SOURCE_LOGSYS ''       " PV_T_LOGSYS unused
          CHANGING LS_FLOW-S_TABLE_NAME  LS_FLOW-T_TABLE_NAME.
        APPEND LS_FLOW TO GT_FLOW.

      ELSE.
        " one Flow row per LOGSYS (the key change)
        LOOP AT LT_LS INTO LV_LS.
          DATA LS_ROW TYPE TY_FLOW.
          LS_ROW = LS_FLOW.
          LS_ROW-SOURCE_LOGSYS = LV_LS.

          PERFORM GET_TABLE_NAME
            USING    LS_ROW-TARGET        LS_ROW-TARGET_TYPE
                     LS_ROW-SOURCE        LS_ROW-SOURCE_TYPE
                     LS_ROW-SOURCE_LOGSYS ''
            CHANGING LS_ROW-S_TABLE_NAME  LS_ROW-T_TABLE_NAME.

          APPEND LS_ROW TO GT_FLOW.
        ENDLOOP.
      ENDIF.

      " Recurse into Transfer Structure or InfoSource source
      IF LS_FLOW-SOURCE_TYPE = 'TRANSFER STRUCTURE' OR LS_FLOW-SOURCE_TYPE = 'INFOSOURCE'.
        PERFORM PROCESS_PROVIDER USING LV_SRC.
      ENDIF.
      " Plain DataSource is always the leaf – do NOT recurse
      CONTINUE.
    ENDIF.

    "==================== Non-DS sources: single row ====================
    CLEAR LS_FLOW-SOURCE_LOGSYS.
    PERFORM GET_TABLE_NAME
      USING    LS_FLOW-TARGET        LS_FLOW-TARGET_TYPE
               LS_FLOW-SOURCE        LS_FLOW-SOURCE_TYPE
               LS_FLOW-SOURCE_LOGSYS ''
      CHANGING LS_FLOW-S_TABLE_NAME  LS_FLOW-T_TABLE_NAME.
    APPEND LS_FLOW TO GT_FLOW.

    " Recurse for any non-DS, non-RSDS source
    IF LS_TRAN-SOURCETYPE <> 'RSDS'.
      PERFORM PROCESS_PROVIDER USING LV_SRC.
    ENDIF.

  ENDLOOP.

ENDFORM.

FORM GET_PROVIDER_TYPE USING    PV_PROVIDER TYPE CHAR30
                                PV_HINT     TYPE RSTRAN-SOURCETYPE
                       CHANGING PV_TYPE     TYPE STRING.
  DATA: LV_MP  TYPE RSDCUBEMULTI-INFOCUBE,
        LV_DSO TYPE RSDODSO-ODSOBJECT.

  " ── InfoSource check first (name ends _IS OR exists in RSKSFIELD) ──
  DATA LV_PROV_STR TYPE STRING.
  LV_PROV_STR = PV_PROVIDER.
  CONDENSE LV_PROV_STR NO-GAPS.
  IF LV_PROV_STR CP '*_IS'.
    PV_TYPE = 'INFOSOURCE'.
    RETURN.
  ENDIF.

  " ── Hint-driven shortcuts ────────────────────────────────────────
  IF PV_HINT = 'IS'.   PV_TYPE = 'INFOSOURCE'.        RETURN. ENDIF.
  IF PV_HINT = 'TRCS'. PV_TYPE = 'TRANSFER STRUCTURE'. RETURN. ENDIF.
  IF PV_HINT = 'ISET'. PV_TYPE = 'INFOSET'.            RETURN. ENDIF.
  IF PV_HINT = 'RSDS'. PV_TYPE = 'DATASOURCE'.         RETURN. ENDIF.

  " ── DB lookups ───────────────────────────────────────────────────
  SELECT SINGLE INFOCUBE INTO LV_MP FROM RSDCUBEMULTI
    WHERE INFOCUBE = PV_PROVIDER AND OBJVERS = 'A'.
  IF SY-SUBRC = 0. PV_TYPE = 'MULTIPROVIDER'. RETURN. ENDIF.

  DATA LS_CUBE TYPE RSDCUBE.
  SELECT SINGLE * INTO LS_CUBE FROM RSDCUBE
    WHERE INFOCUBE = PV_PROVIDER AND OBJVERS = 'A'.
  IF SY-SUBRC = 0.
    IF LS_CUBE-CUBETYPE = 'I'. PV_TYPE = 'INFOSET'.  RETURN. ENDIF.
    PV_TYPE = 'INFOCUBE'. RETURN.
  ENDIF.

  SELECT SINGLE ODSOBJECT INTO LV_DSO FROM RSDODSO
    WHERE ODSOBJECT = PV_PROVIDER AND OBJVERS = 'A'.
  IF SY-SUBRC = 0. PV_TYPE = 'DSO'. RETURN. ENDIF.

  " ── Check RSKSFIELD for InfoSource (catches names not ending _IS) ─
  DATA LV_IS_HIT TYPE RSKSFIELDNEW-ISOURCE.
  DATA LV_IS_KEY TYPE RSKSFIELDNEW-ISOURCE.
  LV_IS_KEY = PV_PROVIDER.
  SELECT SINGLE ISOURCE INTO LV_IS_HIT FROM RSKSFIELDNEW
    WHERE ISOURCE = LV_IS_KEY AND OBJVERS = 'A'.
  IF SY-SUBRC = 0. PV_TYPE = 'INFOSOURCE'. RETURN. ENDIF.

  " ── Check RSTSFIELD for Transfer Structure ───────────────────────
  DATA LV_TS_DS TYPE RSDS-DATASOURCE.
  LV_TS_DS = PV_PROVIDER.
  SELECT SINGLE TRANSTRU INTO @DATA(LV_TS_HIT) FROM RSTSFIELD
    WHERE TRANSTRU = @LV_TS_DS AND OBJVERS = 'A'.
  IF SY-SUBRC = 0. PV_TYPE = 'TRANSFER STRUCTURE'. RETURN. ENDIF.

  PV_TYPE = 'DATASOURCE'.
ENDFORM.
" Resolve physical table names for DSO active table and PSA (Transfer Structure)
FORM GET_TABLE_NAME USING    PV_TARGET      TYPE CHAR30
                             PV_TARGETTYPE  TYPE STRING
                             PV_SOURCE      TYPE CHAR50
                             PV_SOURCETYPE  TYPE STRING
                             PV_S_LOGSYS    TYPE RSSLOGSYS
                             PV_T_LOGSYS    TYPE RSSLOGSYS
                    CHANGING PV_S_TABNAME   TYPE STRING
                             PV_T_TABNAME   TYPE STRING.

  DATA: LV_DS_IN    TYPE RSDS-DATASOURCE,
        LV_DS_PLAIN TYPE RSDS-DATASOURCE,
        LV_PSA_TAB  TYPE DD02L-TABNAME.

  CLEAR: PV_S_TABNAME, PV_T_TABNAME.

  "----------------- Target side -----------------
  IF PV_TARGETTYPE = 'DSO'.
    CONCATENATE '/BIC/A' PV_TARGET '00' INTO PV_T_TABNAME.
  ELSEIF PV_TARGETTYPE = 'INFOSOURCE'.
    PV_T_TABNAME = PV_TARGET.
  ELSEIF PV_TARGETTYPE = 'MD ATTRIBUTE' OR PV_TARGETTYPE = 'MASTER DATA IOBJ'.
    " Attribute/Master data: P-table (time-independent) or Q-table (time-dep)
    " Strip leading 0 to build table name core (BW convention)
    DATA LV_IO_CORE TYPE STRING.
    LV_IO_CORE = PV_TARGET.
    CONDENSE LV_IO_CORE NO-GAPS.
    " Remove _ATTR suffix if present
    REPLACE '_ATTR' IN LV_IO_CORE WITH ''.
    " Strip leading 0 (SAP namespace: /BI0/PMATERIAL not /BI0/P0MATERIAL)
    IF LV_IO_CORE(1) = '0'. LV_IO_CORE = LV_IO_CORE+1. ENDIF.
    DATA LV_P_TRY TYPE DD02L-TABNAME.
    " Try /BI0/P (SAP-delivered) then /BIC/P (customer)
    CONCATENATE '/BI0/P' LV_IO_CORE INTO LV_P_TRY.
    SELECT SINGLE TABNAME INTO PV_T_TABNAME FROM DD02L WHERE TABNAME = LV_P_TRY.
    IF SY-SUBRC <> 0.
      CONCATENATE '/BIC/P' LV_IO_CORE INTO LV_P_TRY.
      SELECT SINGLE TABNAME INTO PV_T_TABNAME FROM DD02L WHERE TABNAME = LV_P_TRY.
    ENDIF.
    IF SY-SUBRC <> 0.
      " Fallback: Q-table (time-dependent attributes)
      CONCATENATE '/BI0/Q' LV_IO_CORE INTO LV_P_TRY.
      SELECT SINGLE TABNAME INTO PV_T_TABNAME FROM DD02L WHERE TABNAME = LV_P_TRY.
      IF SY-SUBRC <> 0.
        CONCATENATE '/BIC/Q' LV_IO_CORE INTO LV_P_TRY.
        SELECT SINGLE TABNAME INTO PV_T_TABNAME FROM DD02L WHERE TABNAME = LV_P_TRY.
      ENDIF.
    ENDIF.
  ELSEIF PV_TARGETTYPE = 'MD TEXT'.
    " Text table: /BI0/T or /BIC/T
    DATA LV_IO_TXT TYPE STRING.
    LV_IO_TXT = PV_TARGET.
    CONDENSE LV_IO_TXT NO-GAPS.
    REPLACE '_TEXT' IN LV_IO_TXT WITH ''.
    IF LV_IO_TXT(1) = '0'. LV_IO_TXT = LV_IO_TXT+1. ENDIF.
    DATA LV_T_TRY TYPE DD02L-TABNAME.
    CONCATENATE '/BI0/T' LV_IO_TXT INTO LV_T_TRY.
    SELECT SINGLE TABNAME INTO PV_T_TABNAME FROM DD02L WHERE TABNAME = LV_T_TRY.
    IF SY-SUBRC <> 0.
      CONCATENATE '/BIC/T' LV_IO_TXT INTO LV_T_TRY.
      SELECT SINGLE TABNAME INTO PV_T_TABNAME FROM DD02L WHERE TABNAME = LV_T_TRY.
    ENDIF.
  ELSEIF PV_TARGETTYPE = 'MD HIERARCHY'.
    " Hierarchy table: /BI0/H or /BIC/H
    DATA LV_IO_HIE TYPE STRING.
    LV_IO_HIE = PV_TARGET.
    CONDENSE LV_IO_HIE NO-GAPS.
    REPLACE '_HIER' IN LV_IO_HIE WITH ''.
    IF LV_IO_HIE(1) = '0'. LV_IO_HIE = LV_IO_HIE+1. ENDIF.
    DATA LV_H_TRY TYPE DD02L-TABNAME.
    CONCATENATE '/BI0/H' LV_IO_HIE INTO LV_H_TRY.
    SELECT SINGLE TABNAME INTO PV_T_TABNAME FROM DD02L WHERE TABNAME = LV_H_TRY.
    IF SY-SUBRC <> 0.
      CONCATENATE '/BIC/H' LV_IO_HIE INTO LV_H_TRY.
      SELECT SINGLE TABNAME INTO PV_T_TABNAME FROM DD02L WHERE TABNAME = LV_H_TRY.
    ENDIF.
  ELSEIF PV_TARGETTYPE = 'TRANSFER STRUCTURE' OR PV_TARGETTYPE = 'DATASOURCE'.
    " Check if actually an InfoSource misclassified
    DATA LV_IS_CHK_T TYPE RSKSFIELDNEW-ISOURCE.
    DATA LV_IS_KEY_T TYPE RSKSFIELDNEW-ISOURCE.
    LV_IS_KEY_T = PV_TARGET.
    SELECT SINGLE ISOURCE INTO LV_IS_CHK_T FROM RSKSFIELDNEW
      WHERE ISOURCE = LV_IS_KEY_T AND OBJVERS = 'A'.
    IF SY-SUBRC = 0.
      PV_T_TABNAME = PV_TARGET.
    ELSE.
      LV_DS_IN = PV_TARGET.
      PERFORM EXTRACT_PLAIN_DSNAME USING LV_DS_IN CHANGING LV_DS_PLAIN.
      CLEAR LV_PSA_TAB.
      PERFORM RESOLVE_PSA_TABLE_NAME_RSDSSEG USING LV_DS_PLAIN PV_T_LOGSYS CHANGING LV_PSA_TAB.
      IF LV_PSA_TAB IS INITIAL.
        CONCATENATE '/BIC/B' LV_DS_PLAIN '00' INTO LV_PSA_TAB.
      ENDIF.
      PV_T_TABNAME = LV_PSA_TAB.
    ENDIF.
  ENDIF.

  "----------------- Source side -----------------
  IF PV_SOURCETYPE = 'DSO'.
    CONCATENATE '/BIC/A' PV_SOURCE '00' INTO PV_S_TABNAME.
  ELSEIF PV_SOURCETYPE = 'INFOSOURCE'.
    " InfoSource: use name as-is — fields from RSKSFIELDNEW, no /BIC/B prefix
    PV_S_TABNAME = PV_SOURCE.
  ELSEIF PV_SOURCETYPE = 'TRANSFER STRUCTURE' OR PV_SOURCETYPE = 'DATASOURCE'.
    " First check: is it actually an InfoSource misclassified?
    DATA LV_IS_CHK_S  TYPE RSKSFIELDNEW-ISOURCE.
    DATA LV_IS_KEY_S  TYPE RSKSFIELDNEW-ISOURCE.
    LV_IS_KEY_S = PV_SOURCE.
    SELECT SINGLE ISOURCE INTO LV_IS_CHK_S FROM RSKSFIELDNEW
      WHERE ISOURCE = LV_IS_KEY_S AND OBJVERS = 'A'.
    IF SY-SUBRC = 0.
      " It is an InfoSource — use name as-is
      PV_S_TABNAME = PV_SOURCE.
    ELSE.
      LV_DS_IN = PV_SOURCE.
      PERFORM EXTRACT_PLAIN_DSNAME USING LV_DS_IN CHANGING LV_DS_PLAIN.

      CLEAR LV_PSA_TAB.
      PERFORM RESOLVE_PSA_TABLE_NAME_RSDSSEG USING LV_DS_PLAIN PV_S_LOGSYS CHANGING LV_PSA_TAB.

      IF LV_PSA_TAB IS INITIAL.
        CONCATENATE '/BIC/B' LV_DS_PLAIN '00' INTO LV_PSA_TAB.
      ENDIF.

      PV_S_TABNAME = LV_PSA_TAB.
    ENDIF.
  ENDIF.

ENDFORM.

* GP resolver + safe report reader                                    *
*---------------------------------------------------------------------*
" GP resolver: 'GP' + RSTRAN-TRANPROG (Active first, then M)
FORM GET_GP_PROGRAM USING PV_TRANID TYPE RSTRAN-TRANID
                    CHANGING PV_PROG TYPE PROGNAME.

  DATA LV_TRANPROG TYPE RSTRAN-TRANPROG.
  CLEAR PV_PROG.

  " 1: Read transformation TRANPROG (active)
  SELECT SINGLE TRANPROG INTO LV_TRANPROG
    FROM RSTRAN
    WHERE TRANID = PV_TRANID
      AND OBJVERS = 'A'.

  " If missing, fallback to M
  IF SY-SUBRC <> 0 OR LV_TRANPROG IS INITIAL.
    SELECT SINGLE TRANPROG INTO LV_TRANPROG
      FROM RSTRAN
      WHERE TRANID = PV_TRANID
        AND OBJVERS = 'M'.
  ENDIF.

  IF LV_TRANPROG IS INITIAL.
    RETURN.
  ENDIF.

  " 2: Prefix GP if needed
  IF LV_TRANPROG CP 'GP*'.
    PV_PROG = LV_TRANPROG.
  ELSE.
    CONCATENATE 'GP' LV_TRANPROG INTO PV_PROG.
  ENDIF.

  " 3: Validate that this program contains METHOD EXECUTE
  DATA LT_SRC TYPE STANDARD TABLE OF STRING.
  READ REPORT PV_PROG INTO LT_SRC STATE 'A'.
  LOOP AT LT_SRC INTO DATA(LINE).
    IF LINE CS 'METHOD EXECUTE'.
      RETURN.
    ENDIF.
  ENDLOOP.

  " 4: Try INACTIVE version
  CLEAR LT_SRC.
  READ REPORT PV_PROG INTO LT_SRC STATE 'I'.
  LOOP AT LT_SRC INTO LINE.
    IF LINE CS 'METHOD EXECUTE'.
      RETURN.
    ENDIF.
  ENDLOOP.

  " 5) As a last fallback: search TRDIR for any GP* program that really has METHOD EXECUTE
  DATA LV_NAME TYPE PROGNAME.

  SELECT NAME
    INTO LV_NAME
    FROM TRDIR
    WHERE NAME LIKE 'GP%'        " CP not allowed in Open SQL; use LIKE
*    AND subc = '1'            " optional: restrict to type 1 (program)
*    AND genflag = ''          " optional: exclude generated templates if present in your system
  .

    " try ACTIVE
    REFRESH LT_SRC.
    READ REPORT LV_NAME INTO LT_SRC STATE 'A'.
    LOOP AT LT_SRC INTO LINE.
      IF LINE CS 'METHOD EXECUTE'.
        PV_PROG = LV_NAME.
        EXIT. " exit LOOP
      ENDIF.
    ENDLOOP.
    IF PV_PROG IS NOT INITIAL.
      EXIT. " exit SELECT loop as well
    ENDIF.

    " try INACTIVE
    REFRESH LT_SRC.
    READ REPORT LV_NAME INTO LT_SRC STATE 'I'.
    LOOP AT LT_SRC INTO LINE.
      IF LINE CS 'METHOD EXECUTE'.
        PV_PROG = LV_NAME.
        EXIT. " exit LOOP
      ENDIF.
    ENDLOOP.
    IF PV_PROG IS NOT INITIAL.
      EXIT. " exit SELECT loop
    ENDIF.

  ENDSELECT.
ENDFORM.

FORM FETCH_REPORT_SOURCE TABLES PT_SRC
                         USING  PV_PROG TYPE PROGNAME
                         CHANGING PV_OK TYPE C.

  DATA LV_EXISTS TYPE TRDIR-NAME.
  CLEAR PV_OK.
  REFRESH PT_SRC.

  " Does program exist at all?
  SELECT SINGLE NAME INTO LV_EXISTS
    FROM TRDIR
    WHERE NAME = PV_PROG.
  IF SY-SUBRC <> 0.
    RETURN. " no such program
  ENDIF.

  " Try ACTIVE version
  READ REPORT PV_PROG INTO PT_SRC STATE 'A'.
  IF SY-SUBRC = 0 AND PT_SRC IS NOT INITIAL.
    PV_OK = 'X'.
    RETURN.
  ENDIF.

  " Try INACTIVE version (GP reports often stored here)
  READ REPORT PV_PROG INTO PT_SRC STATE 'I'.
  IF SY-SUBRC = 0 AND PT_SRC IS NOT INITIAL.
    PV_OK = 'X'.
    RETURN.
  ENDIF.

  " Last fallback: read without specifying state (some systems behave differently)
  READ REPORT PV_PROG INTO PT_SRC.
  IF SY-SUBRC = 0 AND PT_SRC IS NOT INITIAL.
    PV_OK = 'X'.
  ENDIF.

ENDFORM.

FORM EXPORT_ALL_TRANSFORMATIONS.
  DATA LT_TRANIDS TYPE SORTED TABLE OF RSTRAN-TRANID WITH UNIQUE KEY TABLE_LINE.
  DATA LS_ANY    TYPE TY_FLOW.
  DATA LV_TRANID TYPE RSTRAN-TRANID.
  DATA LV_CNT    TYPE I.

  LOOP AT GT_FLOW INTO LS_ANY.
    INSERT LS_ANY-TRANID INTO TABLE LT_TRANIDS.
  ENDLOOP.

  LV_CNT = 0.
  LOOP AT LT_TRANIDS INTO LV_TRANID.
    PERFORM EXPORT_ONE_TRAN USING LV_TRANID.
    LV_CNT = LV_CNT + 1.
    " Free scanner source every 20 transformations to reduce heap pressure
    IF LV_CNT MOD 20 = 0.
      FREE GT_SCAN_SRC.
      REFRESH GT_SCAN_SRC.
    ENDIF.
  ENDLOOP.
ENDFORM.

FORM EXPORT_ONE_TRAN USING PV_TRANID TYPE RSTRAN-TRANID.

  DATA: LV_PROG  TYPE PROGNAME,
        LT_RULES TYPE TY_MAP_TAB,
        LT_ROUT  TYPE TY_ROUTINE_TAB,
        LT_BLK   TYPE STANDARD TABLE OF STRING WITH EMPTY KEY,
        LS_R     TYPE TY_ROUTINE.

  " 1) Resolve GP program (or override)
  IF P_GPPROG IS NOT INITIAL.
    LV_PROG = P_GPPROG.
  ELSE.
    PERFORM GET_GP_PROGRAM USING PV_TRANID CHANGING LV_PROG.
  ENDIF.
  IF LV_PROG IS INITIAL.
    RETURN.
  ENDIF.

  " 2) Clear details table (GV_TRANS_DIR already set by BUILD_SUBFOLDERS)
  REFRESH GT_DETAILS.

  " 3) Mine Start Routine → fills GT_DETAILS (constants/formulas/MD)
  PERFORM RX_ANALYZE_START_ROUTINE USING LV_PROG CHANGING GT_DETAILS.

  " --- NEW: Mine EXECUTE + FORMULA__* (GP class) → enrich GT_DETAILS
  PERFORM RX_ANALYZE_EXECUTE USING LV_PROG CHANGING GT_DETAILS.

  " 4) Collect ALL routines ALWAYS (so Field routines enrich GT_DETAILS)
  REFRESH LT_ROUT.
  PERFORM RX_COLLECT_ROUTINES USING PV_TRANID LV_PROG CHANGING LT_ROUT.
  LOOP AT LT_ROUT INTO LS_R WHERE ROUTINE_TYPE = 'FIELD'.
    LT_BLK = LS_R-CODE.
    PERFORM RX_SCAN_START_BODY TABLES LT_BLK CHANGING GT_DETAILS.
  ENDLOOP.

  " 5) Write routine code files only if P_ROUT = 'X'
  IF P_ROUT = 'X'.
    PERFORM WRITE_ROUTINE_FILES USING PV_TRANID LT_ROUT.
  ENDIF.

  " 6) Collect Mapping & Write Mapping CSV
  IF P_MAP = 'X'.
    REFRESH LT_RULES.
    PERFORM COLLECT_MAPPING USING PV_TRANID CHANGING LT_RULES.
    PERFORM WRITE_MAPPING_FILE USING PV_TRANID LT_RULES.


  ENDIF.

  " 7) ALWAYS write Constants / Formulas / Master Data files
  " Ensure LT_RULES is populated even when P_MAP is off
  IF LT_RULES IS INITIAL.
    PERFORM COLLECT_MAPPING USING PV_TRANID CHANGING LT_RULES.
  ENDIF.
  PERFORM WRITE_CONST_FORMULA_FILES USING PV_TRANID LT_RULES.

  " --- DEBUG (optional): dump details to a text file
  PERFORM DUMP_DETAILS USING PV_TRANID.

  " Memory: free large locals after each transformation
  FREE LT_ROUT.
  FREE LT_RULES.
  FREE LT_BLK.
  REFRESH GT_DETAILS.
ENDFORM.

FORM BUILD_OUT_DIR USING    PV_PROVIDER TYPE CHAR30
                   CHANGING PV_DIR      TYPE STRING.
  DATA LV_PROV TYPE STRING.
  DATA LV_BASE TYPE STRING.
  DATA LV_LEN  TYPE I.
  DATA LV_OFF  TYPE I.
  DATA LV_LAST TYPE C LENGTH 1.

  LV_PROV = PV_PROVIDER.
  REPLACE ALL OCCURRENCES OF REGEX '[^A-Za-z0-9_-]+' IN LV_PROV WITH '_'.
  CONDENSE LV_PROV NO-GAPS.

  IF P_OUTDIR IS NOT INITIAL.
    LV_BASE = P_OUTDIR.
    LV_LEN  = STRLEN( LV_BASE ).
    IF LV_LEN > 0.
      LV_OFF  = LV_LEN - 1.
      LV_LAST = LV_BASE+LV_OFF(1).
    ENDIF.
    IF LV_LAST <> '\'.
      CONCATENATE LV_BASE '\' INTO LV_BASE.
    ENDIF.
    CONCATENATE LV_BASE LV_PROV '\Transformations\' INTO PV_DIR.
  ELSE.
    LV_BASE = P_APPDIR.
    LV_LEN  = STRLEN( LV_BASE ).
    IF LV_LEN > 0.
      LV_OFF  = LV_LEN - 1.
      LV_LAST = LV_BASE+LV_OFF(1).
    ENDIF.
    IF LV_LAST <> '/'.
      CONCATENATE LV_BASE '/' INTO LV_BASE.
    ENDIF.
    CONCATENATE LV_BASE LV_PROV '/Transformations/' INTO PV_DIR.
  ENDIF.
ENDFORM.

*---------------------------------------------------------------------*
* COLLECT_MAPPING – build mapping rows + source/target key flags
*   IN : PV_TRANID  (RSTRAN-TRANID)
*  OUT : PT_OUT     (TY_MAP_TAB)
* Notes:
*  - No control-break statements; explicit grouping per RULEID.
*  - All host variables escaped with @ in Open SQL.
*  - STRING -> FIELDNAME cast via LV_FIELD for GET_KEYFLAG.
*---------------------------------------------------------------------*
FORM COLLECT_MAPPING USING    PV_TRANID TYPE RSTRAN-TRANID
                     CHANGING PT_OUT    TYPE TY_MAP_TAB.

  "---- Declarations ---------------------------------------------------
  DATA: LT_FIELDS  TYPE STANDARD TABLE OF TY_FIELD,
        LT_RULES   TYPE STANDARD TABLE OF TY_RULE_HDR,
        LS_FIELD   TYPE TY_FIELD,
        LS_RULEH   TYPE TY_RULE_HDR,

        LT_SOURCES TYPE STANDARD TABLE OF STRING,
        LT_TARGETS TYPE STANDARD TABLE OF STRING,

        LV_RULE    TYPE STRING,
        LS_OUT     TYPE TY_MAP_ENRICHED,

        LV_SRC     TYPE STRING,
        LV_TGT     TYPE STRING,
        LV_SRC2    TYPE STRING,
        LV_TGT2    TYPE STRING,

        LV_SRC_TAB TYPE DD02L-TABNAME,
        LV_TGT_TAB TYPE DD02L-TABNAME,

        LS_FLOW    TYPE TY_FLOW,

        LV_FIELD   TYPE FIELDNAME,      " for GET_KEYFLAG
        LV_CURR    TYPE INT2,
        LV_PREV    TYPE INT2.

  "---- Resolve physical tables (source/target) for this transformation
  CLEAR: LV_SRC_TAB, LV_TGT_TAB.
  READ TABLE GT_FLOW INTO LS_FLOW WITH KEY TRANID = PV_TRANID.
  IF SY-SUBRC = 0.
    LV_SRC_TAB = LS_FLOW-S_TABLE_NAME.
    LV_TGT_TAB = LS_FLOW-T_TABLE_NAME.
  ENDIF.

  "---- Output buffer
  REFRESH PT_OUT.

  "--------------------------------------------------------------------
  " 1) Load field participation + rule headers (host vars escaped)
  "--------------------------------------------------------------------
  "--- Fields participating in rules
  SELECT TRANID, RULEID, PARAMTYPE, FIELDNM
    FROM RSTRANFIELD
    INTO CORRESPONDING FIELDS OF TABLE @LT_FIELDS
   WHERE TRANID = @PV_TRANID
     AND OBJVERS = 'A'.

  " Preserve presentation order (moved out of SQL)
  SORT LT_FIELDS BY RULEID PARAMTYPE FIELDNM.

  "--- Rule headers
  SELECT TRANID, RULEID, RULETYPE
    FROM RSTRANRULE
    INTO CORRESPONDING FIELDS OF TABLE @LT_RULES
   WHERE TRANID = @PV_TRANID
     AND OBJVERS = 'A'.

  " For BINARY SEARCH later
  SORT LT_RULES BY TRANID RULEID.

  IF LT_FIELDS IS INITIAL.
    RETURN.
  ENDIF.

  "--------------------------------------------------------------------
  " 2) Helper: flush current group (emit rows, compute key flags)
  "    Uses LT_SOURCES, LT_TARGETS, LV_RULE, LV_SRC_TAB, LV_TGT_TAB
  "--------------------------------------------------------------------
  DEFINE _FLUSH_GROUP.

    " A) no participants
    IF LT_SOURCES IS INITIAL AND LT_TARGETS IS INITIAL.
      " nothing to emit

    " B) targets only
    ELSEIF LT_SOURCES IS INITIAL.
      LOOP AT LT_TARGETS INTO LV_TGT.
        CLEAR LS_OUT.
        LS_OUT-SOURCE_FIELD = ''.
        LS_OUT-TARGET_FIELD = LV_TGT.
        LS_OUT-RULE_TYPE    = LV_RULE.

        PERFORM ENRICH_FROM_START USING LV_TGT CHANGING LS_OUT.

        IF LV_TGT_TAB IS NOT INITIAL AND LV_TGT IS NOT INITIAL.
          LV_FIELD = LV_TGT.
          PERFORM GET_KEYFLAG USING LV_TGT_TAB LV_FIELD
                              CHANGING LS_OUT-KEYFLAG_T.
        ENDIF.

        APPEND LS_OUT TO PT_OUT.
      ENDLOOP.

    " C) sources only
    ELSEIF LT_TARGETS IS INITIAL.
      LOOP AT LT_SOURCES INTO LV_SRC.
        CLEAR LS_OUT.
        LS_OUT-SOURCE_FIELD = LV_SRC.
        LS_OUT-TARGET_FIELD = ''.
        LS_OUT-RULE_TYPE    = LV_RULE.

        PERFORM ENRICH_FROM_START USING '' CHANGING LS_OUT.

        IF LV_SRC_TAB IS NOT INITIAL AND LV_SRC IS NOT INITIAL.
          LV_FIELD = LV_SRC.
          PERFORM GET_KEYFLAG USING LV_SRC_TAB LV_FIELD
                              CHANGING LS_OUT-KEYFLAG_S.
        ENDIF.

        APPEND LS_OUT TO PT_OUT.
      ENDLOOP.

    " D) many-to-many -> compute BOTH key flags per pair
    ELSE.
      LOOP AT LT_SOURCES INTO LV_SRC2.
        LOOP AT LT_TARGETS INTO LV_TGT2.
          CLEAR LS_OUT.
          LS_OUT-SOURCE_FIELD = LV_SRC2.
          LS_OUT-TARGET_FIELD = LV_TGT2.
          LS_OUT-RULE_TYPE    = LV_RULE.

          PERFORM ENRICH_FROM_START USING LV_TGT2 CHANGING LS_OUT.

          " Source keyflag
          IF LV_SRC_TAB IS NOT INITIAL AND LV_SRC2 IS NOT INITIAL.
            LV_FIELD = LV_SRC2.
            PERFORM GET_KEYFLAG USING LV_SRC_TAB LV_FIELD
                                CHANGING LS_OUT-KEYFLAG_S.
          ENDIF.

          " Target keyflag
          IF LV_TGT_TAB IS NOT INITIAL AND LV_TGT2 IS NOT INITIAL.
            LV_FIELD = LV_TGT2.
            PERFORM GET_KEYFLAG USING LV_TGT_TAB LV_FIELD
                                CHANGING LS_OUT-KEYFLAG_T.
          ENDIF.

          APPEND LS_OUT TO PT_OUT.
        ENDLOOP.
      ENDLOOP.
    ENDIF.

    " reset accumulators for next rule-id
    REFRESH LT_SOURCES.
    REFRESH LT_TARGETS.
    CLEAR   LV_RULE.

  END-OF-DEFINITION.

  "--------------------------------------------------------------------
  " 3) Walk rows and flush explicitly per RULEID
  "--------------------------------------------------------------------
  CLEAR LV_PREV.

  LOOP AT LT_FIELDS INTO LS_FIELD.

    LV_CURR = LS_FIELD-RULEID.

    " initialize header for the very first group
    IF LV_PREV IS INITIAL.
      READ TABLE LT_RULES INTO LS_RULEH
           WITH KEY TRANID = PV_TRANID RULEID = LV_CURR
           BINARY SEARCH.
      IF SY-SUBRC = 0.
        LV_RULE = LS_RULEH-RULETYPE.
      ELSE.
        CLEAR LV_RULE.
      ENDIF.
      LV_PREV = LV_CURR.
    ENDIF.

    " group break -> flush the previous group first
    IF LV_CURR <> LV_PREV.
      _FLUSH_GROUP.

      " new group's header
      READ TABLE LT_RULES INTO LS_RULEH
           WITH KEY TRANID = PV_TRANID RULEID = LV_CURR
           BINARY SEARCH.
      IF SY-SUBRC = 0.
        LV_RULE = LS_RULEH-RULETYPE.
      ELSE.
        CLEAR LV_RULE.
      ENDIF.

      LV_PREV = LV_CURR.
    ENDIF.

    " accumulate this field to its role
    IF LS_FIELD-PARAMTYPE = '0'.
      APPEND LS_FIELD-FIELDNM TO LT_SOURCES.
    ELSEIF LS_FIELD-PARAMTYPE = '1'.
      APPEND LS_FIELD-FIELDNM TO LT_TARGETS.
    ENDIF.

  ENDLOOP.

  " flush last open group
  IF LV_PREV IS NOT INITIAL.
    _FLUSH_GROUP.
  ENDIF.

  "--------------------------------------------------------------------
  " 4) Final cleanup as in your original version
  "--------------------------------------------------------------------
  DELETE PT_OUT WHERE RULE_TYPE = 'START'.
  DELETE PT_OUT WHERE RULE_TYPE = 'END'.

ENDFORM.

FORM ENSURE_FRONTEND_DIR USING PV_DIR TYPE STRING
                         CHANGING PV_OK TYPE C.

  DATA: LV_DIR   TYPE STRING,
        LT_PARTS TYPE STANDARD TABLE OF STRING,
        LV_PART  TYPE STRING,
        LV_ACC   TYPE STRING,
        LV_RC    TYPE I.

  CLEAR PV_OK.
  IF SY-BATCH = 'X'. RETURN. ENDIF.
  IF PV_DIR IS INITIAL. RETURN. ENDIF.

  LV_DIR = PV_DIR.
  REPLACE ALL OCCURRENCES OF '/' IN LV_DIR WITH '\'.

  SPLIT LV_DIR AT '\' INTO TABLE LT_PARTS.

  LOOP AT LT_PARTS INTO LV_PART.
    IF LV_PART IS INITIAL. CONTINUE. ENDIF.

    IF LV_ACC IS INITIAL.
      LV_ACC = LV_PART.
    ELSE.
      CONCATENATE LV_ACC '\' LV_PART INTO LV_ACC.
    ENDIF.

  ENDLOOP.

  PV_OK = 'X'.

ENDFORM.

FORM WRITE_FLOW_CSV.
  DATA: LT_OUT      TYPE STANDARD TABLE OF STRING,
        LV          TYPE STRING,
        LV_FILE     TYPE STRING,
        LS_F        TYPE TY_FLOW,
        LV_DIR_ONLY TYPE STRING,

  DELETE ADJACENT DUPLICATES FROM GT_FLOW COMPARING TRANID.

  LOOP AT GT_FLOW INTO LS_F.

    " Fix source type: check if TRANSFER STRUCTURE is actually an InfoSource
    IF LS_F-SOURCE_TYPE = 'TRANSFER STRUCTURE'
    OR ( LS_F-SOURCE_TYPE = 'DATASOURCE' AND LS_F-SOURCE_LOGSYS IS INITIAL ).

          LS_F-SOURCE_TYPE = 'INFOSOURCE'.
        ELSEIF LS_F-SOURCE_TYPE = 'DATASOURCE'.
          LS_F-SOURCE_TYPE = 'TRANSFER STRUCTURE'.
        ENDIF.
      ENDIF.
    ENDIF.

      FIND REGEX '_IS$' IN LS_F-TARGET.

        DATA LV_IS_KEY3  TYPE RSKSFIELDNEW-ISOURCE.

        SELECT SINGLE ISOURCE INTO LV_IS_CHK2 FROM RSKSFIELDNEW
          WHERE ISOURCE = LV_IS_KEY3 AND OBJVERS = 'A'.
        IF SY-SUBRC = 0.
          LS_F-TARGET_TYPE = 'INFOSOURCE'.
        ENDIF.

            LS_F-CONTEXT_PROVIDER

            LS_F-TARGET_TYPE
            LS_F-T_TABLE_NAME
            LS_F-SOURCE
            LS_F-SOURCE_TYPE
            LS_F-SOURCE_LOGSYS     " <— NEW COLUMN
            LS_F-S_TABLE_NAME
            LS_F-TRANID
            LS_F-PROGRAM
            ''                     " unused slot
          CHANGING LV.
    APPEND LV TO LT_OUT.
  ENDLOOP.
*  PERFORM GET_FULLPATH_CTX USING 'Flow.csv' CHANGING LV_FILE.
*  PERFORM GET_FULLPATH_CTX USING 'Flow/Flow.csv' CHANGING LV_FILE.
  DATA LV_BASE_C30 TYPE CHAR30.
*  DATA LV_DIR_ONLY TYPE STRING.
  DATA OK          TYPE C.

  LV_BASE_C30 = GV_CTX.                      " provider
  PERFORM SANITIZE_FOR_FILE USING LV_BASE_C30 CHANGING LV_BASE_C30.

  " Compose: <provider>\Flow\<provider>_Flow.csv

  " Ensure folder exists (in case)
  PERFORM GET_DIRNAME USING LV_FILE CHANGING LV_DIR_ONLY.
  PERFORM ENSURE_OUT_DIR USING LV_DIR_ONLY CHANGING OK.

  IF P_OUTDIR IS  NOT INITIAL.

FORM PRINT_SUBTREE TABLES  PT_OUT
                   USING   PV_TARGET
                           PV_PREFIX.
  DATA LT_CHILDREN    TYPE STANDARD TABLE OF TY_FLOW.
  DATA LS_F           TYPE TY_FLOW.
  DATA LV_I           TYPE I.
  DATA LV_N           TYPE I.
  DATA LV_BRANCH      TYPE STRING.
  DATA LV_LINE        TYPE STRING.
  DATA LV_NEXT_PREFIX TYPE STRING.

  REFRESH LT_CHILDREN.
  LOOP AT GT_FLOW INTO LS_F WHERE TARGET = PV_TARGET.
    APPEND LS_F TO LT_CHILDREN.
  ENDLOOP.

  DESCRIBE TABLE LT_CHILDREN LINES LV_N.
  IF LV_N = 0.
    RETURN.
  ENDIF.

  LV_I = 0.
  LOOP AT LT_CHILDREN INTO LS_F.
    LV_I = LV_I + 1.
    IF LV_I < LV_N.
      LV_BRANCH = '├─'.
      CONCATENATE PV_PREFIX '│  ' INTO LV_NEXT_PREFIX.
    ELSE.
      LV_BRANCH = '└─'.
      CONCATENATE PV_PREFIX '   ' INTO LV_NEXT_PREFIX.
    ENDIF.

    CLEAR LV_LINE.
    CONCATENATE PV_PREFIX LV_BRANCH LS_F-SOURCE '  [' LS_F-SOURCE_TYPE ']'
           INTO LV_LINE SEPARATED BY SPACE.
    APPEND LV_LINE TO PT_OUT.

    IF LS_F-SOURCE_TYPE <> 'DATASOURCE' AND LS_F-SOURCE_TYPE <> 'INFOSOURCE'.
      PERFORM PRINT_SUBTREE TABLES PT_OUT USING LS_F-SOURCE LV_NEXT_PREFIX.
    ENDIF.
  ENDLOOP.
ENDFORM.

* Routine collection (METHOD-based + INCLUDE expansion)               *
*---------------------------------------------------------------------*
FORM RX_COLLECT_ROUTINES USING    PV_TRANID TYPE RSTRAN-TRANID
                                  PV_PROG  TYPE PROGNAME
                         CHANGING PT_ROUT  TYPE TY_ROUTINE_TAB.
  DATA LT_SRC TYPE STANDARD TABLE OF STRING.
  DATA LS_R  TYPE TY_ROUTINE.
  DATA LV_LINE  TYPE STRING.
  DATA LV_ID    TYPE STRING.
  DATA LV_MNAME TYPE STRING.
  DATA LV_OK    TYPE C.

  REFRESH PT_ROUT.

*  PERFORM FETCH_REPORT_SOURCE USING PV_PROG CHANGING LT_SRC LV_OK.
  PERFORM FETCH_REPORT_SOURCE TABLES LT_SRC USING PV_PROG CHANGING LV_OK.
*  IF LV_OK IS INITIAL. RETURN. ENDIF.

  " START
  PERFORM RX_EXTRACT_METHOD_FULL TABLES LT_SRC USING 'start_routine' CHANGING LS_R.
  IF LS_R-CODE IS NOT INITIAL.
    LS_R-TRANID       = PV_TRANID.
    LS_R-ROUTINE_TYPE = 'START'.
    CLEAR LS_R-ROUTINE_ID.
    APPEND LS_R TO PT_ROUT.
    CLEAR LS_R.
  ENDIF.

  " END
  PERFORM RX_EXTRACT_METHOD_FULL TABLES LT_SRC USING 'end_routine' CHANGING LS_R.
  IF LS_R-CODE IS NOT INITIAL.
    LS_R-TRANID       = PV_TRANID.
    LS_R-ROUTINE_TYPE = 'END'.
    APPEND LS_R TO PT_ROUT.
    CLEAR LS_R.
  ENDIF.

  " EXPERT
  PERFORM RX_EXTRACT_METHOD_FULL TABLES LT_SRC USING 'expert_routine' CHANGING LS_R.
  IF LS_R-CODE IS NOT INITIAL.
    LS_R-TRANID       = PV_TRANID.
    LS_R-ROUTINE_TYPE = 'EXPERT'.
    APPEND LS_R TO PT_ROUT.
    CLEAR LS_R.
  ENDIF.

  " FIELD routines: METHOD compute_*
  LOOP AT LT_SRC INTO LV_LINE.
    CLEAR LV_ID.
    FIND REGEX '^\s*METHOD\s+compute_([A-Za-z0-9_]+)\.' IN LV_LINE IGNORING CASE SUBMATCHES LV_ID.
    IF SY-SUBRC <> 0.
      FIND REGEX '^\s*METHOD\s+IF_[A-Za-z0-9_]+~compute_([A-Za-z0-9_]+)\.' IN LV_LINE IGNORING CASE SUBMATCHES LV_ID.
    ENDIF.
    IF SY-SUBRC = 0 AND LV_ID IS NOT INITIAL.
      CLEAR LV_MNAME. CONCATENATE 'compute_' LV_ID INTO LV_MNAME.
      PERFORM RX_EXTRACT_METHOD_FULL TABLES LT_SRC USING LV_MNAME CHANGING LS_R.
      IF LS_R-CODE IS NOT INITIAL.
        LS_R-TRANID       = PV_TRANID.
        LS_R-ROUTINE_TYPE = 'FIELD'.
        LS_R-ROUTINE_ID   = LV_ID.
        APPEND LS_R TO PT_ROUT. CLEAR LS_R.
      ENDIF.
    ENDIF.
  ENDLOOP.
ENDFORM.

  DATA LV_END   TYPE I VALUE 0.

  DATA LV_PAT1  TYPE STRING.
  DATA LV_PAT2  TYPE STRING.

  DATA LV_MAX   TYPE I.
  REFRESH PT_BODY.
  CONCATENATE '^\s*METHOD\s+' PV_NAME '\.' INTO LV_PAT1.
  CONCATENATE '^\s*METHOD\s+IF_[A-Z0-9_]+~' PV_NAME '\.' INTO LV_PAT2.
  DESCRIBE TABLE PT_SRC LINES LV_MAX.
  LV_IDX = 1.
  WHILE LV_IDX <= LV_MAX.
    READ TABLE PT_SRC INTO LV_LINE INDEX LV_IDX.
    IF SY-SUBRC <> 0. EXIT. ENDIF.
    IF LV_BEGIN = 0.
      FIND REGEX LV_PAT1 IN LV_LINE IGNORING CASE.
      IF SY-SUBRC <> 0. FIND REGEX LV_PAT2 IN LV_LINE IGNORING CASE. ENDIF.
      IF SY-SUBRC = 0. LV_BEGIN = LV_IDX + 1. ENDIF.
    ELSE.
      FIND REGEX '^\s*ENDMETHOD\.' IN LV_LINE IGNORING CASE.
      IF SY-SUBRC = 0. LV_END = LV_IDX - 1. EXIT. ENDIF.
    ENDIF.
    LV_IDX = LV_IDX + 1.
  ENDWHILE.
  IF LV_BEGIN = 0 OR LV_END < LV_BEGIN. RETURN. ENDIF.
  LV_IDX = LV_BEGIN.
  WHILE LV_IDX <= LV_END.
    READ TABLE PT_SRC INTO LV_LINE INDEX LV_IDX.
    IF SY-SUBRC <> 0. EXIT. ENDIF.
    APPEND LV_LINE TO PT_BODY.
    LV_IDX = LV_IDX + 1.
  ENDWHILE.
ENDFORM.

" Expand INCLUDE <prog> [IF FOUND]. recursively – CHANGING seen set
FORM RX_EXPAND_INCLUDES TABLES  PT_IN
                                 PT_OUT
                          CHANGING PT_SEEN TYPE TY_SEEN_TAB.
  DATA LV     TYPE STRING.
  DATA LV_INC TYPE PROGNAME.
  DATA LT_INC TYPE STANDARD TABLE OF STRING.
  DATA LT_EXP TYPE STANDARD TABLE OF STRING.
  REFRESH PT_OUT.
  LOOP AT PT_IN INTO LV.
    FIND REGEX '^\s*INCLUDE\s+([A-Za-z0-9_/]+)\s*(IF FOUND)?\.' IN LV SUBMATCHES LV_INC.
    IF SY-SUBRC = 0 AND LV_INC IS NOT INITIAL.
      READ TABLE PT_SEEN WITH KEY TABLE_LINE = LV_INC TRANSPORTING NO FIELDS.
      IF SY-SUBRC <> 0.
        INSERT LV_INC INTO TABLE PT_SEEN.
        REFRESH LT_INC.
        READ REPORT LV_INC INTO LT_INC.
        IF SY-SUBRC = 0 AND LT_INC IS NOT INITIAL.
          REFRESH LT_EXP.
          PERFORM RX_EXPAND_INCLUDES TABLES LT_INC LT_EXP CHANGING PT_SEEN.
          APPEND LINES OF LT_EXP TO PT_OUT.
        ELSE.
          CONCATENATE '* [MISSING INCLUDE]' LV INTO LV SEPARATED BY SPACE.
          APPEND LV TO PT_OUT.
        ENDIF.
      ENDIF.
    ELSE.
      APPEND LV TO PT_OUT.
    ENDIF.
  ENDLOOP.
ENDFORM.

FORM RX_SCAN_START_BODY TABLES PT_LINES
                        CHANGING PT_DET TYPE TY_DETAIL_TAB.

  DATA LV       TYPE STRING.
  DATA LV_TGT   TYPE STRING.
  DATA LV_RHS   TYPE STRING.
  DATA LV_TAB   TYPE STRING.
  DATA LS_DET   TYPE TY_DETAIL.
  FIELD-SYMBOLS <DET> TYPE TY_DETAIL.

  LOOP AT PT_LINES INTO LV.

    "--- 1) Assignment to target field: RESULT_FIELDS-<field> = <rhs>
    "    or E_T_* - <field> = <rhs>   (spaces/case tolerant)
    CLEAR: LV_TGT, LV_RHS.
    FIND REGEX '^\s*RESULT_FIELDS-\s*([A-Za-z0-9_]+)\s*=\s*(.+)$'
         IN LV IGNORING CASE SUBMATCHES LV_TGT LV_RHS.
    IF SY-SUBRC <> 0.
      FIND REGEX '^\s*E_T_[A-Za-z0-9_]+\s*-\s*([A-Za-z0-9_]+)\s*=\s*(.+)$'
           IN LV IGNORING CASE SUBMATCHES LV_TGT LV_RHS.
    ENDIF.

    IF SY-SUBRC = 0 AND LV_TGT IS NOT INITIAL.
      READ TABLE PT_DET ASSIGNING <DET> WITH KEY TARGET_FIELD = LV_TGT.
      IF SY-SUBRC <> 0.
        CLEAR LS_DET.
        LS_DET-TARGET_FIELD = LV_TGT.
        INSERT LS_DET INTO TABLE PT_DET ASSIGNING <DET>.
      ENDIF.

      IF <DET> IS ASSIGNED.
        <DET>-ROUTINE_FLAG = 'X'.

        " Try to capture a direct mapping: SOURCE_FIELDS-<src>
        DATA LV_SRC_IOBJ TYPE STRING.
        FIND REGEX 'SOURCE_FIELDS-\s*([A-Za-z0-9_]+)'
             IN LV_RHS IGNORING CASE SUBMATCHES LV_SRC_IOBJ.
        IF SY-SUBRC = 0 AND <DET> IS ASSIGNED AND LV_SRC_IOBJ IS NOT INITIAL
           AND <DET>-SOURCE_FIELD IS INITIAL.
          <DET>-SOURCE_FIELD = LV_SRC_IOBJ.
        ENDIF.

        " 1.a Constant on RHS → pick first quoted literal
        FIND REGEX '''([^'']*)''' IN LV_RHS SUBMATCHES DATA(LV_CONST).
        IF SY-SUBRC = 0 AND LV_CONST IS NOT INITIAL.
          <DET>-CONSTANT_VAL = LV_CONST.
        ELSE.
          " 1.b Otherwise treat as formula if it has operators/paren
          IF LV_RHS CS '+' OR LV_RHS CS '-' OR LV_RHS CS '*' OR LV_RHS CS '/' OR
             LV_RHS CS '(' OR LV_RHS CS ')'.
            <DET>-FORMULA_TEXT = LV_RHS.
          ENDIF.
        ENDIF.
      ENDIF.
    ENDIF.

    "--- 2) Master data source: SELECT ... FROM <table> (tied to last LV_TGT)
    IF LV CS 'SELECT' AND ( LV CS '/BIC/' OR LV CS '/BI0/' OR LV CS ' FROM ' ).
      " *** FIXED: no (?i), no \b — use IGNORING CASE for case-insensitive match
      FIND REGEX 'FROM\s+([A-Za-z0-9_/]+)'
           IN LV IGNORING CASE SUBMATCHES LV_TAB.
      IF SY-SUBRC = 0 AND LV_TGT IS NOT INITIAL AND LV_TAB IS NOT INITIAL.
        READ TABLE PT_DET ASSIGNING <DET> WITH KEY TARGET_FIELD = LV_TGT.
        IF SY-SUBRC <> 0.
          CLEAR LS_DET.
          LS_DET-TARGET_FIELD = LV_TGT.
          INSERT LS_DET INTO TABLE PT_DET ASSIGNING <DET>.
        ENDIF.
        IF <DET> IS ASSIGNED AND <DET>-MASTER_SOURCE IS INITIAL.
          <DET>-MASTER_SOURCE = LV_TAB.
        ENDIF.
      ENDIF.
    ELSEIF LV CS 'READ TABLE'.
      IF LV_TGT IS NOT INITIAL.
        READ TABLE PT_DET ASSIGNING <DET> WITH KEY TARGET_FIELD = LV_TGT.
        IF SY-SUBRC <> 0.
          CLEAR LS_DET.
          LS_DET-TARGET_FIELD = LV_TGT.
          INSERT LS_DET INTO TABLE PT_DET ASSIGNING <DET>.
        ENDIF.
        IF <DET> IS ASSIGNED AND <DET>-MASTER_SOURCE IS INITIAL.
          <DET>-MASTER_SOURCE = 'READ TABLE ...'.
        ENDIF.
      ENDIF.
    ENDIF.

  ENDLOOP.
ENDFORM.

*---------------------------------------------------------------------*
* Sanitize name for file usage (safe on all kernels)
*---------------------------------------------------------------------*
*---------------------------------------------------------------------*
* Sanitize name for file usage (safe on all kernels; no REPLACE-ALL)
*---------------------------------------------------------------------*
FORM SANITIZE_FOR_FILE USING    PV_IN  TYPE CHAR30
                       CHANGING PV_OUT TYPE CHAR30.

  DATA: LV_IN   TYPE STRING,
        LV_OUT  TYPE STRING,
        LV_C    TYPE C LENGTH 1,
        LV_PREV TYPE C LENGTH 1 VALUE ' ',
        LV_I    TYPE I,
        LV_LEN  TYPE I.

  " Work on a STRING copy (avoid CHAR trailing-space quirks)
  LV_IN  = PV_IN.
  LV_LEN = STRLEN( LV_IN ).
  CLEAR LV_OUT.

  LV_I = 0.
  WHILE LV_I < LV_LEN.
    LV_C = LV_IN+LV_I(1).

    " Allow: letters, digits, underscore, hyphen; everything else -> '_'
    IF ( LV_C >= 'A' AND LV_C <= 'Z' )
    OR ( LV_C >= 'a' AND LV_C <= 'z' )
    OR ( LV_C >= '0' AND LV_C <= '9' )
    OR   LV_C = '_'
    OR   LV_C = '-'.
      " keep as-is
    ELSE.
      LV_C = '_'.
    ENDIF.

    " Collapse consecutive underscores
    IF LV_C = '_' AND LV_PREV = '_'.
      " skip
    ELSE.
      CONCATENATE LV_OUT LV_C INTO LV_OUT IN CHARACTER MODE.
      LV_PREV = LV_C.
    ENDIF.

    LV_I = LV_I + 1.
  ENDWHILE.

  " Ensure not empty
  IF LV_OUT IS INITIAL.
    LV_OUT = 'file'.
  ENDIF.

  " Trim to 30 (your original type length)
  IF STRLEN( LV_OUT ) > 30.
    LV_OUT = LV_OUT(30).
  ENDIF.

  PV_OUT = LV_OUT.

ENDFORM." Compose <root>/<tier>/<base><suffix>
FORM MAKE_ABS_FILE_NAME USING PV_ROOT TYPE STRING
                              PV_TIER TYPE STRING
                              PV_BASE TYPE CHAR30
                              PV_SFX  TYPE STRING
                        CHANGING PV_FILE TYPE STRING.

  ENDIF.
  CONCATENATE LV_DIR PV_BASE PV_SFX INTO PV_FILE.
ENDFORM.

" DDIC -> SQL type (legacy-safe; IF/ELSEIF ladder)
FORM MAKE_SQL_TYPE USING    PV_DATATYPE TYPE DD03P-DATATYPE
                             PV_LENG    TYPE DD03P-LENG
                             PV_DECIM   TYPE DD03P-DECIMALS
                   CHANGING PV_SQLTYPE  TYPE STRING.

  DATA LV     TYPE STRING.
  DATA LV_DT  TYPE DD03P-DATATYPE.

  " Work on a local copy; do not modify USING parameter
  LV_DT = PV_DATATYPE.
  TRANSLATE LV_DT TO UPPER CASE.

  IF LV_DT = 'CHAR' OR LV_DT = 'LCHR' OR LV_DT = 'LANG' OR LV_DT = 'CLNT'.
    CONCATENATE 'VARCHAR(' PV_LENG ')' INTO LV.

  ELSEIF LV_DT = 'NUMC'.
    CONCATENATE 'VARCHAR(' PV_LENG ')' INTO LV.

  ELSEIF LV_DT = 'RAW' OR LV_DT = 'LRAW'.
    CONCATENATE 'VARBINARY(' PV_LENG ')' INTO LV.

  ELSEIF LV_DT = 'INT1' OR LV_DT = 'INT2'.
    LV = 'SMALLINT'.

  ELSEIF LV_DT = 'INT4'.
    LV = 'INTEGER'.

  ELSEIF LV_DT = 'FLTP'.
    LV = 'DOUBLE'.

  ELSEIF LV_DT = 'DEC' OR LV_DT = 'CURR' OR LV_DT = 'QUAN'.
    CONCATENATE 'DECIMAL(' PV_LENG ',' PV_DECIM ')' INTO LV.

  ELSEIF LV_DT = 'DATS'.
    LV = 'DATE'.

  ELSEIF LV_DT = 'TIMS'.
    LV = 'TIME'.

  ELSE.
    CONCATENATE 'VARCHAR(' PV_LENG ')' INTO LV.
  ENDIF.

  PV_SQLTYPE = LV.
ENDFORM.

" Physical DDIC table -> CREATE TABLE
*---------------------------------------------------------------------*
* Make DDL from DDIC (global IO: GV_TABNAME -> GT_SQL)
*---------------------------------------------------------------------*
*---------------------------------------------------------------------*
* Make DDL from DDIC (global IO: GV_TABNAME -> GT_SQL)
* - Input : GV_TABNAME (table name)
* - Output: GT_SQL     (CREATE TABLE DDL lines)
* Notes   : Parameter-less (no USING/CHANGING), avoids FORM/PERFORM issues.
*           No substring arithmetic; trailing comma removed via SHIFT.
*---------------------------------------------------------------------*
FORM MAKE_SQL_DDL_FROM_DDIC_2.

  DATA: LT_DD03P TYPE STANDARD TABLE OF DD03P WITH EMPTY KEY,
        LS_DD03P TYPE DD03P,
        LT_KEYS  TYPE STANDARD TABLE OF STRING WITH EMPTY KEY,
        LV_LINE  TYPE STRING,
        LV_TYPE  TYPE STRING,
        LV_K     TYPE STRING.

  " Always start clean
  REFRESH GT_SQL.
  REFRESH LT_KEYS.

  " Read DDIC metadata for the physical table
  CALL FUNCTION 'DDIF_TABL_GET'
    EXPORTING
      NAME      = GV_TABNAME
    TABLES
      DD03P_TAB = LT_DD03P
    EXCEPTIONS
      OTHERS    = 1.

  IF SY-SUBRC <> 0 OR LT_DD03P IS INITIAL.
    RETURN.
  ENDIF.

  " Header
  CONCATENATE 'CREATE TABLE "' GV_TABNAME '" (' INTO LV_LINE.
  APPEND LV_LINE TO GT_SQL.

  " Columns (append with comma; we'll safely remove the final comma later)
  LOOP AT LT_DD03P INTO LS_DD03P WHERE FIELDNAME IS NOT INITIAL.

    PERFORM MAKE_SQL_TYPE USING    LS_DD03P-DATATYPE
                                   LS_DD03P-LENG
                                   LS_DD03P-DECIMALS
                          CHANGING LV_TYPE.

    CLEAR LV_LINE.
    CONCATENATE ' "' LS_DD03P-FIELDNAME '" ' LV_TYPE ',' INTO LV_LINE.
    APPEND LV_LINE TO GT_SQL.

    IF LS_DD03P-KEYFLAG = 'X'.
      APPEND LS_DD03P-FIELDNAME TO LT_KEYS.
    ENDIF.

  ENDLOOP.

  " --- Remove trailing comma from the last column line (no substrings used)
  READ TABLE GT_SQL INTO LV_LINE INDEX LINES( GT_SQL ).
  IF SY-SUBRC = 0.
    DATA TEMP_LAST TYPE STRING.
    TEMP_LAST = LV_LINE.
    " This will drop any trailing ',' without using offsets/lengths
    SHIFT TEMP_LAST RIGHT DELETING TRAILING ','.
    LV_LINE = TEMP_LAST.
    MODIFY GT_SQL FROM LV_LINE INDEX LINES( GT_SQL ).
  ENDIF.

  " Primary key (if any)
  IF LT_KEYS IS NOT INITIAL.
    CLEAR LV_LINE.
    LOOP AT LT_KEYS INTO LV_K.
      IF LV_LINE IS INITIAL.
        LV_LINE = LV_K.
      ELSE.
        CONCATENATE LV_LINE '","' LV_K INTO LV_LINE.
      ENDIF.
    ENDLOOP.
    CONCATENATE ' ,PRIMARY KEY ("' LV_LINE '")' INTO LV_LINE.
    APPEND LV_LINE TO GT_SQL.
  ENDIF.

  " Close
  APPEND ');' TO GT_SQL.

ENDFORM.

*---------------------------------------------------------------------*
* MAKE_SQL_DDL_FROM_IS
*  Generates DDL for an InfoSource communication structure.
*  Uses DDIF_TABL_GET on the InfoSource DDIC structure name.
*  BW registers InfoSource comm structures as DDIC structures named
*  after the InfoSource itself (e.g. N1DGLI13_IS).
*  Falls back to RSTSFIELD (same as TS) if DDIC lookup returns nothing.
*  Input : GV_ISNAME  (InfoSource name, e.g. N1DGLI13_IS)
*  Output: GT_SQL
*---------------------------------------------------------------------*
FORM MAKE_SQL_DDL_FROM_IS.
* Reads InfoSource field definitions from RSKSFIELDNEW.
* Key fields: ISOURCE, OBJVERS, SEGID, POSIT.
* Has FIELDNM, DATATYPE, LENG, DECIMALS — everything needed for DDL.

  DATA: LT_KS   TYPE STANDARD TABLE OF RSKSFIELDNEW WITH EMPTY KEY,
        LS_KS   TYPE RSKSFIELDNEW,
        LV_LINE TYPE STRING,
        LV_TYPE TYPE STRING,
        LV_DT   TYPE DD03P-DATATYPE,
        LV_LEN  TYPE DD03P-LENG,
        LV_DEC  TYPE DD03P-DECIMALS.

  REFRESH GT_SQL.
  IF GV_ISNAME IS INITIAL. RETURN. ENDIF.

  SELECT * FROM RSKSFIELDNEW INTO TABLE LT_KS
    WHERE ISOURCE = GV_ISNAME
      AND OBJVERS = 'A'
      AND SEGID   = '0001'
  ORDER BY POSIT.

  IF LT_KS IS INITIAL. RETURN. ENDIF.

  CONCATENATE 'CREATE TABLE "' GV_ISNAME '_IS" (' INTO LV_LINE.
  APPEND LV_LINE TO GT_SQL.

  LOOP AT LT_KS INTO LS_KS.
    CLEAR: LV_DT, LV_LEN, LV_DEC.
    LV_DT  = LS_KS-DATATYPE.
    LV_LEN = LS_KS-LENG.
    LV_DEC = LS_KS-DECIMALS.
    PERFORM MAKE_SQL_TYPE USING LV_DT LV_LEN LV_DEC CHANGING LV_TYPE.
    CLEAR LV_LINE.
    CONCATENATE ' "' LS_KS-FIELDNM '" ' LV_TYPE ',' INTO LV_LINE.
    APPEND LV_LINE TO GT_SQL.
  ENDLOOP.

  " Remove trailing comma on last field
  READ TABLE GT_SQL INTO LV_LINE INDEX LINES( GT_SQL ).
  IF SY-SUBRC = 0.
    DATA LV_LAST TYPE STRING.
    LV_LAST = LV_LINE.
    SHIFT LV_LAST RIGHT DELETING TRAILING ','.
    LV_LINE = LV_LAST.
    MODIFY GT_SQL FROM LV_LINE INDEX LINES( GT_SQL ).
  ENDIF.
  APPEND ');' TO GT_SQL.

ENDFORM.

*---------------------------------------------------------------------*
* SCAN_ROUTINE_BLOCK (parameter-less, regex-free)
*  in : GT_SCAN_SRC, GV_SCAN_TRANID, GV_SCAN_CONTAINER
* out: GT_DEP_LOG, GT_DDL_TABS, GT_FM_QUEUE
*---------------------------------------------------------------------*
FORM SCAN_ROUTINE_BLOCK.

  FIELD-SYMBOLS <S> TYPE STRING.

  DATA: LN  TYPE I,
        LV  TYPE STRING,
        LVU TYPE STRING,
        C   TYPE C LENGTH 1,
        TAB TYPE DD02L-TABNAME,
        FM  TYPE RS38L_FNAM,
        MTH TYPE STRING,
        INC TYPE PROGNAME,
        DEP TYPE TY_DEP,
        I   TYPE I,
        J   TYPE I,
        LEN TYPE I,
        IDX TYPE I.

  LN = 0.
  LOOP AT GT_SCAN_SRC ASSIGNING <S>.
    LN  = LN + 1.
    LV  = <S>.
    LVU = LV.
    TRANSLATE LVU TO UPPER CASE.
    LEN = STRLEN( LVU ).

    " --------------------
    " SELECT ... FROM <TAB>
    CLEAR TAB. IDX = -1.
    FIND FIRST OCCURRENCE OF ' FROM ' IN LVU MATCH OFFSET IDX.
    IF SY-SUBRC = 0 AND IDX GE 0.
      IDX = IDX + 6. " skip ' FROM '
      WHILE IDX < LEN AND LVU+IDX(1) = ' '. IDX = IDX + 1. ENDWHILE.
      CLEAR TAB. I = IDX.
      WHILE I < LEN.
        C = LVU+I(1).
        IF ( ( C GE 'A' AND C LE 'Z' ) OR ( C GE '0' AND C LE '9' ) OR C = '_' OR C = '/' ).
          CONCATENATE TAB C INTO TAB.
          I = I + 1.
        ELSE.
          EXIT.
        ENDIF.
      ENDWHILE.
      IF TAB IS NOT INITIAL.
        CLEAR DEP.
        DEP-TRANID    = GV_SCAN_TRANID.
        DEP-CONTAINER = GV_SCAN_CONTAINER.
        DEP-DEP_KIND  = 'TABLE'.
        DEP-DEP_NAME  = TAB.
        DEP-WHERE_TXT = 'SELECT FROM'.
        DEP-LINE_NO   = LN.
        APPEND DEP TO GT_DEP_LOG.
        INSERT TAB INTO TABLE GT_DDL_TABS.
      ENDIF.
    ENDIF.

    " UPDATE <TAB>
    CLEAR TAB. IDX = -1.
    FIND FIRST OCCURRENCE OF ' UPDATE ' IN LVU MATCH OFFSET IDX.
    IF SY-SUBRC = 0 AND IDX GE 0.
      IDX = IDX + 8.
      WHILE IDX < LEN AND LVU+IDX(1) = ' '. IDX = IDX + 1. ENDWHILE.
      CLEAR TAB. I = IDX.
      WHILE I < LEN.
        C = LVU+I(1).
        IF ( ( C GE 'A' AND C LE 'Z' ) OR ( C GE '0' AND C LE '9' ) OR C = '_' OR C = '/' ).
          CONCATENATE TAB C INTO TAB.
          I = I + 1.
        ELSE.
          EXIT.
        ENDIF.
      ENDWHILE.
      IF TAB IS NOT INITIAL.
        CLEAR DEP.
        DEP-TRANID    = GV_SCAN_TRANID.
        DEP-CONTAINER = GV_SCAN_CONTAINER.
        DEP-DEP_KIND  = 'TABLE'.
        DEP-DEP_NAME  = TAB.
        DEP-WHERE_TXT = 'UPDATE'.
        DEP-LINE_NO   = LN.
        APPEND DEP TO GT_DEP_LOG.
        INSERT TAB INTO TABLE GT_DDL_TABS.
      ENDIF.
    ENDIF.

    " INSERT [INTO] <TAB>
    CLEAR TAB. IDX = -1.
    FIND FIRST OCCURRENCE OF ' INSERT INTO ' IN LVU MATCH OFFSET IDX.
    IF SY-SUBRC = 0 AND IDX GE 0.
      IDX = IDX + 13.
    ELSE.
      FIND FIRST OCCURRENCE OF ' INSERT ' IN LVU MATCH OFFSET IDX.
      IF SY-SUBRC = 0 AND IDX GE 0.
        IDX = IDX + 8.
      ENDIF.
    ENDIF.
    IF IDX GE 0.
      WHILE IDX < LEN AND LVU+IDX(1) = ' '. IDX = IDX + 1. ENDWHILE.
      CLEAR TAB. I = IDX.
      WHILE I < LEN.
        C = LVU+I(1).
        IF ( ( C GE 'A' AND C LE 'Z' ) OR ( C GE '0' AND C LE '9' ) OR C = '_' OR C = '/' ).
          CONCATENATE TAB C INTO TAB.
          I = I + 1.
        ELSE.
          EXIT.
        ENDIF.
      ENDWHILE.
      IF TAB IS NOT INITIAL.
        CLEAR DEP.
        DEP-TRANID    = GV_SCAN_TRANID.
        DEP-CONTAINER = GV_SCAN_CONTAINER.
        DEP-DEP_KIND  = 'TABLE'.
        DEP-DEP_NAME  = TAB.
        DEP-WHERE_TXT = 'INSERT'.
        DEP-LINE_NO   = LN.
        APPEND DEP TO GT_DEP_LOG.
        INSERT TAB INTO TABLE GT_DDL_TABS.
      ENDIF.
    ENDIF.

    " DELETE [FROM] <TAB>
    CLEAR TAB. IDX = -1.
    FIND FIRST OCCURRENCE OF ' DELETE FROM ' IN LVU MATCH OFFSET IDX.
    IF SY-SUBRC = 0 AND IDX GE 0.
      IDX = IDX + 13.
    ELSE.
      FIND FIRST OCCURRENCE OF ' DELETE ' IN LVU MATCH OFFSET IDX.
      IF SY-SUBRC = 0 AND IDX GE 0.
        IDX = IDX + 8.
      ENDIF.
    ENDIF.
    IF IDX GE 0.
      WHILE IDX < LEN AND LVU+IDX(1) = ' '. IDX = IDX + 1. ENDWHILE.
      CLEAR TAB. I = IDX.
      WHILE I < LEN.
        C = LVU+I(1).
        IF ( ( C GE 'A' AND C LE 'Z' ) OR ( C GE '0' AND C LE '9' ) OR C = '_' OR C = '/' ).
          CONCATENATE TAB C INTO TAB.
          I = I + 1.
        ELSE.
          EXIT.
        ENDIF.
      ENDWHILE.
      IF TAB IS NOT INITIAL.
        CLEAR DEP.
        DEP-TRANID    = GV_SCAN_TRANID.
        DEP-CONTAINER = GV_SCAN_CONTAINER.
        DEP-DEP_KIND  = 'TABLE'.
        DEP-DEP_NAME  = TAB.
        DEP-WHERE_TXT = 'DELETE'.
        DEP-LINE_NO   = LN.
        APPEND DEP TO GT_DEP_LOG.
        INSERT TAB INTO TABLE GT_DDL_TABS.
      ENDIF.
    ENDIF.

    " JOIN <TAB> (multiple per line)
    DATA WORK TYPE STRING.
    WORK = LVU.
    DO.
      CLEAR TAB. IDX = -1.
      FIND FIRST OCCURRENCE OF ' JOIN ' IN WORK MATCH OFFSET IDX.
      IF SY-SUBRC <> 0 OR IDX LT 0.
        EXIT.
      ENDIF.
      IDX = IDX + 6.
      WHILE IDX < STRLEN( WORK ) AND WORK+IDX(1) = ' '. IDX = IDX + 1. ENDWHILE.
      CLEAR TAB. I = IDX.
      WHILE I < STRLEN( WORK ).
        C = WORK+I(1).
        IF ( ( C GE 'A' AND C LE 'Z' ) OR ( C GE '0' AND C LE '9' ) OR C = '_' OR C = '/' ).
          CONCATENATE TAB C INTO TAB.
          I = I + 1.
        ELSE.
          EXIT.
        ENDIF.
      ENDWHILE.
      IF TAB IS NOT INITIAL.
        CLEAR DEP.
        DEP-TRANID    = GV_SCAN_TRANID.
        DEP-CONTAINER = GV_SCAN_CONTAINER.
        DEP-DEP_KIND  = 'TABLE'.
        DEP-DEP_NAME  = TAB.
        DEP-WHERE_TXT = 'JOIN'.
        DEP-LINE_NO   = LN.
        APPEND DEP TO GT_DEP_LOG.
        INSERT TAB INTO TABLE GT_DDL_TABS.
        REPLACE FIRST OCCURRENCE OF TAB IN WORK WITH SPACE.
      ELSE.
        EXIT.
      ENDIF.
    ENDDO.

    " --------------------
    " CALL FUNCTION 'FM'
    CLEAR FM. IDX = -1.
    FIND FIRST OCCURRENCE OF 'CALL FUNCTION' IN LVU MATCH OFFSET IDX.
    IF SY-SUBRC = 0 AND IDX GE 0.
      I = IDX + STRLEN( 'CALL FUNCTION' ).
      J = -1.
      DATA K TYPE I.
      K = I.
      WHILE K < LEN.
        C = LVU+K(1).
        IF C = ''''. J = K. EXIT. ENDIF.
        K = K + 1.
      ENDWHILE.
      IF J GE 0.
        CLEAR FM. K = J + 1.
        WHILE K < LEN.
          C = LVU+K(1).
          IF C = ''''. EXIT. ENDIF.
          CONCATENATE FM C INTO FM.
          K = K + 1.
        ENDWHILE.
      ELSE.
        K = I.
        WHILE K < LEN AND LVU+K(1) = ' '. K = K + 1. ENDWHILE.
        CLEAR FM.
        WHILE K < LEN.
          C = LVU+K(1).
          IF ( ( C GE 'A' AND C LE 'Z' ) OR ( C GE '0' AND C LE '9' ) OR C = '_' ).
            CONCATENATE FM C INTO FM.
            K = K + 1.
          ELSE.
            EXIT.
          ENDIF.
        ENDWHILE.
      ENDIF.

      IF FM IS NOT INITIAL.
        CLEAR DEP.
        DEP-TRANID    = GV_SCAN_TRANID.
        DEP-CONTAINER = GV_SCAN_CONTAINER.
        DEP-DEP_KIND  = 'FM'.
        DEP-DEP_NAME  = FM.
        DEP-WHERE_TXT = 'CALL FUNCTION'.
        DEP-LINE_NO   = LN.
        APPEND DEP TO GT_DEP_LOG.
        INSERT FM INTO TABLE GT_FM_QUEUE.
      ENDIF.
    ENDIF.

    " --------------------
    " METHODS (class=>method and ref->method)
    CLEAR MTH. IDX = -1.
    FIND FIRST OCCURRENCE OF '=>' IN LVU MATCH OFFSET IDX.
    IF SY-SUBRC = 0 AND IDX GE 0.
      I = IDX + 2.
      WHILE I < LEN AND LVU+I(1) = ' '. I = I + 1. ENDWHILE.
      WHILE I < LEN.
        C = LVU+I(1).
        IF ( ( C GE 'A' AND C LE 'Z' ) OR ( C GE '0' AND C LE '9' ) OR C = '_' ).
          CONCATENATE MTH C INTO MTH.
          I = I + 1.
        ELSE.
          EXIT.
        ENDIF.
      ENDWHILE.
      IF MTH IS NOT INITIAL.
        CLEAR DEP.
        DEP-TRANID    = GV_SCAN_TRANID.
        DEP-CONTAINER = GV_SCAN_CONTAINER.
        DEP-DEP_KIND  = 'METHOD'.
        CONCATENATE '=>' MTH INTO DEP-DEP_NAME.
        DEP-WHERE_TXT = 'CALL METHOD'.
        DEP-LINE_NO   = LN.
        APPEND DEP TO GT_DEP_LOG.
      ENDIF.
    ENDIF.

    CLEAR MTH. IDX = -1.
    FIND FIRST OCCURRENCE OF '->' IN LVU MATCH OFFSET IDX.
    IF SY-SUBRC = 0 AND IDX GE 0.
      I = IDX + 2.
      WHILE I < LEN AND LVU+I(1) = ' '. I = I + 1. ENDWHILE.
      WHILE I < LEN.
        C = LVU+I(1).
        IF ( ( C GE 'A' AND C LE 'Z' ) OR ( C GE '0' AND C LE '9' ) OR C = '_' ).
          CONCATENATE MTH C INTO MTH.
          I = I + 1.
        ELSE.
          EXIT.
        ENDIF.
      ENDWHILE.
      IF MTH IS NOT INITIAL.
        CLEAR DEP.
        DEP-TRANID    = GV_SCAN_TRANID.
        DEP-CONTAINER = GV_SCAN_CONTAINER.
        DEP-DEP_KIND  = 'METHOD'.
        CONCATENATE '->' MTH INTO DEP-DEP_NAME.
        DEP-WHERE_TXT = 'CALL METHOD'.
        DEP-LINE_NO   = LN.
        APPEND DEP TO GT_DEP_LOG.
      ENDIF.
    ENDIF.

    " --------------------
    " INCLUDE <prog>.
    CLEAR INC. IDX = -1.
    FIND FIRST OCCURRENCE OF 'INCLUDE ' IN LVU MATCH OFFSET IDX.
    IF SY-SUBRC = 0 AND IDX GE 0.
      I = IDX + 8.
      WHILE I < LEN AND LVU+I(1) = ' '. I = I + 1. ENDWHILE.
      CLEAR INC.
      WHILE I < LEN.
        C = LVU+I(1).
        IF ( ( C GE 'A' AND C LE 'Z' ) OR ( C GE '0' AND C LE '9' ) OR C = '_' OR C = '/' ).
          CONCATENATE INC C INTO INC.
          I = I + 1.
        ELSEIF C = '.'.
          EXIT.
        ELSE.
          EXIT.
        ENDIF.
      ENDWHILE.
      IF INC IS NOT INITIAL.
        CLEAR DEP.
        DEP-TRANID    = GV_SCAN_TRANID.
        DEP-CONTAINER = GV_SCAN_CONTAINER.
        DEP-DEP_KIND  = 'INCLUDE'.
        DEP-DEP_NAME  = INC.
        DEP-WHERE_TXT = 'INCLUDE'.
        DEP-LINE_NO   = LN.
        APPEND DEP TO GT_DEP_LOG.
      ENDIF.
    ENDIF.

  ENDLOOP.

ENDFORM.

*---------------------------------------------------------------------*
* WRITE_DEPS_DDL  (old-kernel safe; routes MD to DDL/MD/)
*---------------------------------------------------------------------*
FORM WRITE_DEPS_DDL.

  IF GT_DDL_TABS IS INITIAL. RETURN. ENDIF.

  DATA LV_DIR  TYPE STRING.
  DATA OK      TYPE C.
  DATA LV_TAB  TYPE DD02L-TABNAME.
  DATA LV_BASE TYPE CHAR30.
  DATA LV_FILE TYPE STRING.
  DATA LT_SQL  TYPE STANDARD TABLE OF STRING WITH EMPTY KEY.
  DATA LV_IS_MD TYPE C.
  DATA LV_EXISTS TYPE DD02L-TABNAME.

  LOOP AT GT_DDL_TABS INTO LV_TAB.

    " Guard: only real DDIC tables
    CLEAR LV_EXISTS.
    SELECT SINGLE TABNAME INTO LV_EXISTS FROM DD02L WHERE TABNAME = LV_TAB.
    IF SY-SUBRC <> 0.
      CONTINUE.
    ENDIF.

    " Decide base folder: MD (P/T/H*) vs Deps
    CLEAR LV_IS_MD.
    IF LV_TAB CP '/BI0/P*' OR LV_TAB CP '/BI0/T*' OR LV_TAB CP '/BI0/H*'
     OR LV_TAB CP '/BIC/P*' OR LV_TAB CP '/BIC/T*' OR LV_TAB CP '/BIC/H*'.
      LV_IS_MD = 'X'.
    ENDIF.

    IF LV_IS_MD = 'X'.
      IF P_OUTDIR IS NOT INITIAL.
        CONCATENATE GV_DDL_DIR 'Master_Data\\' INTO LV_DIR.
      ELSE.
        LV_DIR = GV_DDL_DIR.
      ENDIF.
    ELSE.
      IF P_OUTDIR IS NOT INITIAL.
        CONCATENATE GV_DDL_DIR 'Dependency_Tables\\' INTO LV_DIR.
      ELSE.
        LV_DIR = GV_DDL_DIR.
      ENDIF.
    ENDIF.

    PERFORM ENSURE_OUT_DIR USING LV_DIR CHANGING OK.

    " Build SQL for this table
    REFRESH LT_SQL.
    GV_TABNAME = LV_TAB.
    PERFORM MAKE_SQL_DDL_FROM_DDIC_2.
    LT_SQL = GT_SQL.
    IF LT_SQL IS INITIAL. CONTINUE. ENDIF.

    " Compose file name
    LV_BASE = LV_TAB.
    PERFORM SANITIZE_FOR_FILE USING LV_BASE CHANGING LV_BASE.

    IF LV_IS_MD = 'X'.
      " MD files: keep neutral suffix .txt
      PERFORM MAKE_ABS_FILE_NAME USING LV_DIR '' LV_BASE '.txt' CHANGING LV_FILE.
    ELSE.
      " Dependencies files
      PERFORM MAKE_ABS_FILE_NAME USING LV_DIR '' LV_BASE '_Dep.txt' CHANGING LV_FILE.
    ENDIF.

    " Write file
    IF P_OUTDIR IS NOT INITIAL.
      PERFORM WRITE_DDL_FILE TABLES LT_SQL USING LV_FILE.
    ELSE.
      PERFORM WRITE_FILE_APPSERVER TABLES LT_SQL USING LV_FILE.
    ENDIF.

  ENDLOOP.

ENDFORM.

*---------------------------------------------------------------------*
* CLEAN_DEP_LOG (parameter-less)
* - Validates TABLE via DD02L; INCLUDE via TRDIR
* - Drops stopwords, LT_/IT_/GT_ work tables
* - Normalizes METHOD names, de-duplicates rows
*---------------------------------------------------------------------*
FORM CLEAN_DEP_LOG.

  DATA LT_CLEAN TYPE TY_DEP_TAB.
  DATA LS TYPE TY_DEP.
  DATA LV TYPE STRING.

  LOOP AT GT_DEP_LOG INTO LS.

    " --- METHOD: prettify
    IF LS-DEP_KIND = 'METHOD'.
      IF LS-DEP_NAME CP '->*'.
        CLEAR LV. LV = LS-DEP_NAME+2.         " drop '->'
        CONCATENATE 'Instance: ' LV INTO LS-DEP_NAME.
      ELSEIF LS-DEP_NAME CP '=>*'.
        CLEAR LV. LV = LS-DEP_NAME+2.         " drop '=>'
        CONCATENATE 'Static: ' LV INTO LS-DEP_NAME.
      ENDIF.
    ENDIF.

    " --- TABLE: prune noise & validate
    IF LS-DEP_KIND = 'TABLE'.
      " internal work tables
      IF LS-DEP_NAME CP 'LT_*' OR LS-DEP_NAME CP 'IT_*' OR LS-DEP_NAME CP 'GT_*'.
        CONTINUE.
      ENDIF.

      " stop-word prune (fast)
      LV = LS-DEP_NAME. TRANSLATE LV TO UPPER CASE.
      READ TABLE GT_STOPWORDS WITH KEY TABLE_LINE = LV TRANSPORTING NO FIELDS.
      IF SY-SUBRC = 0.
        CONTINUE.
      ENDIF.

      " DDIC validation
      SELECT SINGLE TABNAME FROM DD02L INTO @DATA(X)
        WHERE TABNAME = @LS-DEP_NAME.
      IF SY-SUBRC <> 0.
        CONTINUE.
      ENDIF.
    ENDIF.

    " --- INCLUDE: validate repo program exists
    IF LS-DEP_KIND = 'INCLUDE'.
      SELECT SINGLE NAME FROM TRDIR INTO @DATA(P)
        WHERE NAME = @LS-DEP_NAME.
      IF SY-SUBRC <> 0.
        CONTINUE.
      ENDIF.
    ENDIF.

    " keep row
    APPEND LS TO LT_CLEAN.
  ENDLOOP.

  " De-dup on business keys (leave line_no for human navigation only)
  SORT LT_CLEAN BY TRANID CONTAINER DEP_KIND DEP_NAME WHERE_TXT LINE_NO.
  DELETE ADJACENT DUPLICATES FROM LT_CLEAN
         COMPARING TRANID CONTAINER DEP_KIND DEP_NAME WHERE_TXT.

  GT_DEP_LOG = LT_CLEAN.

ENDFORM.

*---------------------------------------------------------------------*
* COLLECT_MD_FROM_FACT (parameter-less, old-kernel safe)
*  in : GV_TABNAME = fact table (e.g., /BIC/FN3BPUR)
* out: GT_DDL_TABS (adds /BI0|/BIC/P*, /T*, /H* if present)
*      GT_MD_IOBJ  (InfoObject names discovered)
*---------------------------------------------------------------------*
FORM COLLECT_MD_FROM_FACT.

  DATA: LT_DD03P TYPE STANDARD TABLE OF DD03P WITH EMPTY KEY,
        LS_DD03P TYPE DD03P,
        IO       TYPE RSIOBJNM,
        PREF     TYPE STRING,
        TAB_P    TYPE DD02L-TABNAME,
        TAB_T    TYPE DD02L-TABNAME,
        LV_LIKE  TYPE DD02L-TABNAME,
        LT_H     TYPE STANDARD TABLE OF DD02L-TABNAME WITH EMPTY KEY,
        H        TYPE DD02L-TABNAME,
        LV_DUMMY TYPE DD02L-TABNAME.

  IF GV_TABNAME IS INITIAL.
    RETURN.
  ENDIF.

  REFRESH LT_DD03P.
  CALL FUNCTION 'DDIF_TABL_GET'
    EXPORTING
      NAME      = GV_TABNAME
    TABLES
      DD03P_TAB = LT_DD03P
    EXCEPTIONS
      OTHERS    = 1.
  IF SY-SUBRC <> 0 OR LT_DD03P IS INITIAL.
    RETURN.
  ENDIF.

  LOOP AT LT_DD03P INTO LS_DD03P.
    IF LS_DD03P-FIELDNAME CP 'SID_*'.
      IO = LS_DD03P-FIELDNAME+4.
      CONDENSE IO NO-GAPS.
      IF IO IS INITIAL. CONTINUE. ENDIF.

      " Namespace: 0* -> /BI0/, else -> /BIC/
      IF IO CP '0*'.
        PREF = '/BI0/'.
      ELSE.
        PREF = '/BIC/'.
      ENDIF.

      " P<IOBJ>
      CONCATENATE PREF 'P' IO INTO TAB_P.
      CLEAR LV_DUMMY.
      SELECT SINGLE TABNAME INTO LV_DUMMY FROM DD02L WHERE TABNAME = TAB_P.
      IF SY-SUBRC = 0.
        INSERT TAB_P INTO TABLE GT_DDL_TABS.
      ENDIF.

      " T<IOBJ>
      CONCATENATE PREF 'T' IO INTO TAB_T.
      CLEAR LV_DUMMY.
      SELECT SINGLE TABNAME INTO LV_DUMMY FROM DD02L WHERE TABNAME = TAB_T.
      IF SY-SUBRC = 0.
        INSERT TAB_T INTO TABLE GT_DDL_TABS.
      ENDIF.

      " H<IOBJ>*  (multiple possible)

" Resolve PSA physical table by DS + LOGSYS via RSDSSEG / RSTSODS (ODSNAME_TECH)
" in : PV_DS      plain DataSource (e.g., ZBWKNC3 / 0FI_AR_4)
"      PV_LOGSYS  ECC source system (e.g., ECCCLNT100) — REQUIRED
" out: PV_PSA_TAB /BIC/<technical> (or INITIAL if not f
FORM RESOLVE_PSA_TABLE_NAME_RSDSSEG USING    PV_DS      TYPE RSDS-DATASOURCE
PV_LOGSYS  TYPE RSSLOGSYS
CHANGING PV_PSA_TAB TYPE DD02L-TABNAME.

  DATA: LV_PSA_NAME TYPE RSODSNAME,
        LV_PSA_VER  TYPE CHAR3,   " use proper data element
        LV_TECH     TYPE RSODSTECH.

  CLEAR PV_PSA_TAB.
  IF PV_DS IS INITIAL OR PV_LOGSYS IS INITIAL.
    RETURN.  " caller will decide fallback
  ENDIF.

  " RSDSSEG: PSA name + version for (DS, LOGSYS)
  CLEAR: LV_PSA_NAME, LV_PSA_VER.
  SELECT SINGLE PSA PSAVERS
    INTO (LV_PSA_NAME, LV_PSA_VER)
    FROM RSDSSEG
   WHERE DATASOURCE = PV_DS
     AND LOGSYS     = PV_LOGSYS
     AND OBJVERS    = 'A'.
  IF SY-SUBRC <> 0 OR LV_PSA_NAME IS INITIAL.
    SELECT SINGLE PSA PSAVERS
      INTO (LV_PSA_NAME, LV_PSA_VER)
      FROM RSDSSEG
     WHERE DATASOURCE = PV_DS
       AND LOGSYS     = PV_LOGSYS
       AND OBJVERS    = 'M'.
  ENDIF.
  IF LV_PSA_NAME IS INITIAL OR LV_PSA_VER IS INITIAL.
    RETURN.
  ENDIF.

  " RSTSODS: technical DB core for (PSA, VERSION)
  CLEAR LV_TECH.
  SELECT SINGLE ODSNAME_TECH
    INTO LV_TECH
    FROM RSTSODS
   WHERE ODSNAME = LV_PSA_NAME
     AND VERSION = LV_PSA_VER
     AND OBJSTAT = 'ACT'.
  IF SY-SUBRC <> 0 OR LV_TECH IS INITIAL.
    SELECT SINGLE ODSNAME_TECH
      INTO LV_TECH
      FROM RSTSODS
     WHERE ODSNAME = LV_PSA_NAME
       AND VERSION = LV_PSA_VER.
  ENDIF.
  IF LV_TECH IS INITIAL.
    RETURN.
  ENDIF.

  " *** IMPORTANT: RETURN the resolved PSA table to caller ***
  " LV_TECH is the technical core name (e.g., B0001014000). Prefix with /BIC/.
  IF LV_TECH(5) = '/BIC/'.
    " safety: in case some system already stores the full name (rare)
    PV_PSA_TAB = LV_TECH.
  ELSE.
    CONCATENATE '/BIC/' LV_TECH INTO PV_PSA_TAB.
  ENDIF.

ENDFORM.

" Analyze class LCL_TRANSFORM METHOD execute and its FORMULA__r_s methods.
" Enriches PT_DET (GT_DETAILS) with CONSTANT_VAL / FORMULA_TEXT / MASTER_SOURCE.
FORM RX_ANALYZE_EXECUTE USING    PV_PROG TYPE PROGNAME
                        CHANGING PT_DET  TYPE TY_DETAIL_TAB.

  " -------- Declarations --------
  DATA: LT_SRC  TYPE STANDARD TABLE OF STRING,
        LV_OK   TYPE C,
        LT_EXEC TYPE STANDARD TABLE OF STRING,
        LT_FULL TYPE STANDARD TABLE OF STRING,
        LT_SEEN TYPE TY_SEEN_TAB.

  TYPES: BEGIN OF TY_V2F,
           VAR   TYPE STRING,
           FIELD TYPE STRING,
         END OF TY_V2F.
  DATA LT_V2F TYPE STANDARD TABLE OF TY_V2F WITH EMPTY KEY.

  TYPES: BEGIN OF TY_V2M,
           VAR  TYPE STRING,
           METH TYPE STRING,
         END OF TY_V2M.
  DATA LT_V2M TYPE STANDARD TABLE OF TY_V2M WITH EMPTY KEY.

  TYPES: BEGIN OF TY_V2E,
           VAR   TYPE STRING,
           EXPR  TYPE STRING,
           MDTAB TYPE STRING,
         END OF TY_V2E.
  DATA LT_V2E TYPE STANDARD TABLE OF TY_V2E WITH EMPTY KEY.

  DATA: LV_LINE  TYPE STRING,
        LV_FLD   TYPE STRING,
        LV_VAR   TYPE STRING,
        LV_METH  TYPE STRING,
        LV_EXPR  TYPE STRING,
        LV_MDTAB TYPE STRING.

  DATA: IN_CALL TYPE C VALUE ' ',
        CALLBUF TYPE STRING.

  PERFORM FETCH_REPORT_SOURCE TABLES LT_SRC USING PV_PROG CHANGING LV_OK.

  " Fallback: try state-less read if LV_OK is still initial
  IF LV_OK IS INITIAL OR LT_SRC IS INITIAL.
    READ REPORT PV_PROG INTO LT_SRC. " state-less (kernel-dependent but useful in GP cases)
    IF SY-SUBRC = 0 AND LT_SRC IS NOT INITIAL.
      LV_OK = 'X'.
    ENDIF.
  ENDIF.

  " Only give up if the buffer is really empty
  IF LT_SRC IS INITIAL.
    RETURN.
  ENDIF.

  REFRESH: LT_EXEC, LT_FULL, LT_SEEN.
  PERFORM RX_EXTRACT_METHOD_BODY TABLES LT_SRC LT_EXEC USING 'execute'.
  IF LT_EXEC IS INITIAL.
    RETURN.
  ENDIF.

  PERFORM RX_EXPAND_INCLUDES TABLES LT_EXEC LT_FULL CHANGING LT_SEEN.
  IF LT_FULL IS INITIAL.
    LT_FULL = LT_EXEC.
  ENDIF.

  " -------- 1) Direct patterns in EXECUTE --------
  "     - RESULT_FIELDS-<field> = 'const' | <expr>
  "     - SELECT ... FROM <table>
  PERFORM RX_SCAN_START_BODY TABLES LT_FULL CHANGING PT_DET.

  " -------- 2) Map temp var -> target field (new GP patterns) --------
  " Build LT_V2F (VAR -> FIELD) from:
  "   a) _G_1-<field> = <_R_xxx>.
  "   b) CONVERSION_EXIT ... INPUT <_R_xxx> ... OUTPUT _G_1-<field>.
  REFRESH LT_V2F.

  DATA : LS_V2F_2 TYPE TY_V2F.

  " 2.a  _G_1-<field> = _R_... .
  LOOP AT LT_FULL INTO LV_LINE.
    CLEAR: LV_FLD, LV_VAR.
    FIND REGEX '^\s*_G_1-\s*([A-Za-z0-9_/]+)\s*=\s*([A-Za-z0-9_\-/]+)\s*\.'
         IN LV_LINE IGNORING CASE SUBMATCHES LV_FLD LV_VAR.
    IF SY-SUBRC = 0 AND LV_FLD IS NOT INITIAL AND LV_VAR CP '_R_*'.
      CLEAR LS_V2F_2.
      LS_V2F_2-VAR   = LV_VAR.
      LS_V2F_2-FIELD = LV_FLD.
      APPEND LS_V2F_2 TO LT_V2F.
    ENDIF.
  ENDLOOP.

  " 2.b  CONVERSION_EXIT multi-line: pair INPUT _R_*  with  OUTPUT _G_1-<field>
  DATA: IN_CONV   TYPE C,
        INPUT_VAR TYPE STRING,
        OUT_FLD   TYPE STRING.

  CLEAR: IN_CONV, INPUT_VAR, OUT_FLD.

  LOOP AT LT_FULL INTO LV_LINE.

    " enter a CONVERSION_EXIT ... block (ALLOW both CALL FUNCTION and CALL '...')
    IF IN_CONV IS INITIAL.
      FIND REGEX '^\s*CALL\s+(FUNCTION\s+)?''CONVERSION_EXIT_[A-Za-z0-9_]+'
           IN LV_LINE IGNORING CASE.
      IF SY-SUBRC = 0.
        IN_CONV = 'X'.
        CLEAR: INPUT_VAR, OUT_FLD.
      ENDIF.
    ENDIF.

    IF IN_CONV = 'X'.
      " capture INPUT var
      IF INPUT_VAR IS INITIAL.
        FIND REGEX 'ID\s+''INPUT''\s+FIELD\s+([A-Za-z0-9_\-/]+)'
             IN LV_LINE IGNORING CASE SUBMATCHES LV_VAR.
        IF SY-SUBRC = 0.
          INPUT_VAR = LV_VAR.
        ENDIF.
      ENDIF.

      " capture OUTPUT target (G_1-<field>)
      IF OUT_FLD IS INITIAL.
        FIND REGEX 'ID\s+''OUTPUT''\s+FIELD\s+_G_1-([A-Za-z0-9_/]+)'
             IN LV_LINE IGNORING CASE SUBMATCHES LV_FLD.
        IF SY-SUBRC = 0.
          OUT_FLD = LV_FLD.
        ENDIF.
      ENDIF.

      " end of the CALL ... block — the period closes it
      IF LV_LINE CS '.'.
        IF INPUT_VAR CP '_R_*' AND OUT_FLD IS NOT INITIAL.
          CLEAR LS_V2F_2.
          LS_V2F_2-VAR   = INPUT_VAR.
          LS_V2F_2-FIELD = OUT_FLD.
          APPEND LS_V2F_2 TO LT_V2F.
        ENDIF.
        CLEAR: IN_CONV, INPUT_VAR, OUT_FLD.
      ENDIF.
    ENDIF.

  ENDLOOP.

  DELETE ADJACENT DUPLICATES FROM LT_V2F COMPARING VAR FIELD.

  " -------- 2.x  Enrich PT_DET with DIRECT, CONSTANT, MASTER ----------
  FIELD-SYMBOLS <DET> TYPE TY_DETAIL.
  DATA LS_DET TYPE TY_DETAIL.

  " 2.x.a  DIRECT: _G_1-<tgt> = <_YS_SC_1>-<src>.
  LOOP AT LT_FULL INTO LV_LINE.
    CLEAR: LV_FLD.
    DATA LV_SRC TYPE STRING.
    FIND REGEX '^\s*_G_1-\s*([A-Za-z0-9_/]+)\s*=\s*<\s*_YS_SC_1\s*>\s*-\s*([A-Za-z0-9_/]+)\s*\.'
         IN LV_LINE IGNORING CASE SUBMATCHES LV_FLD LV_SRC.
    IF SY-SUBRC = 0 AND LV_FLD IS NOT INITIAL.
      READ TABLE PT_DET ASSIGNING <DET> WITH KEY TARGET_FIELD = LV_FLD.
      IF SY-SUBRC <> 0.
        CLEAR LS_DET. LS_DET-TARGET_FIELD = LV_FLD.
        INSERT LS_DET INTO TABLE PT_DET ASSIGNING <DET>.
      ENDIF.
      IF <DET> IS ASSIGNED.
        <DET>-SOURCE_FIELD = LV_SRC.
        <DET>-ROUTINE_FLAG = 'X'.
      ENDIF.
    ENDIF.
  ENDLOOP.

  " 2.x.b  CONSTANT: _G_1-<tgt> = 'literal'.
  LOOP AT LT_FULL INTO LV_LINE.
    CLEAR: LV_FLD.
    DATA LV_C TYPE STRING.
    FIND REGEX '^\s*_G_1-\s*([A-Za-z0-9_/]+)\s*=\s*''([^'']*)''\s*\.'
         IN LV_LINE IGNORING CASE SUBMATCHES LV_FLD LV_C.
    IF SY-SUBRC = 0 AND LV_FLD IS NOT INITIAL.
      READ TABLE PT_DET ASSIGNING <DET> WITH KEY TARGET_FIELD = LV_FLD.
      IF SY-SUBRC <> 0.
        CLEAR LS_DET. LS_DET-TARGET_FIELD = LV_FLD.
        INSERT LS_DET INTO TABLE PT_DET ASSIGNING <DET>.
      ENDIF.
      IF <DET> IS ASSIGNED AND <DET>-CONSTANT_VAL IS INITIAL.
        <DET>-CONSTANT_VAL = LV_C.
        <DET>-ROUTINE_FLAG = 'X'.
      ENDIF.
    ENDIF.
  ENDLOOP.

  " Discover MVIEW table once (e.g., /BI0/MMATERIAL) to tag master source
  DATA LV_MVIEW_TAB TYPE STRING.
  CLEAR LV_MVIEW_TAB.
  LOOP AT LT_SRC INTO LV_LINE.
    FIND REGEX 'MVIEW\s+TYPE\s+([A-Za-z0-9_/]+)' IN LV_LINE IGNORING CASE SUBMATCHES LV_MVIEW_TAB.
    IF SY-SUBRC = 0 AND LV_MVIEW_TAB IS NOT INITIAL.
      EXIT.
    ENDIF.
  ENDLOOP.

  " 2.x.c  MASTER: _G_1-<tgt> = _B_1-MVIEW-<attr>.
  LOOP AT LT_FULL INTO LV_LINE.
    CLEAR: LV_FLD.
    DATA LV_ATTR TYPE STRING.
    FIND REGEX '^\s*_G_1-\s*([A-Za-z0-9_/]+)\s*=\s*_B_1-MVIEW-([A-Za-z0-9_]+)\s*\.'
         IN LV_LINE IGNORING CASE SUBMATCHES LV_FLD LV_ATTR.
    IF SY-SUBRC = 0 AND LV_FLD IS NOT INITIAL.
      READ TABLE PT_DET ASSIGNING <DET> WITH KEY TARGET_FIELD = LV_FLD.
      IF SY-SUBRC <> 0.
        CLEAR LS_DET. LS_DET-TARGET_FIELD = LV_FLD.
        INSERT LS_DET INTO TABLE PT_DET ASSIGNING <DET>.
      ENDIF.
      IF <DET> IS ASSIGNED AND <DET>-MASTER_SOURCE IS INITIAL.
        IF LV_MVIEW_TAB IS INITIAL.
          <DET>-MASTER_SOURCE = 'READ TABLE ...'.
        ELSE.
          <DET>-MASTER_SOURCE = LV_MVIEW_TAB.
        ENDIF.
        <DET>-ROUTINE_FLAG = 'X'.
      ENDIF.
    ENDIF.
  ENDLOOP.
  " -------- 3) Map temp var -> FORMULA__r_s (accumulate lines until '.') --------
  CLEAR: IN_CALL, CALLBUF.
  LOOP AT LT_FULL INTO LV_LINE.
    IF IN_CALL IS INITIAL.
      " Start of a call when a line contains the token 'FORMULA__'
      IF LV_LINE CS 'FORMULA__'.
        IN_CALL = 'X'.
        CALLBUF = LV_LINE.
        IF LV_LINE CS '.'.
          IN_CALL = ' '.
        ENDIF.
      ENDIF.
    ELSE.
      CONCATENATE CALLBUF ' ' LV_LINE INTO CALLBUF.
      IF LV_LINE CS '.'.
        IN_CALL = ' '.
      ENDIF.
    ENDIF.

    IF IN_CALL IS INITIAL AND CALLBUF IS NOT INITIAL.
      CLEAR: LV_METH, LV_VAR.
      " Take method token FORMULA__r_s (no objref / no CALL keyword needed)
      FIND REGEX '(FORMULA__([0-9]+)_([0-9]+))'
           IN CALLBUF IGNORING CASE SUBMATCHES LV_METH DATA(D1) DATA(D2).
      " Find 'result = <var>' part
      FIND REGEX 'result\s*=\s*([-A-Za-z0-9_/]+)'
           IN CALLBUF IGNORING CASE SUBMATCHES LV_VAR.
      IF LV_METH IS NOT INITIAL AND LV_VAR IS NOT INITIAL.
        DATA LS_V2M TYPE TY_V2M.
        TRANSLATE LV_METH TO LOWER CASE.      " -> formula__r_s
        LS_V2M-METH = LV_METH.
        LS_V2M-VAR  = LV_VAR.
        APPEND LS_V2M TO LT_V2M.
      ENDIF.
      CLEAR CALLBUF.
    ENDIF.
  ENDLOOP.

  " -------- 4) Open each formula__r_s, read RESULT = <expr> and FROM <tab> -----
  DATA LS_V2M_1 TYPE TY_V2M.
  LOOP AT LT_V2M INTO LS_V2M_1.
    DATA: LT_MB    TYPE STANDARD TABLE OF STRING,
          LT_MFULL TYPE STANDARD TABLE OF STRING,
          LT_SEEN2 TYPE TY_SEEN_TAB.
    REFRESH: LT_MB, LT_MFULL, LT_SEEN2.

    PERFORM RX_EXTRACT_METHOD_BODY TABLES LT_SRC LT_MB USING LS_V2M_1-METH.
    IF LT_MB IS INITIAL.
      CONTINUE.
    ENDIF.
    PERFORM RX_EXPAND_INCLUDES TABLES LT_MB LT_MFULL CHANGING LT_SEEN2.
    IF LT_MFULL IS INITIAL.
      LT_MFULL = LT_MB.
    ENDIF.

    " --- 4.a Capture the COMPLETE body of the FORMULA__r_s method
    " Lines joined with [ | ] – one CSV cell per field.
    " Blank lines and DATA/TYPES declarations stripped.
    CLEAR LV_EXPR.
    DATA LV_BODY_LINE  TYPE STRING.
    DATA LV_BODY_UPPER TYPE STRING.
    LOOP AT LT_MFULL INTO LV_BODY_LINE.
      LV_BODY_UPPER = LV_BODY_LINE.
      TRANSLATE LV_BODY_UPPER TO UPPER CASE.
      CONDENSE LV_BODY_UPPER.
      IF LV_BODY_UPPER IS INITIAL. CONTINUE. ENDIF.
      IF LV_BODY_UPPER CP 'DATA *' OR LV_BODY_UPPER CP 'TYPES *'.
        CONTINUE.
      ENDIF.
      IF LV_EXPR IS INITIAL.
        LV_EXPR = LV_BODY_LINE.
      ELSE.
        CONCATENATE LV_EXPR ' | ' LV_BODY_LINE INTO LV_EXPR.
      ENDIF.
    ENDLOOP.

    " 4.b Scan body for master-data table reference
    CLEAR LV_MDTAB.
    DATA LV_RESVAR TYPE STRING.  " compatibility placeholder
    CLEAR LV_RESVAR.
    LOOP AT LT_MFULL INTO LV_LINE.
      FIND REGEX 'FROM\s+([A-Za-z0-9_/]+)'
           IN LV_LINE IGNORING CASE SUBMATCHES LV_MDTAB.
      IF SY-SUBRC = 0 AND LV_MDTAB IS NOT INITIAL. EXIT. ENDIF.
    ENDLOOP.

    DATA LS_V2E TYPE TY_V2E.
    LS_V2E-VAR   = LS_V2M_1-VAR.
    LS_V2E-EXPR  = LV_EXPR.   " full FORMULA__r_s body
    LS_V2E-MDTAB = LV_MDTAB.
    APPEND LS_V2E TO LT_V2E.
  ENDLOOP.

  " -------- 5) Merge onto PT_DET by target field ----------------------
  DATA LS_V2F_1 TYPE TY_V2F.
  LOOP AT LT_V2F INTO LS_V2F_1.
    READ TABLE LT_V2E INTO DATA(LS_E) WITH KEY VAR = LS_V2F_1-VAR.
    IF SY-SUBRC <> 0.
      CONTINUE.
    ENDIF.

    FIELD-SYMBOLS <DET1> TYPE TY_DETAIL.
    READ TABLE PT_DET ASSIGNING <DET1> WITH KEY TARGET_FIELD = LS_V2F_1-FIELD.
    IF SY-SUBRC <> 0.
*      DATA LS_DET TYPE TY_DETAIL.
      CLEAR LS_DET.
      LS_DET-TARGET_FIELD = LS_V2F_1-FIELD.
      INSERT LS_DET INTO TABLE PT_DET ASSIGNING <DET1>.
    ENDIF.

    IF <DET1> IS ASSIGNED.
      <DET1>-ROUTINE_FLAG = 'X'.

      IF <DET1>-FORMULA_TEXT IS INITIAL AND LS_E-EXPR IS NOT INITIAL.
        <DET1>-FORMULA_TEXT = LS_E-EXPR.
      ENDIF.

      IF <DET1>-MASTER_SOURCE IS INITIAL AND LS_E-MDTAB IS NOT INITIAL.
        <DET1>-MASTER_SOURCE = LS_E-MDTAB.
      ENDIF.

      " Pure quoted literal -> constant
      IF <DET1>-CONSTANT_VAL IS INITIAL AND LS_E-EXPR IS NOT INITIAL.
        DATA LVLEN        TYPE I.
        DATA LV_OFF_LAST  TYPE I.
        DATA LV_INNER_LEN TYPE I.
        DATA LV_TMP_CONST TYPE STRING.
        CLEAR: LVLEN, LV_OFF_LAST, LV_INNER_LEN, LV_TMP_CONST.
        LVLEN = STRLEN( LS_E-EXPR ).
        IF LVLEN GE 2.
          LV_OFF_LAST = LVLEN - 1.
          IF LS_E-EXPR+0(1) = '''' AND LS_E-EXPR+LV_OFF_LAST(1) = ''''.
            LV_INNER_LEN = LVLEN - 2.
            IF LV_INNER_LEN GT 0.
              LV_TMP_CONST = LS_E-EXPR+1(LV_INNER_LEN).
              <DET1>-CONSTANT_VAL = LV_TMP_CONST.
            ENDIF.
          ENDIF.
        ENDIF.
      ENDIF.
    ENDIF.
  ENDLOOP.

ENDFORM.

*---------------------------------------------------------------------*
* GET_KEYFLAG – read DD03L-KEYFLAG for a (table, field)
* Drop-in replacement: auto-normalizes field for DSO Active tables.
* IN : PV_TAB   TYPE DD02L-TABNAME   (physical table name)
*      PV_FIELD TYPE FIELDNAME       (field name, may miss '/BIC/' prefix)
* OUT: PV_KEYFLAG TYPE C             ('X' if part of the table key)
*---------------------------------------------------------------------*
FORM GET_KEYFLAG USING    PV_TAB    TYPE DD02L-TABNAME
                          PV_FIELD  TYPE FIELDNAME
                 CHANGING PV_KEYFLAG TYPE C.

  DATA: LV_FIELD     TYPE FIELDNAME,
        LV_ORIG      TYPE FIELDNAME,
        LV_KEY       TYPE C,
        LV_BIC_ADDED TYPE C.

  CLEAR PV_KEYFLAG.
  IF PV_TAB IS INITIAL OR PV_FIELD IS INITIAL.
    RETURN.
  ENDIF.

  LV_FIELD = PV_FIELD.
  LV_ORIG  = PV_FIELD.
  CLEAR LV_BIC_ADDED.

  " Normalize for DSO Active tables only: /BIC/A<dso>00
  " PSA (/BIC/B*), Cube (/BIC/F*|/BIC/E*), etc. are left as-is.
  IF PV_TAB CP '/BIC/A*'.
    " If no namespace prefix yet, add /BIC/ (do not touch /BI0/ or already /BIC/)


    " --- Add namespace prefix if missing
    IF LV_FIELD+0(1) = '0'.
      SHIFT LV_FIELD LEFT BY 1 PLACES.   " drops the first char
    ELSEIF LV_FIELD+0(5) <> '/BIC/' AND LV_FIELD+0(5) <> '/BI0/'.
      IF LV_FIELD+0(1) = '0'.
        " 0* InfoObjects live under /BI0/
        CONCATENATE '/BI0/' LV_FIELD INTO LV_FIELD.
      ELSE.
        " customer namespace under /BIC/
        CONCATENATE '/BIC/' LV_FIELD INTO LV_FIELD.
      ENDIF.
    ENDIF.
  ENDIF.

  " Try normalized first
  SELECT SINGLE KEYFLAG
    FROM DD03L
    INTO @LV_KEY
   WHERE TABNAME   = @PV_TAB
     AND FIELDNAME = @LV_FIELD.

  " Fallback: original (in case the table truly stores it without /BIC/)
  IF SY-SUBRC <> 0 AND LV_BIC_ADDED = 'X'.
    SELECT SINGLE KEYFLAG
      FROM DD03L
      INTO @LV_KEY
     WHERE TABNAME   = @PV_TAB
       AND FIELDNAME = @LV_ORIG.
  ENDIF.

  PV_KEYFLAG = LV_KEY.

ENDFORM.

FORM COLLECT_ACTIVITY_PROVIDERS.
  DATA: LS_F    TYPE TY_FLOW,
        LS_PAIR TYPE TY_DS_LS,
        LV      TYPE STRING,
        LV_DSO  TYPE RSDODSO-ODSOBJECT,
        LV_CUBE TYPE RSDCUBE-INFOCUBE,
        LV_DS   TYPE RSDS-DATASOURCE.

  REFRESH GT_LOAD.

  LOOP AT GT_FLOW INTO LS_F.
    " --- DSO (target)
    IF LS_F-TARGET_TYPE = 'DSO'.
      APPEND INITIAL LINE TO GT_LOAD ASSIGNING <L>.
      <L>-PROVIDER       = LS_F-TARGET.
      <L>-PROVIDER_TYPE  = 'DSO'.
      <L>-PHYSICAL_TABLE = LS_F-T_TABLE_NAME.
    ENDIF.

    " --- DSO (source)
    IF LS_F-SOURCE_TYPE = 'DSO'.
      APPEND INITIAL LINE TO GT_LOAD ASSIGNING <L>.
      <L>-PROVIDER       = LS_F-SOURCE.
      <L>-PROVIDER_TYPE  = 'DSO'.
      <L>-PHYSICAL_TABLE = LS_F-S_TABLE_NAME.

      <L>-PROVIDER       = LS_F-SOURCE.
      <L>-PROVIDER_TYPE  = 'CUBE'.
      <L>-PHYSICAL_TABLE = LS_F-S_TABLE_NAME.
    ENDIF.

    " --- PSA (from DataSource + LOGSYS) - only if physical PSA table resolved
    IF LS_F-SOURCE_TYPE = 'DATASOURCE' OR LS_F-SOURCE_TYPE = 'TRANSFER STRUCTURE'.
      " Skip InfoSource segments misclassified as DS (no physical PSA table)

        <L>-PROVIDER_TYPE  = 'PSA'.
        <L>-SOURCE_LOGSYS  = LS_F-SOURCE_LOGSYS.
        <L>-PHYSICAL_TABLE = LS_F-S_TABLE_NAME.
      ENDIF.
    ENDIF.

    " InfoSource entries skipped - not actionable in load activity context
  ENDLOOP.

  " Deduplicate by business keys
  SORT GT_LOAD BY PROVIDER PROVIDER_TYPE SOURCE_LOGSYS.
  DELETE ADJACENT DUPLICATES FROM GT_LOAD COMPARING PROVIDER PROVIDER_TYPE SOURCE_LOGSYS.
ENDFORM.

FORM RESOLVE_LAST_LOAD_FOR_ALL.
  DATA LS TYPE TY_LOAD.

  LOOP AT GT_LOAD INTO LS.
    SET RUN TIME CLOCK RESOLUTION LOW.
    CASE LS-PROVIDER_TYPE.
      WHEN 'DSO'.
        PERFORM RESOLVE_LAST_LOAD_DSO USING LS-PROVIDER CHANGING LS-LAST_TS_UTC.
      WHEN 'CUBE'.
        PERFORM RESOLVE_LAST_LOAD_CUBE USING LS-PROVIDER CHANGING LS-LAST_TS_UTC.
      WHEN 'PSA'.
        PERFORM RESOLVE_LAST_LOAD_PSA USING LS-PROVIDER LS-SOURCE_LOGSYS CHANGING LS-LAST_TS_UTC.
    ENDCASE.
    PERFORM NORMALIZE_TS_AND_AGE CHANGING LS.
    MODIFY GT_LOAD FROM LS.
  ENDLOOP.
ENDFORM.

FORM RESOLVE_LAST_LOAD_DSO USING    PV_DSO TYPE RSDODSO-ODSOBJECT
                           CHANGING PV_TS  TYPE RSTIMESTMP.
  " RSSTATMANPART covers both DTP and InfoPackage loads in one table
  DATA: LV_DAT TYPE DATS,
        LV_TIM TYPE TIMS.
  CLEAR PV_TS.

  " 1) DTP path: RSBKDTP -> RSBKREQUEST
  DATA: LT_DTP TYPE STANDARD TABLE OF RSBKDTP-DTP WITH DEFAULT KEY,
        LV_DTP TYPE RSBKDTP-DTP,
        LV_TS  TYPE RSTIMESTMP.
  SELECT DTP INTO TABLE LT_DTP FROM RSBKDTP
    WHERE TGT = PV_DSO AND OBJVERS = 'A'.
  LOOP AT LT_DTP INTO LV_DTP.
    CLEAR LV_TS.
    SELECT MAX( TSTMP_H ) INTO LV_TS FROM RSBKREQUEST
      WHERE DTP = LV_DTP AND TSTATE NOT IN ( '4', '9' ).
    IF LV_TS > PV_TS. PV_TS = LV_TS. ENDIF.
  ENDLOOP.

  " 2) InfoPackage fallback: RSSTATMANPART by DTA = DSO name
  IF PV_TS IS INITIAL.
    SELECT MAX( DATUM_ANF ) INTO LV_DAT FROM RSSTATMANPART
      WHERE DTA = PV_DSO AND STORNOFLAG <> 'X'.
    IF LV_DAT IS NOT INITIAL.
      SELECT MAX( ZEIT_ANF ) INTO LV_TIM FROM RSSTATMANPART
        WHERE DTA = PV_DSO AND STORNOFLAG <> 'X' AND DATUM_ANF = LV_DAT.
      CONVERT DATE LV_DAT TIME LV_TIM INTO TIME STAMP PV_TS TIME ZONE 'UTC'.
    ENDIF.
  ENDIF.
ENDFORM.

  DATA: LV_DAT TYPE DATS,

FORM NORMALIZE_TS_AND_AGE CHANGING PS TYPE TY_LOAD.
  DATA: LV_ZONE  TYPE TTZZ-TZONE,
        LV_DATE  TYPE DATS,
        LV_TIME  TYPE TIMS,
        LV_TODAY TYPE DATS,
        LV_DAYS  TYPE I.

  IF PS-LAST_TS_UTC IS INITIAL.
    PS-LAST_DATE_LOCAL = '00000000'.
    PS-AGE_BUCKET      = '7Y+'.
    PS-NOTE            = 'No successful request found'.
    RETURN.
  ENDIF.

  LV_ZONE = P_TZONE.
  IF LV_ZONE IS INITIAL.
    LV_ZONE = SY-ZONLO.
  ENDIF.

  " Convert UTC TS → local date/time (kernel instruction)
  CONVERT TIME STAMP PS-LAST_TS_UTC TIME ZONE LV_ZONE INTO DATE LV_DATE TIME LV_TIME.
  PS-LAST_DATE_LOCAL = LV_DATE.
  PS-LAST_TIME_LOCAL = LV_TIME.

  LV_TODAY = SY-DATUM.
  LV_DAYS  = LV_TODAY - LV_DATE.
  PS-AGE_DAYS = LV_DAYS.

  IF LV_DAYS < 365.
    PS-AGE_BUCKET = '<1Y'.
  ELSEIF LV_DAYS < 3 * 365.
    PS-AGE_BUCKET = '1Y+'.
  ELSEIF LV_DAYS < 5 * 365.
    PS-AGE_BUCKET = '3Y+'.
  ELSEIF LV_DAYS < 7 * 365.
    PS-AGE_BUCKET = '5Y+'.
  ELSE.
    PS-AGE_BUCKET = '7Y+'.
  ENDIF.
ENDFORM.


FORM BUILD_INACTIVE_LOG.
  DATA: LS     TYPE TY_LOAD,
        LW     TYPE TY_INACTIVE,
        DSO_A  TYPE RSDODSO-ODSOBJECT,
        CUBE_A TYPE RSDCUBE-INFOCUBE.

  "========================
  " 1) DSOs: active check
  "========================
  LOOP AT GT_LOAD INTO LS WHERE PROVIDER_TYPE = 'DSO'.
    CLEAR DSO_A.
    SELECT SINGLE ODSOBJECT INTO DSO_A FROM RSDODSO
      WHERE ODSOBJECT = LS-PROVIDER AND OBJVERS = 'A'.
    IF SY-SUBRC <> 0.
      CLEAR LW. LW-KIND = 'DSO'. LW-NAME = LS-PROVIDER.
      LW-DETAIL = 'No active (A) version in RSDODSO'. APPEND LW TO GT_INACTIVE.
    ENDIF.
  ENDLOOP.

  "=========================
  " 2) Cubes: active check
  "=========================
  LOOP AT GT_LOAD INTO LS WHERE PROVIDER_TYPE = 'CUBE'.
    CLEAR CUBE_A.
    SELECT SINGLE INFOCUBE INTO CUBE_A FROM RSDCUBE
      WHERE INFOCUBE = LS-PROVIDER AND OBJVERS = 'A'.
    IF SY-SUBRC <> 0.
      CLEAR LW. LW-KIND = 'CUBE'. LW-NAME = LS-PROVIDER.
      LW-DETAIL = 'No active (A) version in RSDCUBE'. APPEND LW TO GT_INACTIVE.
    ENDIF.
  ENDLOOP.

  "===============================================================
  " 3) Transformations touching any lineage provider
  "    (avoid IN @itab — loop providers and select per provider)
  "===============================================================
  " Transformations touching any lineage provider – only A-version
  DATA: LT_PROV TYPE STANDARD TABLE OF CHAR30 WITH DEFAULT KEY,
        LV_PROV TYPE CHAR30,
        LS_TRAN TYPE RSTRAN.

  " collect unique providers
  LOOP AT GT_LOAD INTO LS.
    CLEAR LV_PROV.
    LV_PROV = LS-PROVIDER.
    INSERT LV_PROV INTO TABLE LT_PROV.
  ENDLOOP.
  DELETE ADJACENT DUPLICATES FROM LT_PROV.

  " Use preloaded GT_TRAN (loaded at start) — no DB per provider
  LOOP AT LT_PROV INTO LV_PROV.
    LOOP AT GT_TRAN INTO LS_TRAN
      WHERE ( SOURCENAME = LV_PROV OR TARGETNAME = LV_PROV ).
      IF LS_TRAN-OBJSTAT <> 'ACT'.
        CLEAR LW.
        LW-KIND = 'TRANSFORMATION'.
        LW-NAME = LS_TRAN-TRANID.
        CONCATENATE LS_TRAN-SOURCENAME '->' LS_TRAN-TARGETNAME
                    'is not active'
               INTO LW-DETAIL SEPARATED BY SPACE.
        APPEND LW TO GT_INACTIVE.
      ENDIF.
    ENDLOOP.
  ENDLOOP.

  " 4) Stale age entries moved to Issues column in Load Activity CSV
ENDFORM.

                LV_DAYS X-AGE_BUCKET X-NOTE LV_ISSUE

    PERFORM WRITE_FILE_FRONTEND TABLES LT USING F.
  ELSE.
    PERFORM WRITE_FILE_APPSERVER TABLES LT USING F.
  ENDIF.
ENDFORM.

*---------------------------------------------------------------------*
* RESOLVE_MD_TABLE
*  Resolves P-table (time-indep) or Q-table (time-dep) from
*  MASTER_SOURCE which may be a table name already (/BI0/P…)
*  or an InfoObject name (0MATERIAL, ZVENDOR …)
*---------------------------------------------------------------------*
FORM RESOLVE_MD_TABLE USING    PV_MDSRC TYPE STRING
                      CHANGING PV_PTAB  TYPE DD02L-TABNAME.
* Caller must pass a clean InfoObject name (e.g. 0MATERIAL, ZVENDOR).
* DERIVE_MD_IOBJ should be called before this form if the raw value
* is a BW internal table reference such as /BI0/MMATERIAL.
* Probes Q (time-dependent) then P (time-independent) tables in both
* /BI0/ (SAP-delivered) and /BIC/ (customer) namespaces.

  DATA LV_CAND TYPE DD02L-TABNAME.
  DATA LV_HIT  TYPE DD02L-TABNAME.
  DATA LV_IO   TYPE CHAR30.

  CLEAR PV_PTAB.
  IF PV_MDSRC IS INITIAL. RETURN. ENDIF.

  " Strip leading 0 (SAP namespace prefix) to build physical table name core
  " e.g. 0MATERIAL → MATERIAL  (BW convention: /BI0/PMATERIAL, not /BI0/P0MATERIAL)
  LV_IO = PV_MDSRC.
  CONDENSE LV_IO NO-GAPS.
  IF LV_IO(1) = '0'.
    LV_IO = LV_IO+1.
  ENDIF.

  DATA LT_CANDS TYPE STANDARD TABLE OF DD02L-TABNAME WITH DEFAULT KEY.
  DATA LV_TRY   TYPE DD02L-TABNAME.
  " Try Q (time-dependent) before P (time-independent); SAP namespace before customer
  CONCATENATE '/BI0/Q' LV_IO INTO LV_TRY. APPEND LV_TRY TO LT_CANDS.
  CONCATENATE '/BI0/P' LV_IO INTO LV_TRY. APPEND LV_TRY TO LT_CANDS.
  CONCATENATE '/BIC/Q' LV_IO INTO LV_TRY. APPEND LV_TRY TO LT_CANDS.
  CONCATENATE '/BIC/P' LV_IO INTO LV_TRY. APPEND LV_TRY TO LT_CANDS.

  LOOP AT LT_CANDS INTO LV_CAND.
    CLEAR LV_HIT.
    SELECT SINGLE TABNAME INTO LV_HIT FROM DD02L WHERE TABNAME = LV_CAND.
    IF SY-SUBRC = 0. PV_PTAB = LV_HIT. RETURN. ENDIF.
  ENDLOOP.

ENDFORM.

*---------------------------------------------------------------------*
* DERIVE_MD_IOBJ
*  Converts a raw BW master-data table reference into a clean
*  InfoObject name with SAP namespace prefix (0...).
*
*  Examples:
*    /BI0/MMATERIAL  → 0MATERIAL   (M = SID master table)
*    /BI0/VMATERIAL  → 0MATERIAL   (V = text table)
*    /BIC/MZVENDOR   → 0ZVENDOR
*    0MATERIAL       → 0MATERIAL   (already clean, pass through)
*    MATERIAL        → 0MATERIAL   (no prefix, assume SAP NS)
*
*  Type-char stripping applies to: M V P Q H D (common BW suffixes).
*  If the name is already an InfoObject (starts with 0 or Z/Y without
*  a namespace prefix), it is returned as-is (with 0-prefix added if
*  the first char is not 0).
*---------------------------------------------------------------------*
FORM DERIVE_MD_IOBJ USING    PV_RAW  TYPE STRING
                    CHANGING PV_IOBJ TYPE STRING.

  DATA LV_WORK TYPE STRING.
  DATA LV_CORE TYPE STRING.

  CLEAR PV_IOBJ.
  LV_WORK = PV_RAW.
  CONDENSE LV_WORK NO-GAPS.

  IF LV_WORK IS INITIAL. RETURN. ENDIF.

  " Strip BW namespace prefix /BI0/ or /BIC/  (5 characters)
  IF LV_WORK CP '/BI0/*' OR LV_WORK CP '/BIC/*'.
    LV_CORE = LV_WORK+5.          " drop first 5 chars: /BI0/ or /BIC/
    " The next character is the BW type-char (M, V, P, Q, H, D, etc.)
    " Strip it to get the raw InfoObject name core
    IF LV_CORE(1) CA 'MVPQHD'.
      LV_CORE = LV_CORE+1.        " drop type char
    ENDIF.
    " Prepend SAP namespace prefix 0
    CONCATENATE '0' LV_CORE INTO PV_IOBJ.
  ELSE.
    " No namespace prefix: treat as InfoObject name directly
    LV_CORE = LV_WORK.
    IF LV_CORE(1) <> '0'.
      CONCATENATE '0' LV_CORE INTO PV_IOBJ.
    ELSE.
      PV_IOBJ = LV_CORE.
    ENDIF.
  ENDIF.

ENDFORM.
*  Full lineage traversal for a master data InfoObject.
*  Covers attribute, text, and hierarchy load chains down to DS.
*---------------------------------------------------------------------*
FORM PROCESS_MD_INFOOBJECT USING PV_IOBJ TYPE CHAR30.

  DATA: LT_TRAN_MD  TYPE STANDARD TABLE OF RSTRAN WITH EMPTY KEY,
        LS_TRAN     TYPE RSTRAN,
        LS_FLOW     TYPE TY_FLOW,
        LV_SRC      TYPE CHAR30,
        LV_SRC_D    TYPE RSDS-DATASOURCE,
        LV_DS_PLAIN TYPE RSDS-DATASOURCE,
        LV_PROG     TYPE PROGNAME,
        LV_ATTR     TYPE CHAR35,
        LV_TEXT     TYPE CHAR35,
        LV_HIER     TYPE CHAR35.

  " Always collect ALL variants: direct + _ATTR / _TEXT / _HIER
  " Direct (e.g. 0MATERIAL itself as target — covers key figure loads)
  SELECT * FROM RSTRAN INTO TABLE LT_TRAN_MD
    WHERE OBJVERS = 'A' AND TARGETNAME = PV_IOBJ.

  " Attribute load (0MATERIAL_ATTR)
  CONCATENATE PV_IOBJ '_ATTR' INTO LV_ATTR.
  SELECT * FROM RSTRAN APPENDING TABLE LT_TRAN_MD
    WHERE OBJVERS = 'A' AND TARGETNAME = LV_ATTR.

  " Text load (0MATERIAL_TEXT)
  CONCATENATE PV_IOBJ '_TEXT' INTO LV_TEXT.
  SELECT * FROM RSTRAN APPENDING TABLE LT_TRAN_MD
    WHERE OBJVERS = 'A' AND TARGETNAME = LV_TEXT.

  " Hierarchy load (0MATERIAL_HIER)
  CONCATENATE PV_IOBJ '_HIER' INTO LV_HIER.
  SELECT * FROM RSTRAN APPENDING TABLE LT_TRAN_MD
    WHERE OBJVERS = 'A' AND TARGETNAME = LV_HIER.

  IF LT_TRAN_MD IS INITIAL.
    CLEAR LS_FLOW.
    LS_FLOW-CONTEXT_PROVIDER = GV_CTX.
    LS_FLOW-TARGET            = PV_IOBJ.
    LS_FLOW-TARGET_TYPE       = 'MASTER DATA IOBJ'.
    LS_FLOW-SOURCE            = '(no transformation found)'.
    APPEND LS_FLOW TO GT_FLOW. RETURN.
  ENDIF.

  LOOP AT LT_TRAN_MD INTO LS_TRAN.
    CLEAR: LS_FLOW, LV_SRC, LV_SRC_D, LV_DS_PLAIN, LV_PROG.

    LS_FLOW-CONTEXT_PROVIDER = GV_CTX.
    LS_FLOW-TARGET           = LS_TRAN-TARGETNAME.
    LS_FLOW-TRANID           = LS_TRAN-TRANID.

    " Set target type using RSTRAN-TARGETTYPE (ATTR/TEXTS/HIER/ODS etc.)
    CASE LS_TRAN-TARGETTYPE.
      WHEN 'ATTR'.
        LS_FLOW-TARGET_TYPE = 'MD ATTRIBUTE'.
        " Ensure target shows as <IOBJ>_ATTR for clarity
        IF LS_TRAN-TARGETNAME = PV_IOBJ.
          CONCATENATE PV_IOBJ '_ATTR' INTO LS_FLOW-TARGET.
        ENDIF.
      WHEN 'TEXTS'.
        LS_FLOW-TARGET_TYPE = 'MD TEXT'.
        IF LS_TRAN-TARGETNAME = PV_IOBJ.
          CONCATENATE PV_IOBJ '_TEXT' INTO LS_FLOW-TARGET.
        ENDIF.
      WHEN 'HIER'.
        LS_FLOW-TARGET_TYPE = 'MD HIERARCHY'.
        IF LS_TRAN-TARGETNAME = PV_IOBJ.
          CONCATENATE PV_IOBJ '_HIER' INTO LS_FLOW-TARGET.
        ENDIF.
      WHEN OTHERS.
        " Fallback: use name matching
        IF LS_TRAN-TARGETNAME = LV_ATTR.
          LS_FLOW-TARGET_TYPE = 'MD ATTRIBUTE'.
        ELSEIF LS_TRAN-TARGETNAME = LV_TEXT.
          LS_FLOW-TARGET_TYPE = 'MD TEXT'.
        ELSEIF LS_TRAN-TARGETNAME = LV_HIER.
          LS_FLOW-TARGET_TYPE = 'MD HIERARCHY'.
        ELSE.
          LS_FLOW-TARGET_TYPE = 'MASTER DATA IOBJ'.
        ENDIF.
    ENDCASE.

    IF LS_TRAN-TRANPROG IS NOT INITIAL.
      IF LS_TRAN-TRANPROG CP 'GP*'.
        LV_PROG = LS_TRAN-TRANPROG.
      ELSE.
        CONCATENATE 'GP' LS_TRAN-TRANPROG INTO LV_PROG.
      ENDIF.
    ENDIF.
    LS_FLOW-PROGRAM = LV_PROG.

    LV_SRC = LS_TRAN-SOURCENAME.
    PERFORM GET_PROVIDER_TYPE USING LV_SRC LS_TRAN-SOURCETYPE
                              CHANGING LS_FLOW-SOURCE_TYPE.

    " Additional check: if TRANSFER STRUCTURE, verify against RSKSFIELDNEW
    IF LS_FLOW-SOURCE_TYPE = 'TRANSFER STRUCTURE'.
      DATA LV_IS_CHK_MD TYPE RSKSFIELDNEW-ISOURCE.
      DATA LV_IS_KEY_MD TYPE RSKSFIELDNEW-ISOURCE.
      LV_IS_KEY_MD = LV_SRC.
      SELECT SINGLE ISOURCE INTO LV_IS_CHK_MD FROM RSKSFIELDNEW
        WHERE ISOURCE = LV_IS_KEY_MD AND OBJVERS = 'A'.
      IF SY-SUBRC = 0.
        LS_FLOW-SOURCE_TYPE = 'INFOSOURCE'.
      ENDIF.
    ENDIF.

    LV_SRC_D = LV_SRC.
    PERFORM EXTRACT_PLAIN_DSNAME USING LV_SRC_D CHANGING LV_DS_PLAIN.
    IF LV_DS_PLAIN IS INITIAL. LV_DS_PLAIN = LV_SRC. ENDIF.
    LS_FLOW-SOURCE = LV_DS_PLAIN.

    IF LS_FLOW-SOURCE_TYPE = 'DATASOURCE' OR
       LS_FLOW-SOURCE_TYPE = 'TRANSFER STRUCTURE'.
      DATA LT_LS2 TYPE SORTED TABLE OF RSSLOGSYS WITH UNIQUE KEY TABLE_LINE.
      DATA LV_LS2 TYPE RSSLOGSYS.
      REFRESH LT_LS2.
      SELECT LOGSYS INTO TABLE LT_LS2 FROM RSDSSEG
        WHERE DATASOURCE = LV_DS_PLAIN AND OBJVERS = 'A'.
      IF LT_LS2 IS INITIAL.
        SELECT LOGSYS INTO TABLE LT_LS2 FROM RSDS
          WHERE DATASOURCE = LV_DS_PLAIN AND OBJVERS = 'A'.
      ENDIF.
      IF LT_LS2 IS INITIAL.
        PERFORM GET_TABLE_NAME
          USING    LS_FLOW-TARGET LS_FLOW-TARGET_TYPE
                   LS_FLOW-SOURCE LS_FLOW-SOURCE_TYPE
                   LS_FLOW-SOURCE_LOGSYS ''
          CHANGING LS_FLOW-S_TABLE_NAME LS_FLOW-T_TABLE_NAME.
        APPEND LS_FLOW TO GT_FLOW.
      ELSE.
        LOOP AT LT_LS2 INTO LV_LS2.
          DATA LS_ROW2 TYPE TY_FLOW.
          LS_ROW2 = LS_FLOW.
          LS_ROW2-SOURCE_LOGSYS = LV_LS2.
          PERFORM GET_TABLE_NAME
            USING    LS_ROW2-TARGET LS_ROW2-TARGET_TYPE
                     LS_ROW2-SOURCE LS_ROW2-SOURCE_TYPE
                     LS_ROW2-SOURCE_LOGSYS ''
            CHANGING LS_ROW2-S_TABLE_NAME LS_ROW2-T_TABLE_NAME.
          APPEND LS_ROW2 TO GT_FLOW.
        ENDLOOP.
      ENDIF.
      IF LS_FLOW-SOURCE_TYPE = 'TRANSFER STRUCTURE'.
        PERFORM PROCESS_PROVIDER USING LV_SRC.
      ENDIF.
      CONTINUE.
    ENDIF.

    PERFORM GET_TABLE_NAME
      USING    LS_FLOW-TARGET LS_FLOW-TARGET_TYPE
               LS_FLOW-SOURCE LS_FLOW-SOURCE_TYPE
               LS_FLOW-SOURCE_LOGSYS ''
      CHANGING LS_FLOW-S_TABLE_NAME LS_FLOW-T_TABLE_NAME.
    APPEND LS_FLOW TO GT_FLOW.
    PERFORM PROCESS_PROVIDER USING LV_SRC.
  ENDLOOP.

ENDFORM.
*======================================================================
*  DIRECT UPDATE DSO EXTRACTION
*  Tables used: RSDODSO, RSDODSOT, RSBKDTP, RSPCCHAIN, RSBKREQUEST, RSTRAN
*======================================================================
FORM EXTRACT_DIRECT_UPDATE_DSOS.

  DATA: LT_DSO   TYPE STANDARD TABLE OF RSDODSO WITH EMPTY KEY,
        LS_DSO   TYPE RSDODSO,
        LV_FILE  TYPE STRING,
        LV_LINE  TYPE STRING,
        LV_DESC  TYPE STRING,
        LV_ZONE  TYPE TTZZ-TZONE,
        LV_TS    TYPE RSTIMESTMP,
        LV_DATE  TYPE DATS,
        LV_TIME  TYPE TIMS,
        LV_ODSO  TYPE STRING,
        LV_DSONAME TYPE STRING.

  DATA LT_OUT TYPE STANDARD TABLE OF STRING WITH EMPTY KEY.

  IF GV_DUDS_DIR IS INITIAL. RETURN. ENDIF.

  " ODSOTYPE = '2' Write-Optimised, 'F' Direct Update
  SELECT * FROM RSDODSO INTO TABLE LT_DSO
    WHERE OBJVERS = 'A'
      AND ( ODSOTYPE = '2' OR ODSOTYPE = 'F' ).

  LOOP AT LT_DSO INTO LS_DSO.
    REFRESH LT_OUT.

    CONCATENATE '=== Direct Update DSO : ' LS_DSO-ODSOBJECT ' ===' INTO LV_LINE.
    APPEND LV_LINE TO LT_OUT.
    APPEND '' TO LT_OUT.

    CLEAR LV_DESC.
    SELECT SINGLE TXTLG INTO LV_DESC FROM RSDODSOT
      WHERE ODSOBJECT = LS_DSO-ODSOBJECT AND OBJVERS = 'A' AND LANGU = SY-LANGU.
    CONCATENATE 'Description  : ' LV_DESC INTO LV_LINE.
    APPEND LV_LINE TO LT_OUT.
    CONCATENATE 'DSO Type     : ' LS_DSO-ODSOTYPE INTO LV_LINE.
    APPEND LV_LINE TO LT_OUT.
    APPEND '' TO LT_OUT.

    APPEND '--- Loaders ---' TO LT_OUT.

    " 1) Open Hub / DTPs sourcing from this DSO
    DATA LT_OHD TYPE STANDARD TABLE OF RSBKDTP WITH EMPTY KEY.
    DATA LS_OHD TYPE RSBKDTP.
    SELECT * FROM RSBKDTP INTO TABLE LT_OHD
      WHERE SRC = LS_DSO-ODSOBJECT AND OBJVERS = 'A'.
    IF LT_OHD IS INITIAL.
      APPEND 'OpenHub/DTP  : (none found as source)' TO LT_OUT.
    ELSE.
      LOOP AT LT_OHD INTO LS_OHD.
        CONCATENATE 'OpenHub/DTP : ' LS_OHD-DTP ' -> ' LS_OHD-TGT INTO LV_LINE.
        APPEND LV_LINE TO LT_OUT.
      ENDLOOP.
    ENDIF.
    FREE LT_OHD.

    " 2) Process Chains via RSPCCHAIN
    DATA LT_PC TYPE STANDARD TABLE OF RSPCCHAIN WITH EMPTY KEY.
    DATA LS_PC TYPE RSPCCHAIN.
    LV_ODSO = LS_DSO-ODSOBJECT.
    SELECT * FROM RSPCCHAIN INTO TABLE LT_PC WHERE VARIANTE = LV_ODSO.
    IF LT_PC IS INITIAL.
      APPEND 'ProcessChain : (none found)' TO LT_OUT.
    ELSE.
      LOOP AT LT_PC INTO LS_PC.
        CONCATENATE 'ProcessChain : ' LS_PC-CHAIN_ID ' Type=' LS_PC-TYPE INTO LV_LINE.
        APPEND LV_LINE TO LT_OUT.
      ENDLOOP.
    ENDIF.
    FREE LT_PC.

    " 3) DTPs loading INTO this DSO
    DATA LT_DTP TYPE STANDARD TABLE OF RSBKDTP WITH EMPTY KEY.
    DATA LS_DTP TYPE RSBKDTP.
    SELECT * FROM RSBKDTP INTO TABLE LT_DTP
      WHERE TGT = LS_DSO-ODSOBJECT AND OBJVERS = 'A'.
    IF LT_DTP IS INITIAL.
      APPEND 'DTP (inbound) : (none found)' TO LT_OUT.
    ELSE.
      LOOP AT LT_DTP INTO LS_DTP.
        CONCATENATE 'DTP : ' LS_DTP-DTP ' Source=' LS_DTP-SRC INTO LV_LINE.
        APPEND LV_LINE TO LT_OUT.
      ENDLOOP.
    ENDIF.
    FREE LT_DTP.

    " 4) Last load timestamp
    CLEAR LV_TS.
    " TGT in RSBKREQUEST = DTP name; resolve via RSBKDTP
    DATA LT_DTPU TYPE STANDARD TABLE OF RSBKDTP-DTP WITH DEFAULT KEY.
    DATA LV_DTPU TYPE RSBKDTP-DTP.
    DATA LV_TSU  TYPE RSTIMESTMP.
    CLEAR LV_TS.
    SELECT DTP INTO TABLE LT_DTPU FROM RSBKDTP
      WHERE TGT = LS_DSO-ODSOBJECT AND OBJVERS = 'A'.
    LOOP AT LT_DTPU INTO LV_DTPU.
      CLEAR LV_TSU.
      SELECT MAX( TSTMP_H ) INTO LV_TSU FROM RSBKREQUEST
        WHERE DTP = LV_DTPU AND TSTATE NOT IN ( '4', '9' ).
      IF LV_TSU > LV_TS. LV_TS = LV_TSU. ENDIF.
    ENDLOOP.
    IF LV_TS IS NOT INITIAL.
      LV_ZONE = P_TZONE.
      IF LV_ZONE IS INITIAL. LV_ZONE = SY-ZONLO. ENDIF.
      CONVERT TIME STAMP LV_TS TIME ZONE LV_ZONE INTO DATE LV_DATE TIME LV_TIME.
      APPEND '' TO LT_OUT.
      CONCATENATE 'Last Loaded  : ' LV_DATE ' ' LV_TIME INTO LV_LINE.
    ELSE.
      LV_LINE = 'Last Loaded  : (no successful request found)'.
    ENDIF.
    APPEND LV_LINE TO LT_OUT.

    " 5) Downstream targets
    APPEND '' TO LT_OUT.
    APPEND '--- Downstream Targets ---' TO LT_OUT.
    DATA LT_DOWN TYPE STANDARD TABLE OF RSTRAN WITH EMPTY KEY.
    DATA LS_DOWN TYPE RSTRAN.
    SELECT * FROM RSTRAN INTO TABLE LT_DOWN
      WHERE SOURCENAME = LS_DSO-ODSOBJECT AND OBJVERS = 'A'.
    IF LT_DOWN IS INITIAL.
      APPEND '(no downstream transformations found)' TO LT_OUT.
    ELSE.
      LOOP AT LT_DOWN INTO LS_DOWN.
        CONCATENATE 'Target : ' LS_DOWN-TARGETNAME ' [' LS_DOWN-TARGETTYPE ']' INTO LV_LINE.
        APPEND LV_LINE TO LT_OUT.
      ENDLOOP.
    ENDIF.
    FREE LT_DOWN.

    LV_DSONAME = LS_DSO-ODSOBJECT.
    CONDENSE LV_DSONAME NO-GAPS.
    IF P_OUTDIR IS INITIAL.
      CONCATENATE GV_DUDS_DIR GV_CTX '_DUDS_' LV_DSONAME '.txt' INTO LV_FILE.
    ELSE.
      CONCATENATE GV_DUDS_DIR LV_DSONAME '.txt' INTO LV_FILE.
    ENDIF.
    IF P_OUTDIR IS NOT INITIAL.
      PERFORM WRITE_FILE_FRONTEND TABLES LT_OUT USING LV_FILE.
    ELSE.
      PERFORM WRITE_FILE_APPSERVER TABLES LT_OUT USING LV_FILE.
    ENDIF.
    FREE LT_OUT.
  ENDLOOP.

ENDFORM.

*======================================================================
*  INFOSET EXTRACTION
*  Tables: RSDCUBE, RSDCUBET, RSDCUBEMULTI, RSDCUBEIOBJ
*======================================================================
FORM EXTRACT_INFOSETS.

  DATA: LT_IS    TYPE STANDARD TABLE OF RSDCUBE WITH EMPTY KEY,
        LS_IS    TYPE RSDCUBE,
        LV_FILE  TYPE STRING,
        LV_LINE  TYPE STRING,
        LV_DESC  TYPE STRING,
        LV_PTYPE TYPE STRING,
        LV_POS   TYPE CHAR4,
        LV_ISNAME TYPE STRING.

  DATA LT_OUT TYPE STANDARD TABLE OF STRING WITH EMPTY KEY.
  DATA LS_FL  TYPE TY_FLOW.

  IF GV_IS_DIR IS INITIAL. RETURN. ENDIF.

  SELECT * FROM RSDCUBE INTO TABLE LT_IS WHERE OBJVERS = 'A' AND CUBETYPE = 'I'.

  LOOP AT LT_IS INTO LS_IS.
    " Only process InfoSets appearing in current flow
    READ TABLE GT_FLOW INTO LS_FL WITH KEY TARGET = LS_IS-INFOCUBE.
    IF SY-SUBRC <> 0.
      READ TABLE GT_FLOW INTO LS_FL WITH KEY SOURCE = LS_IS-INFOCUBE.
      IF SY-SUBRC <> 0. CONTINUE. ENDIF.
    ENDIF.

    REFRESH LT_OUT.
    CONCATENATE '=== InfoSet : ' LS_IS-INFOCUBE ' ===' INTO LV_LINE.
    APPEND LV_LINE TO LT_OUT.
    APPEND '' TO LT_OUT.

    CLEAR LV_DESC.
    SELECT SINGLE TXTLG INTO LV_DESC FROM RSDCUBET
      WHERE INFOCUBE = LS_IS-INFOCUBE AND OBJVERS = 'A' AND LANGU = SY-LANGU.
    CONCATENATE 'Description : ' LV_DESC INTO LV_LINE.
    APPEND LV_LINE TO LT_OUT.
    APPEND '' TO LT_OUT.

    " Member InfoProviders via RSDCUBEMULTI
    APPEND '--- InfoProviders ---' TO LT_OUT.
    DATA LT_ISMEM TYPE STANDARD TABLE OF RSDCUBEMULTI WITH EMPTY KEY.
    DATA LS_ISMEM TYPE RSDCUBEMULTI.
    SELECT * FROM RSDCUBEMULTI INTO TABLE LT_ISMEM
      WHERE INFOCUBE = LS_IS-INFOCUBE AND OBJVERS = 'A'.
    IF LT_ISMEM IS INITIAL.
      APPEND '(no member providers found)' TO LT_OUT.
    ELSE.
      LOOP AT LT_ISMEM INTO LS_ISMEM.
        PERFORM GET_PROVIDER_TYPE USING LS_ISMEM-PARTCUBE '' CHANGING LV_PTYPE.
        CONCATENATE 'Provider : ' LS_ISMEM-PARTCUBE ' [' LV_PTYPE ']' INTO LV_LINE.
        APPEND LV_LINE TO LT_OUT.
      ENDLOOP.
    ENDIF.
    FREE LT_ISMEM.
    APPEND '' TO LT_OUT.

    " Join conditions: stored internally, not in a transparent table
    APPEND '--- Join Conditions ---' TO LT_OUT.
    APPEND '(Use RSA1/InfoSet editor to view joins — not stored in transparent table)' TO LT_OUT.
    APPEND '' TO LT_OUT.

    " Field mapping via RSDCUBEIOBJ
    APPEND '--- Field Mapping (InfoObjects) ---' TO LT_OUT.
    DATA LT_FLD TYPE STANDARD TABLE OF RSDCUBEIOBJ WITH EMPTY KEY.
    DATA LS_FLD TYPE RSDCUBEIOBJ.
    SELECT * FROM RSDCUBEIOBJ INTO TABLE LT_FLD
      WHERE INFOCUBE = LS_IS-INFOCUBE AND OBJVERS = 'A'.
    IF LT_FLD IS INITIAL.
      APPEND '(no InfoObjects found)' TO LT_OUT.
    ELSE.
      APPEND 'Position;InfoObject' TO LT_OUT.
      LOOP AT LT_FLD INTO LS_FLD.
        LV_POS = LS_FLD-POSIT.
        CONCATENATE LV_POS ';' LS_FLD-IOBJNM INTO LV_LINE.
        APPEND LV_LINE TO LT_OUT.
      ENDLOOP.
    ENDIF.
    FREE LT_FLD.

    LV_ISNAME = LS_IS-INFOCUBE.
    CONDENSE LV_ISNAME NO-GAPS.
    IF P_OUTDIR IS INITIAL.
      CONCATENATE GV_IS_DIR GV_CTX '_IS_' LV_ISNAME '.txt' INTO LV_FILE.
    ELSE.
      CONCATENATE GV_IS_DIR LV_ISNAME '.txt' INTO LV_FILE.
    ENDIF.
    IF P_OUTDIR IS NOT INITIAL.
      PERFORM WRITE_FILE_FRONTEND TABLES LT_OUT USING LV_FILE.
    ELSE.
      PERFORM WRITE_FILE_APPSERVER TABLES LT_OUT USING LV_FILE.
    ENDIF.
    FREE LT_OUT.
  ENDLOOP.

ENDFORM.

*======================================================================
*  MULTIPROVIDER EXTRACTION
*  Tables: RSDCUBEMULTI, RSDCUBET, RSDCUBEIOBJ, RSDIOBJ
*======================================================================
FORM EXTRACT_MULTIPROVIDERS.

  DATA: LV_FILE   TYPE STRING,
        LV_LINE   TYPE STRING,
        LV_DESC   TYPE STRING,
        LV_NAME   TYPE CHAR30,
        LV_PTYPE  TYPE STRING,
        LV_POSIT  TYPE CHAR4,
        LV_MPNAME TYPE STRING,
        LV_SRC30  TYPE CHAR30.

  DATA LT_OUT  TYPE STANDARD TABLE OF STRING WITH EMPTY KEY.
  DATA LT_MP   TYPE STANDARD TABLE OF RSDCUBEMULTI WITH EMPTY KEY.
  DATA LS_MP   TYPE RSDCUBEMULTI.
  DATA LS_FL   TYPE TY_FLOW.

  IF GV_MP_DIR IS INITIAL. RETURN. ENDIF.

  " Collect MPs from current flow
  REFRESH GT_MP_NAMES.
  LOOP AT GT_FLOW INTO LS_FL.
    IF LS_FL-TARGET_TYPE = 'MULTIPROVIDER'.
      INSERT LS_FL-TARGET INTO TABLE GT_MP_NAMES.
    ENDIF.
    IF LS_FL-SOURCE_TYPE = 'MULTIPROVIDER'.
      LV_SRC30 = LS_FL-SOURCE.
      INSERT LV_SRC30 INTO TABLE GT_MP_NAMES.
    ENDIF.
  ENDLOOP.
  INSERT GV_CTX INTO TABLE GT_MP_NAMES.

  LOOP AT GT_MP_NAMES INTO LV_NAME.
    " Verify it is really an MP
    CLEAR LS_MP.
    SELECT SINGLE * FROM RSDCUBEMULTI INTO LS_MP
      WHERE INFOCUBE = LV_NAME AND OBJVERS = 'A'.
    IF SY-SUBRC <> 0. CONTINUE. ENDIF.

    REFRESH LT_OUT.
    REFRESH LT_MP.
    CONCATENATE '=== MultiProvider : ' LV_NAME ' ===' INTO LV_LINE.
    APPEND LV_LINE TO LT_OUT.
    APPEND '' TO LT_OUT.

    CLEAR LV_DESC.
    SELECT SINGLE TXTLG INTO LV_DESC FROM RSDCUBET
      WHERE INFOCUBE = LV_NAME AND OBJVERS = 'A' AND LANGU = SY-LANGU.
    CONCATENATE 'Description : ' LV_DESC INTO LV_LINE.
    APPEND LV_LINE TO LT_OUT.
    APPEND '' TO LT_OUT.

    " Member InfoProviders
    APPEND '--- Member InfoProviders ---' TO LT_OUT.
    SELECT * FROM RSDCUBEMULTI INTO TABLE LT_MP
      WHERE INFOCUBE = LV_NAME AND OBJVERS = 'A'.
    LOOP AT LT_MP INTO LS_MP.
      PERFORM GET_PROVIDER_TYPE USING LS_MP-PARTCUBE '' CHANGING LV_PTYPE.
      CONCATENATE 'Member : ' LS_MP-PARTCUBE ' [' LV_PTYPE ']' INTO LV_LINE.
      APPEND LV_LINE TO LT_OUT.
    ENDLOOP.
    APPEND '' TO LT_OUT.
    FREE LT_MP.

    " Communication Structure (InfoObjects in MP)
    APPEND '--- Communication Structure (InfoObjects) ---' TO LT_OUT.
    DATA LT_IMAP TYPE STANDARD TABLE OF RSDCUBEIOBJ WITH EMPTY KEY.
    DATA LS_IMAP TYPE RSDCUBEIOBJ.
    SELECT * FROM RSDCUBEIOBJ INTO TABLE LT_IMAP
      WHERE INFOCUBE = LV_NAME AND OBJVERS = 'A'.
    IF LT_IMAP IS INITIAL.
      APPEND '(no InfoObjects found)' TO LT_OUT.
    ELSE.
      APPEND 'Position;InfoObject;Type;Description' TO LT_OUT.
      LOOP AT LT_IMAP INTO LS_IMAP.
        LV_POSIT = LS_IMAP-POSIT.
        " Classify InfoObject type
        DATA LV_ICLASS2 TYPE STRING.
        DATA LS_IO2 TYPE RSDIOBJ.
        CLEAR: LV_ICLASS2, LS_IO2.
        SELECT SINGLE * FROM RSDIOBJ INTO LS_IO2
          WHERE IOBJNM = LS_IMAP-IOBJNM AND OBJVERS = 'A'.
        CASE LS_IO2-IOBJTP.
          WHEN 'CHA'.
            IF LS_IO2-ATRONLYFL = 'X'.
              LV_ICLASS2 = 'ATTRIBUTE'.
            ELSE.
              LV_ICLASS2 = 'CHARACTERISTIC'.
            ENDIF.
          WHEN 'KYF'. LV_ICLASS2 = 'KEY FIGURE'.
          WHEN 'UNI'. LV_ICLASS2 = 'UNIT'.
          WHEN 'TIM'. LV_ICLASS2 = 'TIME CHARACTERISTIC'.
          WHEN OTHERS.
            IF LS_IO2-IOBJTP IS INITIAL.
              LV_ICLASS2 = 'CHARACTERISTIC'.  " default for technical fields
            ELSE.
              LV_ICLASS2 = LS_IO2-IOBJTP.
            ENDIF.
        ENDCASE.
        " Get description
        DATA LV_IOBJ_DESC TYPE STRING.
        CLEAR LV_IOBJ_DESC.
        SELECT SINGLE TXTLG INTO LV_IOBJ_DESC FROM RSDIOBJT
          WHERE IOBJNM = LS_IMAP-IOBJNM AND OBJVERS = 'A' AND LANGU = SY-LANGU.
        CONCATENATE LV_POSIT ';' LS_IMAP-IOBJNM ';' LV_ICLASS2 ';' LV_IOBJ_DESC INTO LV_LINE.
        APPEND LV_LINE TO LT_OUT.
      ENDLOOP.
    ENDIF.
    FREE LT_IMAP.

    LV_MPNAME = LV_NAME.
    CONDENSE LV_MPNAME NO-GAPS.
    IF P_OUTDIR IS INITIAL.
      CONCATENATE GV_MP_DIR GV_CTX '_MP_' LV_MPNAME '.txt' INTO LV_FILE.
    ELSE.
      CONCATENATE GV_MP_DIR LV_MPNAME '.txt' INTO LV_FILE.
    ENDIF.
    IF P_OUTDIR IS NOT INITIAL.
      PERFORM WRITE_FILE_FRONTEND TABLES LT_OUT USING LV_FILE.
    ELSE.
      PERFORM WRITE_FILE_APPSERVER TABLES LT_OUT USING LV_FILE.
    ENDIF.
    FREE LT_OUT.
  ENDLOOP.

ENDFORM.

*======================================================================
*======================================================================
*  BEX QUERY EXTRACTION
*  Output structure per query:
*   /BEx_Query/<QueryName>/Detail.txt      — header + variables
*   /BEx_Query/<QueryName>/CKF.txt         — calculated key figures
*   /BEx_Query/<QueryName>/RKF.txt         — restricted key figures
*  Plus at root:
*   /BEx_Query/<Provider>_BEx_Query_List.csv — consolidated list
*  Tables: RSZCOMPIC, RSZCOMPDIR, RSZELTDIR, RSZELTTXT, RSZGLOBV
*  DEFTP values: 'CKF'=Calculated KF, 'RKF'=Restricted KF,
*                'KYF'=Key Figure, 'CHA'=Characteristic, 'FLT'=Filter
*======================================================================
FORM EXTRACT_BEX_QUERIES.

  DATA: LT_QNAMES  TYPE SORTED TABLE OF RSZCOMPIC-COMPUID WITH UNIQUE KEY TABLE_LINE,
        LV_QNAME   TYPE RSZCOMPIC-COMPUID,
        LV_FILE    TYPE STRING,
        LV_LINE    TYPE STRING,
        LV_COMPUID TYPE RSZCOMPDIR-COMPUID,
        LV_ICUBE   TYPE RSZCOMPIC-INFOCUBE,
        LV_QDESC   TYPE STRING,
        LV_ETXT    TYPE STRING,
        LV_VTYPE_C TYPE CHAR1,
        LV_QFNAME  TYPE STRING,
        LV_PROV    TYPE CHAR30.

  DATA LT_LIST   TYPE STANDARD TABLE OF STRING WITH EMPTY KEY.
  DATA LT_DETAIL TYPE STANDARD TABLE OF STRING WITH EMPTY KEY.
  DATA LT_CKF    TYPE STANDARD TABLE OF STRING WITH EMPTY KEY.
  DATA LT_RKF    TYPE STANDARD TABLE OF STRING WITH EMPTY KEY.
  DATA LS_FL     TYPE TY_FLOW.
  DATA LV_SRC    TYPE STRING.

  IF GV_BEX_DIR IS INITIAL. RETURN. ENDIF.

  " --- Mode 1: auto-detect queries from flow sources ---
  LOOP AT GT_FLOW INTO LS_FL.
    LV_SRC = LS_FL-SOURCE.
    CONDENSE LV_SRC NO-GAPS.
    IF LV_SRC IS INITIAL. CONTINUE. ENDIF.
    CLEAR LV_COMPUID.
    " Match source name to query COMPID, verify it is a query (DEFTP='REP')
    SELECT SINGLE d~compuid INTO LV_COMPUID
      FROM rszcompdir AS d
      JOIN rszeltdir  AS e ON e~eltuid = d~compuid AND e~objvers = d~objvers
      WHERE d~compid  = LV_SRC
        AND d~objvers = 'A'
        AND e~deftp   = 'REP'.
    IF SY-SUBRC = 0.
      INSERT LV_COMPUID INTO TABLE LT_QNAMES.
    ENDIF.
  ENDLOOP.

  " --- Mode 2: P_BEX=X — all queries on provider (DEFTP=REP only) ---
  IF P_BEX = 'X'.
    DATA LT_QPROV TYPE STANDARD TABLE OF RSZCOMPIC WITH EMPTY KEY.
    DATA LS_QPROV TYPE RSZCOMPIC.
    LOOP AT GT_MP_NAMES INTO LV_PROV.
      " Only query-level rows: join RSZCOMPIC with RSZELTDIR on DEFTP='REP'
      SELECT ic~compuid FROM rszcompic AS ic
        JOIN rszeltdir AS e ON e~eltuid = ic~compuid AND e~objvers = ic~objvers
        INTO LS_QPROV-COMPUID
        WHERE ic~infocube = LV_PROV
          AND ic~objvers  = 'A'
          AND e~deftp     = 'REP'.
        INSERT LS_QPROV-COMPUID INTO TABLE LT_QNAMES.
      ENDSELECT.
      FREE LT_QPROV.
    ENDLOOP.
    LV_PROV = GV_CTX.
    SELECT ic~compuid FROM rszcompic AS ic
      JOIN rszeltdir AS e ON e~eltuid = ic~compuid AND e~objvers = ic~objvers
      INTO LS_QPROV-COMPUID
      WHERE ic~infocube = LV_PROV
        AND ic~objvers  = 'A'
        AND e~deftp     = 'REP'.
      INSERT LS_QPROV-COMPUID INTO TABLE LT_QNAMES.
    ENDSELECT.
    FREE LT_QPROV.
  ENDIF.

  IF LT_QNAMES IS INITIAL. RETURN. ENDIF.

  APPEND 'QueryTechName;Description;InfoProvider' TO LT_LIST.

  " --- One file per query in BEx_Query folder ---
  LOOP AT LT_QNAMES INTO LV_QNAME.
    REFRESH: LT_DETAIL, LT_CKF, LT_RKF.

    DATA LS_QCOMP TYPE RSZCOMPDIR.
    CLEAR LS_QCOMP.
    SELECT SINGLE * FROM RSZCOMPDIR INTO LS_QCOMP
      WHERE COMPUID = LV_QNAME AND OBJVERS = 'A'.

    CLEAR LV_QDESC.
    SELECT SINGLE TXTLG INTO LV_QDESC FROM RSZELTTXT
      WHERE ELTUID = LV_QNAME AND LANGU = SY-LANGU.

    CLEAR LV_ICUBE.
    SELECT SINGLE INFOCUBE INTO LV_ICUBE FROM RSZCOMPIC
      WHERE COMPUID = LV_QNAME AND OBJVERS = 'A'.

    " Add to consolidated list
    CONCATENATE LS_QCOMP-COMPID ';' LV_QDESC ';' LV_ICUBE INTO LV_LINE.
    APPEND LV_LINE TO LT_LIST.

    " === Single file: header + variables + CKF + RKF ===
    CONCATENATE '=== BEx Query : ' LS_QCOMP-COMPID ' ===' INTO LV_LINE.
    APPEND LV_LINE TO LT_DETAIL.
    APPEND '' TO LT_DETAIL.
    CONCATENATE 'Description  : ' LV_QDESC INTO LV_LINE.
    APPEND LV_LINE TO LT_DETAIL.
    CONCATENATE 'InfoProvider : ' LV_ICUBE INTO LV_LINE.
    APPEND LV_LINE TO LT_DETAIL.
    APPEND '' TO LT_DETAIL.

    " Variables
    APPEND '--- Variables ---' TO LT_DETAIL.
    DATA LT_QVAR TYPE STANDARD TABLE OF RSZGLOBV WITH EMPTY KEY.
    DATA LS_QVAR TYPE RSZGLOBV.
    SELECT * FROM RSZGLOBV INTO TABLE LT_QVAR
      WHERE QUERYNM = LV_QNAME AND OBJVERS = 'A'.
    IF LT_QVAR IS INITIAL.
      APPEND '(no variables)' TO LT_DETAIL.
    ELSE.
      APPEND 'Variable;CharIO;VarType;DefaultLow' TO LT_DETAIL.
      LOOP AT LT_QVAR INTO LS_QVAR.
        LV_VTYPE_C = LS_QVAR-VARTYP.
        CONCATENATE LS_QVAR-VNAM ';' LS_QVAR-IOBJNM ';'
                    LV_VTYPE_C ';' LS_QVAR-LOW INTO LV_LINE.
        APPEND LV_LINE TO LT_DETAIL.
      ENDLOOP.
    ENDIF.
    FREE LT_QVAR.
    APPEND '' TO LT_DETAIL.

    " Elements — split into CKF and RKF sections
    DATA LT_QELS TYPE STANDARD TABLE OF RSZELTDIR WITH EMPTY KEY.
    DATA LS_QEL  TYPE RSZELTDIR.
    SELECT * FROM RSZELTDIR INTO TABLE LT_QELS
      WHERE SOURCE_ELTUID = LV_QNAME AND OBJVERS = 'A'.
    IF LT_QELS IS INITIAL.
      SELECT * FROM RSZELTDIR INTO TABLE LT_QELS
        WHERE MAPNAME = LS_QCOMP-COMPID AND OBJVERS = 'A'.
    ENDIF.

    LOOP AT LT_QELS INTO LS_QEL.
      CLEAR LV_ETXT.
      SELECT SINGLE TXTLG INTO LV_ETXT FROM RSZELTTXT
        WHERE ELTUID = LS_QEL-ELTUID AND LANGU = SY-LANGU.
      CONCATENATE LS_QEL-ELTUID ';' LV_ETXT INTO LV_LINE.
      IF LS_QEL-DEFTP = 'CKF'.
        APPEND LV_LINE TO LT_CKF.
      ELSEIF LS_QEL-DEFTP = 'RKF'.
        APPEND LV_LINE TO LT_RKF.
      ENDIF.
    ENDLOOP.
    FREE LT_QELS.

    " Append CKF section
    APPEND '--- Calculated Key Figures (CKF) ---' TO LT_DETAIL.
    IF LT_CKF IS INITIAL.
      APPEND '(none)' TO LT_DETAIL.
    ELSE.
      APPEND 'ElementUID;Description' TO LT_DETAIL.
      APPEND LINES OF LT_CKF TO LT_DETAIL.
    ENDIF.
    APPEND '' TO LT_DETAIL.

    " Append RKF section
    APPEND '--- Restricted Key Figures (RKF) ---' TO LT_DETAIL.
    IF LT_RKF IS INITIAL.
      APPEND '(none)' TO LT_DETAIL.
    ELSE.
      APPEND 'ElementUID;Description' TO LT_DETAIL.
      APPEND LINES OF LT_RKF TO LT_DETAIL.
    ENDIF.

    " Write single file: /BEx_Query/<QueryName>.txt
    LV_QFNAME = LS_QCOMP-COMPID.
    IF LV_QFNAME IS INITIAL. LV_QFNAME = LV_QNAME. ENDIF.
    CONDENSE LV_QFNAME NO-GAPS.
    REPLACE ALL OCCURRENCES OF '/' IN LV_QFNAME WITH '_'.
    IF P_OUTDIR IS INITIAL.
      CONCATENATE GV_BEX_DIR GV_CTX '_BEX_' LV_QFNAME '.txt' INTO LV_FILE.
    ELSE.
      CONCATENATE GV_BEX_DIR LV_QFNAME '.txt' INTO LV_FILE.
    ENDIF.
    IF P_OUTDIR IS NOT INITIAL.
      PERFORM WRITE_FILE_FRONTEND TABLES LT_DETAIL USING LV_FILE.
    ELSE.
      PERFORM WRITE_FILE_APPSERVER TABLES LT_DETAIL USING LV_FILE.
    ENDIF.
    FREE LT_DETAIL.

  ENDLOOP.

  " --- Write consolidated list CSV at BEx root ---
  DATA LV_BASE TYPE CHAR30.
  LV_BASE = GV_CTX.
  PERFORM SANITIZE_FOR_FILE USING LV_BASE CHANGING LV_BASE.
  CONCATENATE GV_BEX_DIR LV_BASE '_BEx_Query_List.csv' INTO LV_FILE.
  IF P_OUTDIR IS NOT INITIAL.
    PERFORM WRITE_FILE_FRONTEND TABLES LT_LIST USING LV_FILE.
  ELSE.
    PERFORM WRITE_FILE_APPSERVER TABLES LT_LIST USING LV_FILE.
  ENDIF.

ENDFORM.
